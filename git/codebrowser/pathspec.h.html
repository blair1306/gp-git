4/* Pathspec magic */
5#define PATHSPEC_FROMTOP (1<<0)
6#define PATHSPEC_MAXDEPTH (1<<1)
7#define PATHSPEC_LITERAL (1<<2)
8#define PATHSPEC_GLOB (1<<3)
9#define PATHSPEC_ICASE (1<<4)
10#define PATHSPEC_EXCLUDE (1<<5)
11#define PATHSPEC_ATTR (1<<6)

14 PATHSPEC_MAXDEPTH | \

81static inline int ps_strncmp(const struct pathspec_item *item,
82 const char *s1, const char *s2, size_t n)
83{
84 if (item->magic & PATHSPEC_ICASE)
85 return strncasecmp(s1, s2, n);
86 else
87 return strncmp(s1, s2, n);
88}
89
90static inline int ps_strcmp(const struct pathspec_item *item,
91 const char *s1, const char *s2)
92{
93 if (item->magic & PATHSPEC_ICASE)
94 return strcasecmp(s1, s2);
95 else
96 return strcmp(s1, s2);
97}
