87static int parse_msg_id(const char *text)
88{
89 int i;
90
91 if (!msg_id_info[0].downcased) {
92 /* convert id_string to lower case, without underscores. */
93 for (i = 0; i < FSCK_MSG_MAX; i++) {
94 const char *p = msg_id_info[i].id_string;
95 int len = strlen(p);
96 char *q = xmalloc(len);
97
98 msg_id_info[i].downcased = q;
99 while (*p)
100 if (*p == '_')
101 p++;
102 else
103 *(q)++ = tolower(*(p)++);
104 *q = '\0';
105 }
106 }
107
108 for (i = 0; i < FSCK_MSG_MAX; i++)
109 if (!strcmp(text, msg_id_info[i].downcased))
110 return i;
111
112 return -1;
113}

115static int fsck_msg_type(enum fsck_msg_id msg_id,
116 struct fsck_options *options)
117{
118 int msg_type;
119
120 assert(msg_id >= 0 && msg_id < FSCK_MSG_MAX);
121
122 if (options->msg_type)
123 msg_type = options->msg_type[msg_id];
124 else {
125 msg_type = msg_id_info[msg_id].msg_type;
126 if (options->strict && msg_type == FSCK_WARN)
127 msg_type = FSCK_ERROR;
128 }
129
130 return msg_type;

133static void init_skiplist(struct fsck_options *options, const char *path)
134{
135 static struct oid_array skiplist = OID_ARRAY_INIT;
136 int sorted, fd;
137 char buffer[GIT_MAX_HEXSZ + 1];
138 struct object_id oid;
139
140 if (options->skiplist)
141 sorted = options->skiplist->sorted;
142 else {
143 sorted = 1;
144 options->skiplist = &skiplist;
145 }
146
147 fd = open(path, O_RDONLY);
148 if (fd < 0)
149 die("Could not open skip list: %s", path);
150 for (;;) {
151 const char *p;
152 int result = read_in_full(fd, buffer, sizeof(buffer));
153 if (result < 0)
154 die_errno("Could not read '%s'", path);
155 if (!result)
156 break;
157 if (parse_oid_hex(buffer, &oid, &p) || *p != '\n')
158 die("Invalid SHA-1: %s", buffer);
159 oid_array_append(&skiplist, &oid);
160 if (sorted && skiplist.nr > 1 &&
161 oidcmp(&skiplist.oid[skiplist.nr - 2],
162 &oid) > 0)
163 sorted = 0;
164 }
165 close(fd);
166
167 if (sorted)
168 skiplist.sorted = 1;
169}

191void fsck_set_msg_type(struct fsck_options *options,
192 const char *msg_id, const char *msg_type)
193{
194 int id = parse_msg_id(msg_id), type;
195
196 if (id < 0)
197 die("Unhandled message id: %s", msg_id);
198 type = parse_msg_type(msg_type);
199
200 if (type != FSCK_ERROR && msg_id_info[id].msg_type == FSCK_FATAL)
201 die("Cannot demote %s to %s", msg_id, msg_type);
202
203 if (!options->msg_type) {
204 int i;
205 int *msg_type;
206 ALLOC_ARRAY(msg_type, FSCK_MSG_MAX);
207 for (i = 0; i < FSCK_MSG_MAX; i++)
208 msg_type[i] = fsck_msg_type(i, options);
209 options->msg_type = msg_type;
210 }
211
212 options->msg_type[id] = type;
213}

215void fsck_set_msg_types(struct fsck_options *options, const char *values)
216{
217 char *buf = xstrdup(values), *to_free = buf;
218 int done = 0;
219
220 while (!done) {
221 int len = strcspn(buf, " ,|"), equal;
222
223 done = !buf[len];
224 if (!len) {
225 buf++;
226 continue;
227 }
228 buf[len] = '\0';
229
230 for (equal = 0;
231 equal < len && buf[equal] != '=' && buf[equal] != ':';
232 equal++)
233 buf[equal] = tolower(buf[equal]);
234 buf[equal] = '\0';
235
236 if (!strcmp(buf, "skiplist")) {
237 if (equal == len)
238 die("skiplist requires a path");
239 init_skiplist(options, buf + equal + 1);
240 buf += len + 1;
241 continue;
242 }
243
244 if (equal == len)
245 die("Missing '=': '%s'", buf);
246
247 fsck_set_msg_type(options, buf, buf + equal + 1);
248 buf += len + 1;
249 }
250 free(to_free);
251}

253static void append_msg_id(struct strbuf *sb, const char *msg_id)
254{
255 for (;;) {
256 char c = *(msg_id)++;
257
258 if (!c)
259 break;
260 if (c != '_')
261 strbuf_addch(sb, tolower(c));
262 else {
263 assert(*msg_id);
264 strbuf_addch(sb, *(msg_id)++);
265 }
266 }
267
268 strbuf_addstr(sb, ": ");
269}

271__attribute__((format (printf, 4, 5)))
272static int report(struct fsck_options *options, struct object *object,
273 enum fsck_msg_id id, const char *fmt, ...)
274{
275 va_list ap;
276 struct strbuf sb = STRBUF_INIT;
277 int msg_type = fsck_msg_type(id, options), result;
278
279 if (msg_type == FSCK_IGNORE)
280 return 0;
281
282 if (options->skiplist && object &&
283 oid_array_lookup(options->skiplist, &object->oid) >= 0)
284 return 0;
285
286 if (msg_type == FSCK_FATAL)
287 msg_type = FSCK_ERROR;
288 else if (msg_type == FSCK_INFO)
289 msg_type = FSCK_WARN;
290
291 append_msg_id(&sb, msg_id_info[id].id_string);
292
293 va_start(ap, fmt);
294 strbuf_vaddf(&sb, fmt, ap);
295 result = options->error_func(options, object, msg_type, sb.buf);
296 strbuf_release(&sb);
297 va_end(ap);
298
299 return result;
300}

340static int fsck_walk_tree(struct tree *tree, void *data, struct fsck_options *options)
341{
342 struct tree_desc desc;
343 struct name_entry entry;
344 int res = 0;
345 const char *name;
346
347 if (parse_tree(tree))
348 return -1;
349
350 name = get_object_name(options, &tree->object);
351 if (init_tree_desc_gently(&desc, tree->buffer, tree->size))
352 return -1;
353 while (tree_entry_gently(&desc, &entry)) {
354 struct object *obj;
355 int result;
356
357 if (S_ISGITLINK(entry.mode))
358 continue;
359
360 if (S_ISDIR(entry.mode)) {
361 obj = &lookup_tree(entry.oid)->object;
362 if (name)
363 put_object_name(options, obj, "%s%s/", name,
364 entry.path);
365 result = options->walk(obj, OBJ_TREE, data, options);
366 }
367 else if (S_ISREG(entry.mode) || S_ISLNK(entry.mode)) {
368 obj = &lookup_blob(entry.oid)->object;
369 if (name)
370 put_object_name(options, obj, "%s%s", name,
371 entry.path);
372 result = options->walk(obj, OBJ_BLOB, data, options);
373 }
374 else {
375 result = error("in tree %s: entry %s has bad mode %.6o",
376 describe_object(options, &tree->object), entry.path, entry.mode);
377 }
378 if (result < 0)
379 return result;
380 if (!res)
381 res = result;
382 }
383 return res;
384}

386static int fsck_walk_commit(struct commit *commit, void *data, struct fsck_options *options)
387{
388 int counter = 0, generation = 0, name_prefix_len = 0;
389 struct commit_list *parents;
390 int res;
391 int result;
392 const char *name;
393
394 if (parse_commit(commit))
395 return -1;
396
397 name = get_object_name(options, &commit->object);
398 if (name)
399 put_object_name(options, &commit->tree->object, "%s:", name);
400
401 result = options->walk((struct object *)commit->tree, OBJ_TREE, data, options);
402 if (result < 0)
403 return result;
404 res = result;
405
406 parents = commit->parents;
407 if (name && parents) {
408 int len = strlen(name), power;
409
410 if (len && name[len - 1] == '^') {
411 generation = 1;
412 name_prefix_len = len - 1;
413 }
414 else { /* parse ~<generation> suffix */
415 for (generation = 0, power = 1;
416 len && isdigit(name[len - 1]);
417 power *= 10)
418 generation += power * (name[--len] - '0');
419 if (power > 1 && len && name[len - 1] == '~')
420 name_prefix_len = len - 1;
421 }
422 }
423
424 while (parents) {
425 if (name) {
426 struct object *obj = &parents->item->object;
427
428 if (++counter > 1)
429 put_object_name(options, obj, "%s^%d",
430 name, counter);
431 else if (generation > 0)
432 put_object_name(options, obj, "%.*s~%d",
433 name_prefix_len, name, generation + 1);
434 else
435 put_object_name(options, obj, "%s^", name);
436 }
437 result = options->walk((struct object *)parents->item, OBJ_COMMIT, data, options);
438 if (result < 0)
439 return result;
440 if (!res)
441 res = result;
442 parents = parents->next;
443 }
444 return res;
445}

458int fsck_walk(struct object *obj, void *data, struct fsck_options *options)
459{
460 if (!obj)
461 return -1;
462
463 if (obj->type == OBJ_NONE)
464 parse_object(&obj->oid);
465
466 switch (obj->type) {
467 case OBJ_BLOB:
468 return 0;
469 case OBJ_TREE:
470 return fsck_walk_tree((struct tree *)obj, data, options);
471 case OBJ_COMMIT:
472 return fsck_walk_commit((struct commit *)obj, data, options);
473 case OBJ_TAG:
474 return fsck_walk_tag((struct tag *)obj, data, options);
475 default:
476 error("Unknown object type for %s", describe_object(options, obj));
477 return -1;
478 }
479}

481/*
482 * The entries in a tree are ordered in the _path_ order,
483 * which means that a directory entry is ordered by adding
484 * a slash to the end of it.
485 *
486 * So a directory called "a" is ordered _after_ a file
487 * called "a.c", because "a/" sorts after "a.c".
488 */
489#define TREE_UNORDERED (-1)
490#define TREE_HAS_DUPS (-2)
491
492static int verify_ordered(unsigned mode1, const char *name1, unsigned mode2, const char *name2)
493{
494 int len1 = strlen(name1);
495 int len2 = strlen(name2);
496 int len = len1 < len2 ? len1 : len2;
497 unsigned char c1, c2;
498 int cmp;
499
500 cmp = memcmp(name1, name2, len);
501 if (cmp < 0)
502 return 0;
503 if (cmp > 0)
504 return TREE_UNORDERED;
505
506 /*
507 * Ok, the first <len> characters are the same.
508 * Now we need to order the next one, but turn
509 * a '\0' into a '/' for a directory entry.
510 */
511 c1 = name1[len];
512 c2 = name2[len];
513 if (!c1 && !c2)
514 /*
515 * git-write-tree used to write out a nonsense tree that has
516 * entries with the same name, one blob and one tree. Make
517 * sure we do not have duplicate entries.
518 */
519 return TREE_HAS_DUPS;
520 if (!c1 && S_ISDIR(mode1))
521 c1 = '/';
522 if (!c2 && S_ISDIR(mode2))
523 c2 = '/';
524 return c1 < c2 ? 0 : TREE_UNORDERED;
525}

527static int fsck_tree(struct tree *item, struct fsck_options *options)
528{
529 int retval = 0;
530 int has_null_sha1 = 0;
531 int has_full_path = 0;
532 int has_empty_name = 0;
533 int has_dot = 0;
534 int has_dotdot = 0;
535 int has_dotgit = 0;
536 int has_zero_pad = 0;
537 int has_bad_modes = 0;
538 int has_dup_entries = 0;
539 int not_properly_sorted = 0;
540 struct tree_desc desc;
541 unsigned o_mode;
542 const char *o_name;
543
544 if (init_tree_desc_gently(&desc, item->buffer, item->size)) {
545 retval += report(options, &item->object, FSCK_MSG_BAD_TREE, "cannot be parsed as a tree");
546 return retval;
547 }
548
549 o_mode = 0;
550 o_name = NULL;
551
552 while (desc.size) {
553 unsigned mode;
554 const char *name;
555 const struct object_id *oid;
556
557 oid = tree_entry_extract(&desc, &name, &mode);
558
559 has_null_sha1 |= is_null_oid(oid);
560 has_full_path |= !!strchr(name, '/');
561 has_empty_name |= !*name;
562 has_dot |= !strcmp(name, ".");
563 has_dotdot |= !strcmp(name, "..");
564 has_dotgit |= (!strcmp(name, ".git") ||
565 is_hfs_dotgit(name) ||
566 is_ntfs_dotgit(name));
567 has_zero_pad |= *(char *)desc.buffer == '0';
568 if (update_tree_entry_gently(&desc)) {
569 retval += report(options, &item->object, FSCK_MSG_BAD_TREE, "cannot be parsed as a tree");
570 break;
571 }
572
573 switch (mode) {
574 /*
575 * Standard modes..
576 */
577 case S_IFREG | 0755:
578 case S_IFREG | 0644:
579 case S_IFLNK:
580 case S_IFDIR:
581 case S_IFGITLINK:
582 break;
583 /*
584 * This is nonstandard, but we had a few of these
585 * early on when we honored the full set of mode
586 * bits..
587 */
588 case S_IFREG | 0664:
589 if (!options->strict)
590 break;
591 default:
592 has_bad_modes = 1;
593 }
594
595 if (o_name) {
596 switch (verify_ordered(o_mode, o_name, mode, name)) {
597 case TREE_UNORDERED:
598 not_properly_sorted = 1;
599 break;
600 case TREE_HAS_DUPS:
601 has_dup_entries = 1;
602 break;
603 default:
604 break;
605 }
606 }
607
608 o_mode = mode;
609 o_name = name;
610 }
611
612 if (has_null_sha1)
613 retval += report(options, &item->object, FSCK_MSG_NULL_SHA1, "contains entries pointing to null sha1");
614 if (has_full_path)
615 retval += report(options, &item->object, FSCK_MSG_FULL_PATHNAME, "contains full pathnames");
616 if (has_empty_name)
617 retval += report(options, &item->object, FSCK_MSG_EMPTY_NAME, "contains empty pathname");
618 if (has_dot)
619 retval += report(options, &item->object, FSCK_MSG_HAS_DOT, "contains '.'");
620 if (has_dotdot)
621 retval += report(options, &item->object, FSCK_MSG_HAS_DOTDOT, "contains '..'");
622 if (has_dotgit)
623 retval += report(options, &item->object, FSCK_MSG_HAS_DOTGIT, "contains '.git'");
624 if (has_zero_pad)
625 retval += report(options, &item->object, FSCK_MSG_ZERO_PADDED_FILEMODE, "contains zero-padded file modes");
626 if (has_bad_modes)
627 retval += report(options, &item->object, FSCK_MSG_BAD_FILEMODE, "contains bad file modes");
628 if (has_dup_entries)
629 retval += report(options, &item->object, FSCK_MSG_DUPLICATE_ENTRIES, "contains duplicate file entries");
630 if (not_properly_sorted)
631 retval += report(options, &item->object, FSCK_MSG_TREE_NOT_SORTED, "not properly sorted");
632 return retval;
633}

635static int verify_headers(const void *data, unsigned long size,
636 struct object *obj, struct fsck_options *options)
637{
638 const char *buffer = (const char *)data;
639 unsigned long i;
640
641 for (i = 0; i < size; i++) {
642 switch (buffer[i]) {
643 case '\0':
644 return report(options, obj,
645 FSCK_MSG_NUL_IN_HEADER,
646 "unterminated header: NUL at offset %ld", i);
647 case '\n':
648 if (i + 1 < size && buffer[i + 1] == '\n')
649 return 0;
650 }
651 }
652
653 /*
654 * We did not find double-LF that separates the header
655 * and the body. Not having a body is not a crime but
656 * we do want to see the terminating LF for the last header
657 * line.
658 */
659 if (size && buffer[size - 1] == '\n')
660 return 0;
661
662 return report(options, obj,
663 FSCK_MSG_UNTERMINATED_HEADER, "unterminated header");
664}

666static int fsck_ident(const char **ident, struct object *obj, struct fsck_options *options)
667{
668 const char *p = *ident;
669 char *end;
670
671 *ident = strchrnul(*ident, '\n');
672 if (**ident == '\n')
673 (*ident)++;
674
675 if (*p == '<')
676 return report(options, obj, FSCK_MSG_MISSING_NAME_BEFORE_EMAIL, "invalid author/committer line - missing space before email");
677 p += strcspn(p, "<>\n");
678 if (*p == '>')
679 return report(options, obj, FSCK_MSG_BAD_NAME, "invalid author/committer line - bad name");
680 if (*p != '<')
681 return report(options, obj, FSCK_MSG_MISSING_EMAIL, "invalid author/committer line - missing email");
682 if (p[-1] != ' ')
683 return report(options, obj, FSCK_MSG_MISSING_SPACE_BEFORE_EMAIL, "invalid author/committer line - missing space before email");
684 p++;
685 p += strcspn(p, "<>\n");
686 if (*p != '>')
687 return report(options, obj, FSCK_MSG_BAD_EMAIL, "invalid author/committer line - bad email");
688 p++;
689 if (*p != ' ')
690 return report(options, obj, FSCK_MSG_MISSING_SPACE_BEFORE_DATE, "invalid author/committer line - missing space before date");
691 p++;
692 if (*p == '0' && p[1] != ' ')
693 return report(options, obj, FSCK_MSG_ZERO_PADDED_DATE, "invalid author/committer line - zero-padded date");
694 if (date_overflows(parse_timestamp(p, &end, 10)))
695 return report(options, obj, FSCK_MSG_BAD_DATE_OVERFLOW, "invalid author/committer line - date causes integer overflow");
696 if ((end == p || *end != ' '))
697 return report(options, obj, FSCK_MSG_BAD_DATE, "invalid author/committer line - bad date");
698 p = end + 1;
699 if ((*p != '+' && *p != '-') ||
700 !isdigit(p[1]) ||
701 !isdigit(p[2]) ||
702 !isdigit(p[3]) ||
703 !isdigit(p[4]) ||
704 (p[5] != '\n'))
705 return report(options, obj, FSCK_MSG_BAD_TIMEZONE, "invalid author/committer line - bad time zone");
706 p += 6;
707 return 0;
708}

710static int fsck_commit_buffer(struct commit *commit, const char *buffer,
711 unsigned long size, struct fsck_options *options)
712{
713 unsigned char tree_sha1[20], sha1[20];
714 struct commit_graft *graft;
715 unsigned parent_count, parent_line_count = 0, author_count;
716 int err;
717 const char *buffer_begin = buffer;
718
719 if (verify_headers(buffer, size, &commit->object, options))
720 return -1;
721
722 if (!skip_prefix(buffer, "tree ", &buffer))
723 return report(options, &commit->object, FSCK_MSG_MISSING_TREE, "invalid format - expected 'tree' line");
724 if (get_sha1_hex(buffer, tree_sha1) || buffer[40] != '\n') {
725 err = report(options, &commit->object, FSCK_MSG_BAD_TREE_SHA1, "invalid 'tree' line format - bad sha1");
726 if (err)
727 return err;
728 }
729 buffer += 41;
730 while (skip_prefix(buffer, "parent ", &buffer)) {
731 if (get_sha1_hex(buffer, sha1) || buffer[40] != '\n') {
732 err = report(options, &commit->object, FSCK_MSG_BAD_PARENT_SHA1, "invalid 'parent' line format - bad sha1");
733 if (err)
734 return err;
735 }
736 buffer += 41;
737 parent_line_count++;
738 }
739 graft = lookup_commit_graft(&commit->object.oid);
740 parent_count = commit_list_count(commit->parents);
741 if (graft) {
742 if (graft->nr_parent == -1 && !parent_count)
743 ; /* shallow commit */
744 else if (graft->nr_parent != parent_count) {
745 err = report(options, &commit->object, FSCK_MSG_MISSING_GRAFT, "graft objects missing");
746 if (err)
747 return err;
748 }
749 } else {
750 if (parent_count != parent_line_count) {
751 err = report(options, &commit->object, FSCK_MSG_MISSING_PARENT, "parent objects missing");
752 if (err)
753 return err;
754 }
755 }
756 author_count = 0;
757 while (skip_prefix(buffer, "author ", &buffer)) {
758 author_count++;
759 err = fsck_ident(&buffer, &commit->object, options);
760 if (err)
761 return err;
762 }
763 if (author_count < 1)
764 err = report(options, &commit->object, FSCK_MSG_MISSING_AUTHOR, "invalid format - expected 'author' line");
765 else if (author_count > 1)
766 err = report(options, &commit->object, FSCK_MSG_MULTIPLE_AUTHORS, "invalid format - multiple 'author' lines");
767 if (err)
768 return err;
769 if (!skip_prefix(buffer, "committer ", &buffer))
770 return report(options, &commit->object, FSCK_MSG_MISSING_COMMITTER, "invalid format - expected 'committer' line");
771 err = fsck_ident(&buffer, &commit->object, options);
772 if (err)
773 return err;
774 if (!commit->tree) {
775 err = report(options, &commit->object, FSCK_MSG_BAD_TREE, "could not load commit's tree %s", sha1_to_hex(tree_sha1));
776 if (err)
777 return err;
778 }
779 if (memchr(buffer_begin, '\0', size)) {
780 err = report(options, &commit->object, FSCK_MSG_NUL_IN_COMMIT,
781 "NUL byte in the commit object body");
782 if (err)
783 return err;
784 }
785 return 0;
786}

789 unsigned long size, struct fsck_options *options)

791 const char *buffer = data ? data : get_commit_buffer(commit, &size);
792 int ret = fsck_commit_buffer(commit, buffer, size, options);
793 if (!data)

795 return ret;

798static int fsck_tag_buffer(struct tag *tag, const char *data,
799 unsigned long size, struct fsck_options *options)
800{
801 unsigned char sha1[20];
802 int ret = 0;
803 const char *buffer;
804 char *to_free = NULL, *eol;
805 struct strbuf sb = STRBUF_INIT;
806
807 if (data)
808 buffer = data;
809 else {
810 enum object_type type;
811
812 buffer = to_free =
813 read_sha1_file(tag->object.oid.hash, &type, &size);
814 if (!buffer)
815 return report(options, &tag->object,
816 FSCK_MSG_MISSING_TAG_OBJECT,
817 "cannot read tag object");
818
819 if (type != OBJ_TAG) {
820 ret = report(options, &tag->object,
821 FSCK_MSG_TAG_OBJECT_NOT_TAG,
822 "expected tag got %s",
823 typename(type));
824 goto done;
825 }
826 }
827
828 ret = verify_headers(buffer, size, &tag->object, options);
829 if (ret)
830 goto done;
831
832 if (!skip_prefix(buffer, "object ", &buffer)) {
833 ret = report(options, &tag->object, FSCK_MSG_MISSING_OBJECT, "invalid format - expected 'object' line");
834 goto done;
835 }
836 if (get_sha1_hex(buffer, sha1) || buffer[40] != '\n') {
837 ret = report(options, &tag->object, FSCK_MSG_BAD_OBJECT_SHA1, "invalid 'object' line format - bad sha1");
838 if (ret)
839 goto done;
840 }
841 buffer += 41;
842
843 if (!skip_prefix(buffer, "type ", &buffer)) {
844 ret = report(options, &tag->object, FSCK_MSG_MISSING_TYPE_ENTRY, "invalid format - expected 'type' line");
845 goto done;
846 }
847 eol = strchr(buffer, '\n');
848 if (!eol) {
849 ret = report(options, &tag->object, FSCK_MSG_MISSING_TYPE, "invalid format - unexpected end after 'type' line");
850 goto done;
851 }
852 if (type_from_string_gently(buffer, eol - buffer, 1) < 0)
853 ret = report(options, &tag->object, FSCK_MSG_BAD_TYPE, "invalid 'type' value");
854 if (ret)
855 goto done;
856 buffer = eol + 1;
857
858 if (!skip_prefix(buffer, "tag ", &buffer)) {
859 ret = report(options, &tag->object, FSCK_MSG_MISSING_TAG_ENTRY, "invalid format - expected 'tag' line");
860 goto done;
861 }
862 eol = strchr(buffer, '\n');
863 if (!eol) {
864 ret = report(options, &tag->object, FSCK_MSG_MISSING_TAG, "invalid format - unexpected end after 'type' line");
865 goto done;
866 }
867 strbuf_addf(&sb, "refs/tags/%.*s", (int)(eol - buffer), buffer);
868 if (check_refname_format(sb.buf, 0)) {
869 ret = report(options, &tag->object, FSCK_MSG_BAD_TAG_NAME,
870 "invalid 'tag' name: %.*s",
871 (int)(eol - buffer), buffer);
872 if (ret)
873 goto done;
874 }
875 buffer = eol + 1;
876
877 if (!skip_prefix(buffer, "tagger ", &buffer)) {
878 /* early tags do not contain 'tagger' lines; warn only */
879 ret = report(options, &tag->object, FSCK_MSG_MISSING_TAGGER_ENTRY, "invalid format - expected 'tagger' line");
880 if (ret)
881 goto done;
882 }
883 else
884 ret = fsck_ident(&buffer, &tag->object, options);
885
886done:
887 strbuf_release(&sb);
888 free(to_free);
889 return ret;
890}

892static int fsck_tag(struct tag *tag, const char *data,
893 unsigned long size, struct fsck_options *options)
894{
895 struct object *tagged = tag->tagged;
896
897 if (!tagged)
898 return report(options, &tag->object, FSCK_MSG_BAD_TAG_OBJECT, "could not load tagged object");
899
900 return fsck_tag_buffer(tag, data, size, options);
901}
902
903int fsck_object(struct object *obj, void *data, unsigned long size,
904 struct fsck_options *options)
905{
906 if (!obj)
907 return report(options, obj, FSCK_MSG_BAD_OBJECT_SHA1, "no valid object to fsck");
908
909 if (obj->type == OBJ_BLOB)
910 return 0;
911 if (obj->type == OBJ_TREE)
912 return fsck_tree((struct tree *) obj, options);
913 if (obj->type == OBJ_COMMIT)
914 return fsck_commit((struct commit *) obj, (const char *) data,
915 size, options);
916 if (obj->type == OBJ_TAG)
917 return fsck_tag((struct tag *) obj, (const char *) data,
918 size, options);
919
920 return report(options, obj, FSCK_MSG_UNKNOWN_TYPE, "unknown type '%d' (internal fsck error)",
921 obj->type);
922}

924int fsck_error_function(struct fsck_options *o,
925 struct object *obj, int msg_type, const char *message)
926{
927 if (msg_type == FSCK_WARN) {
928 warning("object %s: %s", describe_object(o, obj), message);
929 return 0;
930 }
931 error("object %s: %s", describe_object(o, obj), message);
932 return 1;
933}
