126static inline int missing__target(int code, int result)
127{
128 return /* file:// URL -- do we ever use one??? */
129 (result == CURLE_FILE_COULDNT_READ_FILE) ||
130 /* http:// and https:// URL */
131 (code == 404 && result == CURLE_HTTP_RETURNED_ERROR) ||
132 /* ftp:// URL */
133 (code == 550 && result == CURLE_FTP_COULDNT_RETR_FILE)
134 ;
135}

146/* Options for http_get_*() */
147struct http_get_options {
148 unsigned no_cache:1,
149 keep_error:1,
150 initial_request:1;
151
152 /* If non-NULL, returns the content-type of the response. */
153 struct strbuf *content_type;
154
155 /*
156 * If non-NULL, and content_type above is non-NULL, returns
157 * the charset parameter from the content-type. If none is
158 * present, returns an empty string.
159 */
160 struct strbuf *charset;
161
162 /*
163 * If non-NULL, returns the URL we ended up at, including any
164 * redirects we followed.
165 */
166 struct strbuf *effective_url;
167
168 /*
169 * If both base_url and effective_url are non-NULL, the base URL will
170 * be munged to reflect any redirections going from the requested url
171 * to effective_url. See the definition of update_url_from_redirect
172 * for details.
173 */
174 struct strbuf *base_url;
175};
