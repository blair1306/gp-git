30 long icurr;

33typedef struct s_chastore {
34 chanode_t *head, *tail;
35 long isize, nsize;
36 chanode_t *ancur;

38 long scurr;
39} chastore_t;

43 char const *ptr;
44 long size;
45 unsigned long ha;
46} xrecord_t;

48typedef struct s_xdfile {

50 long nrec;

53 long dstart, dend;
54 xrecord_t **recs;

56 long *rindex;
57 long nreff;
58 unsigned long *ha;
59} xdfile_t;
