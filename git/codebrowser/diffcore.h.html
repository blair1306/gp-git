12/* We internally use unsigned short as the score value,
13 * and rely on an int capable to hold 32-bits. -B can take
14 * -Bmerge_score/break_score format and the two scores are
15 * passed around in one int (high 16-bit for merge and low 16-bit
16 * for break).
17 */
18#define MAX_SCORE 60000.0
19#define DEFAULT_RENAME_SCORE 30000 /* rename/copy similarity minimum (50%) */
20#define DEFAULT_BREAK_SCORE 30000 /* minimum for break to happen (50%) */
21#define DEFAULT_MERGE_SCORE 36000 /* maximum for break-merge to happen (60%) */
22
23#define MINIMUM_BREAK_SIZE 400 /* do not break a file smaller than this */

27struct diff_filespec {
28 struct object_id oid;
29 char *path;
30 void *data;
31 void *cnt_data;
32 unsigned long size;
33 int count; /* Reference count */
34 int rename_used; /* Count of rename users */
35 unsigned short mode; /* file mode */
36 unsigned oid_valid : 1; /* if true, use oid and trust mode;
37 * if false, use the name and read from
38 * the filesystem.
39 */
40#define DIFF_FILE_VALID(spec) (((spec)->mode) != 0)
41 unsigned should_free : 1; /* data should be free()'ed */
42 unsigned should_munmap : 1; /* data should be munmap()'ed */
43 unsigned dirty_submodule : 2; /* For submodules: its work tree is dirty */
44#define DIRTY_SUBMODULE_UNTRACKED 1
45#define DIRTY_SUBMODULE_MODIFIED 2
46 unsigned is_stdin : 1;
47 unsigned has_more_entries : 1; /* only appear in combined diff */
48 /* data should be considered "binary"; -1 means "don't know yet" */
49 signed int is_binary : 2;
50 struct userdiff_driver *driver;
51};

65struct diff_filepair {
66 struct diff_filespec *one;
67 struct diff_filespec *two;
68 unsigned short int score;
69 char status; /* M C R A D U etc. (see Documentation/diff-format.txt or DIFF_STATUS_* in diff.h) */
70 unsigned broken_pair : 1;
71 unsigned renamed_pair : 1;
72 unsigned is_unmerged : 1;
73 unsigned done_skip_stat_unmatch : 1;
74 unsigned skip_stat_unmatch_result : 1;
75};
76#define DIFF_PAIR_UNMERGED(p) ((p)->is_unmerged)
77
78#define DIFF_PAIR_RENAME(p) ((p)->renamed_pair)
79
80#define DIFF_PAIR_BROKEN(p) \
81 ( (!DIFF_FILE_VALID((p)->one) != !DIFF_FILE_VALID((p)->two)) && \
82 ((p)->broken_pair != 0) )
83
84#define DIFF_PAIR_TYPE_CHANGED(p) \
85 ((S_IFMT & (p)->one->mode) != (S_IFMT & (p)->two->mode))
86
87#define DIFF_PAIR_MODE_CHANGED(p) ((p)->one->mode != (p)->two->mode)
