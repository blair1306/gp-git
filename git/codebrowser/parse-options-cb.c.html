12int parse_opt_abbrev_cb(const struct option *opt, const char *arg, int unset)
13{
14 int v;
15
16 if (!arg) {
17 v = unset ? 0 : DEFAULT_ABBREV;
18 } else {
19 v = strtol(arg, (char **)&arg, 10);
20 if (*arg)
21 return opterror(opt, "expects a numerical value", 0);
22 if (v && v < MINIMUM_ABBREV)
23 v = MINIMUM_ABBREV;
24 else if (v > 40)
25 v = 40;
26 }
27 *(int *)(opt->value) = v;
28 return 0;
29}

44int parse_opt_color_flag_cb(const struct option *opt, const char *arg,
45 int unset)
46{
47 int value;
48
49 if (!arg)
50 arg = unset ? "never" : (const char *)opt->defval;
51 value = git_config_colorbool(NULL, arg);
52 if (value < 0)
53 return opterror(opt,
54 "expects \"always\", \"auto\", or \"never\"", 0);
55 *(int *)opt->value = value;

59int parse_opt_verbosity_cb(const struct option *opt, const char *arg,
60 int unset)
61{
62 int *target = opt->value;
63
64 if (unset)
65 /* --no-quiet, --no-verbose */
66 *target = 0;
67 else if (opt->short_name == 'v') {
68 if (*target >= 0)
69 (*target)++;
70 else
71 *target = 1;
72 } else {
73 if (*target <= 0)
74 (*target)--;
75 else
76 *target = -1;
77 }
78 return 0;
79}

81int parse_opt_commits(const struct option *opt, const char *arg, int unset)
82{
83 struct object_id oid;
84 struct commit *commit;
85
86 if (!arg)
87 return -1;
88 if (get_oid(arg, &oid))
89 return error("malformed object name %s", arg);
90 commit = lookup_commit_reference(&oid);
91 if (!commit)
92 return error("no such commit %s", arg);
93 commit_list_insert(commit, opt->value);
94 return 0;
95}

120struct option *parse_options_concat(struct option *a, struct option *b)
121{
122 struct option *ret;
123 size_t i, a_len = 0, b_len = 0;
124
125 for (i = 0; a[i].type != OPTION_END; i++)
126 a_len++;
127 for (i = 0; b[i].type != OPTION_END; i++)
128 b_len++;
129
130 ALLOC_ARRAY(ret, st_add3(a_len, b_len, 1));
131 for (i = 0; i < a_len; i++)
132 ret[i] = a[i];
133 for (i = 0; i < b_len; i++)
134 ret[a_len + i] = b[i];
135 ret[a_len + b_len] = b[b_len]; /* final OPTION_END */
136
137 return ret;
138}

173/**
174 * Recreates the command-line option in the strbuf.
175 */
176static int recreate_opt(struct strbuf *sb, const struct option *opt,
177 const char *arg, int unset)
178{
179 strbuf_reset(sb);
180
181 if (opt->long_name) {
182 strbuf_addstr(sb, unset ? "--no-" : "--");
183 strbuf_addstr(sb, opt->long_name);
184 if (arg) {
185 strbuf_addch(sb, '=');
186 strbuf_addstr(sb, arg);
187 }
188 } else if (opt->short_name && !unset) {
189 strbuf_addch(sb, '-');
190 strbuf_addch(sb, opt->short_name);
191 if (arg)
192 strbuf_addstr(sb, arg);
193 } else
194 return -1;
195
196 return 0;
197}

199/**
200 * For an option opt, recreates the command-line option in opt->value which
201 * must be an char* initialized to NULL. This is useful when we need to pass
202 * the command-line option to another command. Since any previous value will be
203 * overwritten, this callback should only be used for options where the last
204 * one wins.
205 */
206int parse_opt_passthru(const struct option *opt, const char *arg, int unset)
207{
208 static struct strbuf sb = STRBUF_INIT;
209 char **opt_value = opt->value;
210
211 if (recreate_opt(&sb, opt, arg, unset) < 0)
212 return -1;
213
214 free(*opt_value);
215
216 *opt_value = strbuf_detach(&sb, NULL);
217
218 return 0;
219}

221/**
222 * For an option opt, recreate the command-line option, appending it to
223 * opt->value which must be a argv_array. This is useful when we need to pass
224 * the command-line option, which can be specified multiple times, to another
225 * command.
226 */
227int parse_opt_passthru_argv(const struct option *opt, const char *arg, int unset)
228{
229 static struct strbuf sb = STRBUF_INIT;
230 struct argv_array *opt_value = opt->value;
231
232 if (recreate_opt(&sb, opt, arg, unset) < 0)
233 return -1;
234
235 argv_array_push(opt_value, sb.buf);
236
237 return 0;
238}
