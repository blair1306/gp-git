12int optbug(const struct option *opt, const char *reason)
13{
14 if (opt->long_name) {
15 if (opt->short_name)
16 return error("BUG: switch '%c' (--%s) %s",
17 opt->short_name, opt->long_name, reason);
18 return error("BUG: option '%s' %s", opt->long_name, reason);
19 }
20 return error("BUG: switch '%c' %s", opt->short_name, reason);
21}

23static int get_arg(struct parse_opt_ctx_t *p, const struct option *opt,
24 int flags, const char **arg)
25{
26 if (p->opt) {
27 *arg = p->opt;
28 p->opt = NULL;
29 } else if (p->argc == 1 && (opt->flags & PARSE_OPT_LASTARG_DEFAULT)) {
30 *arg = (const char *)opt->defval;
31 } else if (p->argc > 1) {
32 p->argc--;
33 *arg = *++p->argv;
34 } else
35 return opterror(opt, "requires a value", flags);
36 return 0;
37}

47static int opt_command_mode_error(const struct option *opt,
48 const struct option *all_opts,
49 int flags)
50{
51 const struct option *that;
52 struct strbuf message = STRBUF_INIT;
53 struct strbuf that_name = STRBUF_INIT;
54
55 /*
56 * Find the other option that was used to set the variable
57 * already, and report that this is not compatible with it.
58 */
59 for (that = all_opts; that->type != OPTION_END; that++) {
60 if (that == opt ||
61 that->type != OPTION_CMDMODE ||
62 that->value != opt->value ||
63 that->defval != *(int *)opt->value)
64 continue;
65
66 if (that->long_name)
67 strbuf_addf(&that_name, "--%s", that->long_name);
68 else
69 strbuf_addf(&that_name, "-%c", that->short_name);
70 strbuf_addf(&message, ": incompatible with %s", that_name.buf);
71 strbuf_release(&that_name);
72 opterror(opt, message.buf, flags);
73 strbuf_release(&message);
74 return -1;
75 }
76 return opterror(opt, ": incompatible with something else", flags);
77}

79static int get_value(struct parse_opt_ctx_t *p,
80 const struct option *opt,
81 const struct option *all_opts,
82 int flags)
83{
84 const char *s, *arg;
85 const int unset = flags & OPT_UNSET;
86 int err;
87
88 if (unset && p->opt)
89 return opterror(opt, "takes no value", flags);
90 if (unset && (opt->flags & PARSE_OPT_NONEG))
91 return opterror(opt, "isn't available", flags);
92 if (!(flags & OPT_SHORT) && p->opt && (opt->flags & PARSE_OPT_NOARG))
93 return opterror(opt, "takes no value", flags);
94
95 switch (opt->type) {
96 case OPTION_LOWLEVEL_CALLBACK:
97 return (*(parse_opt_ll_cb *)opt->callback)(p, opt, unset);
98
99 case OPTION_BIT:
100 if (unset)
101 *(int *)opt->value &= ~opt->defval;
102 else
103 *(int *)opt->value |= opt->defval;
104 return 0;
105
106 case OPTION_NEGBIT:
107 if (unset)
108 *(int *)opt->value |= opt->defval;
109 else
110 *(int *)opt->value &= ~opt->defval;
111 return 0;
112
113 case OPTION_COUNTUP:
114 if (*(int *)opt->value < 0)
115 *(int *)opt->value = 0;
116 *(int *)opt->value = unset ? 0 : *(int *)opt->value + 1;
117 return 0;
118
119 case OPTION_SET_INT:
120 *(int *)opt->value = unset ? 0 : opt->defval;
121 return 0;
122
123 case OPTION_CMDMODE:
124 /*
125 * Giving the same mode option twice, although is unnecessary,
126 * is not a grave error, so let it pass.
127 */
128 if (*(int *)opt->value && *(int *)opt->value != opt->defval)
129 return opt_command_mode_error(opt, all_opts, flags);
130 *(int *)opt->value = opt->defval;
131 return 0;
132
133 case OPTION_STRING:
134 if (unset)
135 *(const char **)opt->value = NULL;
136 else if (opt->flags & PARSE_OPT_OPTARG && !p->opt)
137 *(const char **)opt->value = (const char *)opt->defval;
138 else
139 return get_arg(p, opt, flags, (const char **)opt->value);
140 return 0;
141
142 case OPTION_FILENAME:
143 err = 0;
144 if (unset)
145 *(const char **)opt->value = NULL;
146 else if (opt->flags & PARSE_OPT_OPTARG && !p->opt)
147 *(const char **)opt->value = (const char *)opt->defval;
148 else
149 err = get_arg(p, opt, flags, (const char **)opt->value);
150
151 if (!err)
152 fix_filename(p->prefix, (const char **)opt->value);
153 return err;
154
155 case OPTION_CALLBACK:
156 if (unset)
157 return (*opt->callback)(opt, NULL, 1) ? (-1) : 0;
158 if (opt->flags & PARSE_OPT_NOARG)
159 return (*opt->callback)(opt, NULL, 0) ? (-1) : 0;
160 if (opt->flags & PARSE_OPT_OPTARG && !p->opt)
161 return (*opt->callback)(opt, NULL, 0) ? (-1) : 0;
162 if (get_arg(p, opt, flags, &arg))
163 return -1;
164 return (*opt->callback)(opt, arg, 0) ? (-1) : 0;
165
166 case OPTION_INTEGER:
167 if (unset) {
168 *(int *)opt->value = 0;
169 return 0;
170 }
171 if (opt->flags & PARSE_OPT_OPTARG && !p->opt) {
172 *(int *)opt->value = opt->defval;
173 return 0;
174 }
175 if (get_arg(p, opt, flags, &arg))
176 return -1;
177 *(int *)opt->value = strtol(arg, (char **)&s, 10);
178 if (*s)
179 return opterror(opt, "expects a numerical value", flags);
180 return 0;
181
182 case OPTION_MAGNITUDE:
183 if (unset) {
184 *(unsigned long *)opt->value = 0;
185 return 0;
186 }
187 if (opt->flags & PARSE_OPT_OPTARG && !p->opt) {
188 *(unsigned long *)opt->value = opt->defval;
189 return 0;
190 }
191 if (get_arg(p, opt, flags, &arg))
192 return -1;
193 if (!git_parse_ulong(arg, opt->value))
194 return opterror(opt,
195 "expects a non-negative integer value with an optional k/m/g suffix",
196 flags);
197 return 0;
198
199 default:
200 die("should not happen, someone must be hit on the forehead");
201 }
202}

204static int parse_short_opt(struct parse_opt_ctx_t *p, const struct option *options)
205{
206 const struct option *all_opts = options;
207 const struct option *numopt = NULL;
208
209 for (; options->type != OPTION_END; options++) {
210 if (options->short_name == *p->opt) {
211 p->opt = p->opt[1] ? p->opt + 1 : NULL;
212 return get_value(p, options, all_opts, OPT_SHORT);
213 }
214
215 /*
216 * Handle the numerical option later, explicit one-digit
217 * options take precedence over it.
218 */
219 if (options->type == OPTION_NUMBER)
220 numopt = options;
221 }
222 if (numopt && isdigit(*p->opt)) {
223 size_t len = 1;
224 char *arg;
225 int rc;
226
227 while (isdigit(p->opt[len]))
228 len++;
229 arg = xmemdupz(p->opt, len);
230 p->opt = p->opt[len] ? p->opt + len : NULL;
231 rc = (*numopt->callback)(numopt, arg, 0) ? (-1) : 0;
232 free(arg);
233 return rc;
234 }
235 return -2;
236}

238static int parse_long_opt(struct parse_opt_ctx_t *p, const char *arg,
239 const struct option *options)
240{
241 const struct option *all_opts = options;
242 const char *arg_end = strchrnul(arg, '=');
243 const struct option *abbrev_option = NULL, *ambiguous_option = NULL;
244 int abbrev_flags = 0, ambiguous_flags = 0;
245
246 for (; options->type != OPTION_END; options++) {
247 const char *rest, *long_name = options->long_name;
248 int flags = 0, opt_flags = 0;
249
250 if (!long_name)
251 continue;
252
253again:
254 if (!skip_prefix(arg, long_name, &rest))
255 rest = NULL;
256 if (options->type == OPTION_ARGUMENT) {
257 if (!rest)
258 continue;
259 if (*rest == '=')
260 return opterror(options, "takes no value", flags);
261 if (*rest)
262 continue;
263 p->out[p->cpidx++] = arg - 2;
264 return 0;
265 }
266 if (!rest) {
267 /* abbreviated? */
268 if (!strncmp(long_name, arg, arg_end - arg)) {
269is_abbreviated:
270 if (abbrev_option) {
271 /*
272 * If this is abbreviated, it is
273 * ambiguous. So when there is no
274 * exact match later, we need to
275 * error out.
276 */
277 ambiguous_option = abbrev_option;
278 ambiguous_flags = abbrev_flags;
279 }
280 if (!(flags & OPT_UNSET) && *arg_end)
281 p->opt = arg_end + 1;
282 abbrev_option = options;
283 abbrev_flags = flags ^ opt_flags;
284 continue;
285 }
286 /* negation allowed? */
287 if (options->flags & PARSE_OPT_NONEG)
288 continue;
289 /* negated and abbreviated very much? */
290 if (starts_with("no-", arg)) {
291 flags |= OPT_UNSET;
292 goto is_abbreviated;
293 }
294 /* negated? */
295 if (!starts_with(arg, "no-")) {
296 if (starts_with(long_name, "no-")) {
297 long_name += 3;
298 opt_flags |= OPT_UNSET;
299 goto again;
300 }
301 continue;
302 }
303 flags |= OPT_UNSET;
304 if (!skip_prefix(arg + 3, long_name, &rest)) {
305 /* abbreviated and negated? */
306 if (starts_with(long_name, arg + 3))
307 goto is_abbreviated;
308 else
309 continue;
310 }
311 }
312 if (*rest) {
313 if (*rest != '=')
314 continue;
315 p->opt = rest + 1;
316 }
317 return get_value(p, options, all_opts, flags ^ opt_flags);
318 }
319
320 if (ambiguous_option)
321 return error("Ambiguous option: %s "
322 "(could be --%s%s or --%s%s)",
323 arg,
324 (ambiguous_flags & OPT_UNSET) ? "no-" : "",
325 ambiguous_option->long_name,
326 (abbrev_flags & OPT_UNSET) ? "no-" : "",
327 abbrev_option->long_name);
328 if (abbrev_option)
329 return get_value(p, abbrev_option, all_opts, abbrev_flags);
330 return -2;
331}

333static int parse_nodash_opt(struct parse_opt_ctx_t *p, const char *arg,
334 const struct option *options)
335{
336 const struct option *all_opts = options;
337
338 for (; options->type != OPTION_END; options++) {
339 if (!(options->flags & PARSE_OPT_NODASH))
340 continue;
341 if (options->short_name == arg[0] && arg[1] == '\0')
342 return get_value(p, options, all_opts, OPT_SHORT);
343 }
344 return -2;
345}

347static void check_typos(const char *arg, const struct option *options)
348{
349 if (strlen(arg) < 3)
350 return;
351
352 if (starts_with(arg, "no-")) {
353 error ("did you mean `--%s` (with two dashes ?)", arg);
354 exit(129);
355 }
356
357 for (; options->type != OPTION_END; options++) {
358 if (!options->long_name)
359 continue;
360 if (starts_with(options->long_name, arg)) {
361 error ("did you mean `--%s` (with two dashes ?)", arg);
362 exit(129);
363 }
364 }

367static void parse_options_check(const struct option *opts)
368{
369 int err = 0;
370 char short_opts[128];
371
372 memset(short_opts, '\0', sizeof(short_opts));
373 for (; opts->type != OPTION_END; opts++) {
374 if ((opts->flags & PARSE_OPT_LASTARG_DEFAULT) &&
375 (opts->flags & PARSE_OPT_OPTARG))
376 err |= optbug(opts, "uses incompatible flags "
377 "LASTARG_DEFAULT and OPTARG");
378 if (opts->short_name) {
379 if (0x7F <= opts->short_name)
380 err |= optbug(opts, "invalid short name");
381 else if (short_opts[opts->short_name]++)
382 err |= optbug(opts, "short name already used");
383 }
384 if (opts->flags & PARSE_OPT_NODASH &&
385 ((opts->flags & PARSE_OPT_OPTARG) ||
386 !(opts->flags & PARSE_OPT_NOARG) ||
387 !(opts->flags & PARSE_OPT_NONEG) ||
388 opts->long_name))
389 err |= optbug(opts, "uses feature "
390 "not supported for dashless options");
391 switch (opts->type) {
392 case OPTION_COUNTUP:
393 case OPTION_BIT:
394 case OPTION_NEGBIT:
395 case OPTION_SET_INT:
396 case OPTION_NUMBER:
397 if ((opts->flags & PARSE_OPT_OPTARG) ||
398 !(opts->flags & PARSE_OPT_NOARG))
399 err |= optbug(opts, "should not accept an argument");
400 default:
401 ; /* ok. (usually accepts an argument) */
402 }
403 if (opts->argh &&
404 strcspn(opts->argh, " _") != strlen(opts->argh))
405 err |= optbug(opts, "multi-word argh should use dash to separate words");
406 }
407 if (err)
408 exit(128);
409}

411void parse_options_start(struct parse_opt_ctx_t *ctx,
412 int argc, const char **argv, const char *prefix,
413 const struct option *options, int flags)
414{
415 memset(ctx, 0, sizeof(*ctx));
416 ctx->argc = ctx->total = argc - 1;
417 ctx->argv = argv + 1;
418 ctx->out = argv;
419 ctx->prefix = prefix;
420 ctx->cpidx = ((flags & PARSE_OPT_KEEP_ARGV0) != 0);
421 ctx->flags = flags;
422 if ((flags & PARSE_OPT_KEEP_UNKNOWN) &&
423 (flags & PARSE_OPT_STOP_AT_NON_OPTION))
424 die("STOP_AT_NON_OPTION and KEEP_UNKNOWN don't go together");
425 parse_options_check(options);
426}

432int parse_options_step(struct parse_opt_ctx_t *ctx,
433 const struct option *options,
434 const char * const usagestr[])
435{
436 int internal_help = !(ctx->flags & PARSE_OPT_NO_INTERNAL_HELP);
437 int err = 0;
438
439 /* we must reset ->opt, unknown short option leave it dangling */
440 ctx->opt = NULL;
441
442 for (; ctx->argc; ctx->argc--, ctx->argv++) {
443 const char *arg = ctx->argv[0];
444
445 if (*arg != '-' || !arg[1]) {
446 if (parse_nodash_opt(ctx, arg, options) == 0)
447 continue;
448 if (ctx->flags & PARSE_OPT_STOP_AT_NON_OPTION)
449 return PARSE_OPT_NON_OPTION;
450 ctx->out[ctx->cpidx++] = ctx->argv[0];
451 continue;
452 }
453
454 /* lone -h asks for help */
455 if (internal_help && ctx->total == 1 && !strcmp(arg + 1, "h"))
456 goto show_usage;
457
458 if (arg[1] != '-') {
459 ctx->opt = arg + 1;
460 switch (parse_short_opt(ctx, options)) {
461 case -1:
462 goto show_usage_error;
463 case -2:
464 if (ctx->opt)
465 check_typos(arg + 1, options);
466 if (internal_help && *ctx->opt == 'h')
467 goto show_usage;
468 goto unknown;
469 }
470 if (ctx->opt)
471 check_typos(arg + 1, options);
472 while (ctx->opt) {
473 switch (parse_short_opt(ctx, options)) {
474 case -1:
475 goto show_usage_error;
476 case -2:
477 if (internal_help && *ctx->opt == 'h')
478 goto show_usage;
479
480 /* fake a short option thing to hide the fact that we may have
481 * started to parse aggregated stuff
482 *
483 * This is leaky, too bad.
484 */
485 ctx->argv[0] = xstrdup(ctx->opt - 1);
486 *(char *)ctx->argv[0] = '-';
487 goto unknown;
488 }
489 }
490 continue;
491 }
492
493 if (!arg[2]) { /* "--" */
494 if (!(ctx->flags & PARSE_OPT_KEEP_DASHDASH)) {
495 ctx->argc--;
496 ctx->argv++;
497 }
498 break;
499 }
500
501 if (internal_help && !strcmp(arg + 2, "help-all"))
502 return usage_with_options_internal(ctx, usagestr, options, 1, 0);
503 if (internal_help && !strcmp(arg + 2, "help"))
504 goto show_usage;
505 switch (parse_long_opt(ctx, arg + 2, options)) {
506 case -1:
507 goto show_usage_error;
508 case -2:
509 goto unknown;
510 }
511 continue;
512unknown:
513 if (!(ctx->flags & PARSE_OPT_KEEP_UNKNOWN))
514 return PARSE_OPT_UNKNOWN;
515 ctx->out[ctx->cpidx++] = ctx->argv[0];
516 ctx->opt = NULL;
517 }
518 return PARSE_OPT_DONE;
519
520 show_usage_error:
521 err = 1;
522 show_usage:
523 return usage_with_options_internal(ctx, usagestr, options, 0, err);
524}

526int parse_options_end(struct parse_opt_ctx_t *ctx)
527{
528 memmove(ctx->out + ctx->cpidx, ctx->argv, ctx->argc * sizeof(*ctx->out));
529 ctx->out[ctx->cpidx + ctx->argc] = NULL;
530 return ctx->cpidx + ctx->argc;
531}

533int parse_options(int argc, const char **argv, const char *prefix,
534 const struct option *options, const char * const usagestr[],
535 int flags)
536{
537 struct parse_opt_ctx_t ctx;
538
539 parse_options_start(&ctx, argc, argv, prefix, options, flags);
540 switch (parse_options_step(&ctx, options, usagestr)) {
541 case PARSE_OPT_HELP:
542 exit(129);
543 case PARSE_OPT_NON_OPTION:
544 case PARSE_OPT_DONE:
545 break;
546 default: /* PARSE_OPT_UNKNOWN */
547 if (ctx.argv[0][1] == '-') {
548 error("unknown option `%s'", ctx.argv[0] + 2);
549 } else if (isascii(*ctx.opt)) {
550 error("unknown switch `%c'", *ctx.opt);
551 } else {
552 error("unknown non-ascii option in string: `%s'",
553 ctx.argv[0]);
554 }
555 usage_with_options(usagestr, options);
556 }
557
558 precompose_argv(argc, argv);
559 return parse_options_end(&ctx);
560}

562static int usage_argh(const struct option *opts, FILE *outfile)
563{
564 const char *s;
565 int literal = (opts->flags & PARSE_OPT_LITERAL_ARGHELP) || !opts->argh;
566 if (opts->flags & PARSE_OPT_OPTARG)
567 if (opts->long_name)
568 s = literal ? "[=%s]" : "[=<%s>]";
569 else
570 s = literal ? "[%s]" : "[<%s>]";
571 else
572 s = literal ? " %s" : " <%s>";
573 return utf8_fprintf(outfile, s, opts->argh ? _(opts->argh) : _("..."));
574}

579static int usage_with_options_internal(struct parse_opt_ctx_t *ctx,
580 const char * const *usagestr,
581 const struct option *opts, int full, int err)
582{
583 FILE *outfile = err ? stderr : stdout;
584
585 if (!usagestr)
586 return PARSE_OPT_HELP;
587
588 if (!err && ctx && ctx->flags & PARSE_OPT_SHELL_EVAL)
589 fprintf(outfile, "cat <<\\EOF\n");
590
591 fprintf_ln(outfile, _("usage: %s"), _(*usagestr++));
592 while (*usagestr && **usagestr)
593 /*
594 * TRANSLATORS: the colon here should align with the
595 * one in "usage: %s" translation.
596 */
597 fprintf_ln(outfile, _(" or: %s"), _(*usagestr++));
598 while (*usagestr) {
599 if (**usagestr)
600 fprintf_ln(outfile, _(" %s"), _(*usagestr));
601 else
602 putchar('\n');
603 usagestr++;
604 }
605
606 if (opts->type != OPTION_GROUP)
607 fputc('\n', outfile);
608
609 for (; opts->type != OPTION_END; opts++) {
610 size_t pos;
611 int pad;
612
613 if (opts->type == OPTION_GROUP) {
614 fputc('\n', outfile);
615 if (*opts->help)
616 fprintf(outfile, "%s\n", _(opts->help));
617 continue;
618 }
619 if (!full && (opts->flags & PARSE_OPT_HIDDEN))
620 continue;
621
622 pos = fprintf(outfile, " ");
623 if (opts->short_name) {
624 if (opts->flags & PARSE_OPT_NODASH)
625 pos += fprintf(outfile, "%c", opts->short_name);
626 else
627 pos += fprintf(outfile, "-%c", opts->short_name);
628 }
629 if (opts->long_name && opts->short_name)
630 pos += fprintf(outfile, ", ");
631 if (opts->long_name)
632 pos += fprintf(outfile, "--%s", opts->long_name);
633 if (opts->type == OPTION_NUMBER)
634 pos += utf8_fprintf(outfile, _("-NUM"));
635
636 if ((opts->flags & PARSE_OPT_LITERAL_ARGHELP) ||
637 !(opts->flags & PARSE_OPT_NOARG))
638 pos += usage_argh(opts, outfile);
639
640 if (pos <= USAGE_OPTS_WIDTH)
641 pad = USAGE_OPTS_WIDTH - pos;
642 else {
643 fputc('\n', outfile);
644 pad = USAGE_OPTS_WIDTH;
645 }
646 fprintf(outfile, "%*s%s\n", pad + USAGE_GAP, "", _(opts->help));
647 }
648 fputc('\n', outfile);
649
650 if (!err && ctx && ctx->flags & PARSE_OPT_SHELL_EVAL)
651 fputs("EOF\n", outfile);
652
653 return PARSE_OPT_HELP;
654}

671#undef opterror
672int opterror(const struct option *opt, const char *reason, int flags)
673{
674 if (flags & OPT_SHORT)
675 return error("switch `%c' %s", opt->short_name, reason);
676 if (flags & OPT_UNSET)
677 return error("option `no-%s' %s", opt->long_name, reason);
678 return error("option `%s' %s", opt->long_name, reason);
679}
