183static struct userdiff_driver *userdiff_find_by_namelen(const char *k, int len)
184{
185 int i;
186 for (i = 0; i < ndrivers; i++) {
187 struct userdiff_driver *drv = drivers + i;
188 if (!strncmp(drv->name, k, len) && !drv->name[len])
189 return drv;
190 }
191 for (i = 0; i < ARRAY_SIZE(builtin_drivers); i++) {
192 struct userdiff_driver *drv = builtin_drivers + i;
193 if (!strncmp(drv->name, k, len) && !drv->name[len])
194 return drv;
195 }
196 return NULL;
197}

223int userdiff_config(const char *k, const char *v)
224{
225 struct userdiff_driver *drv;
226 const char *name, *type;
227 int namelen;
228
229 if (parse_config_key(k, "diff", &name, &namelen, &type) || !name)
230 return 0;
231
232 drv = userdiff_find_by_namelen(name, namelen);
233 if (!drv) {
234 ALLOC_GROW(drivers, ndrivers+1, drivers_alloc);
235 drv = &drivers[ndrivers++];
236 memset(drv, 0, sizeof(*drv));
237 drv->name = xmemdupz(name, namelen);
238 drv->binary = -1;
239 }
240
241 if (!strcmp(type, "funcname"))
242 return parse_funcname(&drv->funcname, k, v, 0);
243 if (!strcmp(type, "xfuncname"))
244 return parse_funcname(&drv->funcname, k, v, REG_EXTENDED);
245 if (!strcmp(type, "binary"))
246 return parse_tristate(&drv->binary, k, v);
247 if (!strcmp(type, "command"))
248 return git_config_string(&drv->external, k, v);
249 if (!strcmp(type, "textconv"))
250 return git_config_string(&drv->textconv, k, v);
251 if (!strcmp(type, "cachetextconv"))
252 return parse_bool(&drv->textconv_want_cache, k, v);
253 if (!strcmp(type, "wordregex"))
254 return git_config_string(&drv->word_regex, k, v);
255
256 return 0;
257}

264struct userdiff_driver *userdiff_find_by_path(const char *path)
265{
266 static struct attr_check *check;
267
268 if (!check)
269 check = attr_check_initl("diff", NULL);
270 if (!path)
271 return NULL;
272 if (git_check_attr(path, check))
273 return NULL;
274
275 if (ATTR_TRUE(check->items[0].value))
276 return &driver_true;
277 if (ATTR_FALSE(check->items[0].value))
278 return &driver_false;
279 if (ATTR_UNSET(check->items[0].value))
280 return NULL;
281 return userdiff_find_by_name(check->items[0].value);
282}

284struct userdiff_driver *userdiff_get_textconv(struct userdiff_driver *driver)
285{
286 if (!driver->textconv)
287 return NULL;
288
289 if (driver->textconv_want_cache && !driver->textconv_cache) {
290 struct notes_cache *c = xmalloc(sizeof(*c));
291 struct strbuf name = STRBUF_INIT;
292
293 strbuf_addf(&name, "textconv/%s", driver->name);
294 notes_cache_init(c, name.buf, driver->textconv);
295 driver->textconv_cache = c;
296 }
297
298 return driver;
299}
