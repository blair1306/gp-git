5static void report_error(const char *class, int ch)
6{
7 printf("%s classifies char %d (0x%02x) wrongly\n", class, ch, ch);

11static int is_in(const char *s, int ch)
12{
13 /* We can't find NUL using strchr. It's classless anyway. */
14 if (ch == '\0')
15 return 0;
16 return !!strchr(s, ch);
17}

19#define TEST_CLASS(t,s) { \
20 int i; \
21 for (i = 0; i < 256; i++) { \
22 if (is_in(s, i) != t(i)) \
23 report_error(#t, i); \
24 } \
25}
