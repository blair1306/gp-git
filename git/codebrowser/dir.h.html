39/*
40 * Each excludes file will be parsed into a fresh exclude_list which
41 * is appended to the relevant exclude_list_group (either EXC_DIRS or
42 * EXC_FILE). An exclude_list within the EXC_CMDL exclude_list_group
43 * can also be used to represent the list of --exclude values passed
44 * via CLI args.
45 */

48 int alloc;
49

57};

83/*
84 * Untracked cache
85 *
86 * The following inputs are sufficient to determine what files in a
87 * directory are excluded:
88 *
89 * - The list of files and directories of the directory in question
90 * - The $GIT_DIR/index
91 * - dir_struct flags
92 * - The content of $GIT_DIR/info/exclude
93 * - The content of core.excludesfile
94 * - The content (or the lack) of .gitignore of all parent directories
95 * from $GIT_WORK_TREE
96 * - The check_only flag in read_directory_recursive (for
97 * DIR_HIDE_EMPTY_DIRECTORIES)
98 *
99 * The first input can be checked using directory mtime. In many
100 * filesystems, directory mtime (stat_data field) is updated when its
101 * files or direct subdirs are added or removed.
102 *
103 * The second one can be hooked from cache_tree_invalidate_path().
104 * Whenever a file (or a submodule) is added or removed from a
105 * directory, we invalidate that directory.
106 *
107 * The remaining inputs are easy, their SHA-1 could be used to verify
108 * their contents (exclude_sha1[], info_exclude_sha1[] and
109 * excludes_file_sha1[])
110 */
111struct untracked_cache_dir {
112 struct untracked_cache_dir **dirs;
113 char **untracked;
114 struct stat_data stat_data;
115 unsigned int untracked_alloc, dirs_nr, dirs_alloc;
116 unsigned int untracked_nr;
117 unsigned int check_only : 1;
118 /* all data except 'dirs' in this struct are good */
119 unsigned int valid : 1;
120 unsigned int recurse : 1;
121 /* null SHA-1 means this directory does not have .gitignore */
122 unsigned char exclude_sha1[20];
123 char name[FLEX_ARRAY];
124};

144struct dir_struct {
145 int nr, alloc;
146 int ignored_nr, ignored_alloc;
147 enum {
148 DIR_SHOW_IGNORED = 1<<0,
149 DIR_SHOW_OTHER_DIRECTORIES = 1<<1,
150 DIR_HIDE_EMPTY_DIRECTORIES = 1<<2,

156 } flags;
157 struct dir_entry **entries;

159
160 /* Exclude info */
161 const char *exclude_per_dir;
162
163 /*
164 * We maintain three groups of exclude pattern lists:
165 *
166 * EXC_CMDL lists patterns explicitly given on the command line.
167 * EXC_DIRS lists patterns obtained from per-directory ignore files.
168 * EXC_FILE lists patterns from fallback ignore files, e.g.
169 * - .git/info/exclude
170 * - core.excludesfile
171 *
172 * Each group contains multiple exclude lists, a single list
173 * per source.
174 */
175#define EXC_CMDL 0
176#define EXC_DIRS 1
177#define EXC_FILE 2
178 struct exclude_list_group exclude_list_group[3];
179
180 /*
181 * Temporary variables which are used during loading of the
182 * per-directory exclude lists.
183 *
184 * exclude_stack points to the top of the exclude_stack, and
185 * basebuf contains the full path to the current
186 * (sub)directory in the traversal. Exclude points to the
187 * matching exclude struct if the directory is excluded.
188 */
189 struct exclude_stack *exclude_stack;
190 struct exclude *exclude;
191 struct strbuf basebuf;
192
193 /* Enable untracked file cache if set */
194 struct untracked_cache *untracked;
195 struct sha1_stat ss_info_exclude;
196 struct sha1_stat ss_excludes_file;
197 unsigned unmanaged_exclude_files;
198};

326static inline int ce_path_match(const struct cache_entry *ce,
327 const struct pathspec *pathspec,
328 char *seen)
329{
330 return match_pathspec(pathspec, ce->name, ce_namelen(ce), 0, seen,
331 S_ISDIR(ce->ce_mode) || S_ISGITLINK(ce->ce_mode));
332}

334static inline int dir_path_match(const struct dir_entry *ent,
335 const struct pathspec *pathspec,
336 int prefix, char *seen)
337{
338 int has_trailing_dir = ent->len && ent->name[ent->len - 1] == '/';
339 int len = has_trailing_dir ? ent->len - 1 : ent->len;
340 return match_pathspec(pathspec, ent->name, len, prefix, seen,
341 has_trailing_dir);
342}
