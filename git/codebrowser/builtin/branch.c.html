60static int parse_branch_color_slot(const char *slot)
61{
62 if (!strcasecmp(slot, "plain"))
63 return BRANCH_COLOR_PLAIN;
64 if (!strcasecmp(slot, "reset"))
65 return BRANCH_COLOR_RESET;
66 if (!strcasecmp(slot, "remote"))
67 return BRANCH_COLOR_REMOTE;
68 if (!strcasecmp(slot, "local"))
69 return BRANCH_COLOR_LOCAL;
70 if (!strcasecmp(slot, "current"))
71 return BRANCH_COLOR_CURRENT;
72 if (!strcasecmp(slot, "upstream"))
73 return BRANCH_COLOR_UPSTREAM;
74 return -1;
75}

77static int git_branch_config(const char *var, const char *value, void *cb)
78{
79 const char *slot_name;
80
81 if (starts_with(var, "column."))
82 return git_column_config(var, value, "branch", &colopts);
83 if (!strcmp(var, "color.branch")) {
84 branch_use_color = git_config_colorbool(var, value);
85 return 0;
86 }
87 if (skip_prefix(var, "color.branch.", &slot_name)) {
88 int slot = parse_branch_color_slot(slot_name);
89 if (slot < 0)
90 return 0;
91 if (!value)
92 return config_error_nonbool(var);
93 return color_parse(value, branch_colors[slot]);
94 }
95 return git_default_config(var, value, cb);
96}

105static int branch_merged(int kind, const char *name,
106 struct commit *rev, struct commit *head_rev)
107{
108 /*
109 * This checks whether the merge bases of branch and HEAD (or
110 * the other branch this branch builds upon) contains the
111 * branch, which means that the branch has already been merged
112 * safely to HEAD (or the other branch).
113 */
114 struct commit *reference_rev = NULL;
115 const char *reference_name = NULL;
116 void *reference_name_to_free = NULL;
117 int merged;
118
119 if (kind == FILTER_REFS_BRANCHES) {
120 struct branch *branch = branch_get(name);
121 const char *upstream = branch_get_upstream(branch, NULL);
122 struct object_id oid;
123
124 if (upstream &&
125 (reference_name = reference_name_to_free =
126 resolve_refdup(upstream, RESOLVE_REF_READING,
127 oid.hash, NULL)) != NULL)
128 reference_rev = lookup_commit_reference(&oid);
129 }
130 if (!reference_rev)
131 reference_rev = head_rev;
132
133 merged = in_merge_bases(rev, reference_rev);
134
135 /*
136 * After the safety valve is fully redefined to "check with
137 * upstream, if any, otherwise with HEAD", we should just
138 * return the result of the in_merge_bases() above without
139 * any of the following code, but during the transition period,
140 * a gentle reminder is in order.
141 */
142 if ((head_rev != reference_rev) &&
143 in_merge_bases(rev, head_rev) != merged) {
144 if (merged)
145 warning(_("deleting branch '%s' that has been merged to\n"
146 " '%s', but not yet merged to HEAD."),
147 name, reference_name);
148 else
149 warning(_("not deleting branch '%s' that is not yet merged to\n"
150 " '%s', even though it is merged to HEAD."),
151 name, reference_name);
152 }
153 free(reference_name_to_free);
154 return merged;
155}

157static int check_branch_commit(const char *branchname, const char *refname,
158 const struct object_id *oid, struct commit *head_rev,
159 int kinds, int force)
160{
161 struct commit *rev = lookup_commit_reference(oid);
162 if (!rev) {
163 error(_("Couldn't look up commit object for '%s'"), refname);
164 return -1;
165 }
166 if (!force && !branch_merged(kinds, branchname, rev, head_rev)) {
167 error(_("The branch '%s' is not fully merged.\n"
168 "If you are sure you want to delete it, "
169 "run 'git branch -D %s'."), branchname, branchname);
170 return -1;
171 }
172 return 0;
173}

184static int delete_branches(int argc, const char **argv, int force, int kinds,
185 int quiet)
186{
187 struct commit *head_rev = NULL;
188 struct object_id oid;
189 char *name = NULL;
190 const char *fmt;
191 int i;
192 int ret = 0;
193 int remote_branch = 0;
194 struct strbuf bname = STRBUF_INIT;
195 unsigned allowed_interpret;
196
197 switch (kinds) {
198 case FILTER_REFS_REMOTES:
199 fmt = "refs/remotes/%s";
200 /* For subsequent UI messages */
201 remote_branch = 1;
202 allowed_interpret = INTERPRET_BRANCH_REMOTE;
203
204 force = 1;
205 break;
206 case FILTER_REFS_BRANCHES:
207 fmt = "refs/heads/%s";
208 allowed_interpret = INTERPRET_BRANCH_LOCAL;
209 break;
210 default:
211 die(_("cannot use -a with -d"));
212 }
213
214 if (!force) {
215 head_rev = lookup_commit_reference(&head_oid);
216 if (!head_rev)
217 die(_("Couldn't look up commit object for HEAD"));
218 }
219 for (i = 0; i < argc; i++, strbuf_release(&bname)) {
220 char *target = NULL;
221 int flags = 0;
222
223 strbuf_branchname(&bname, argv[i], allowed_interpret);
224 free(name);
225 name = mkpathdup(fmt, bname.buf);
226
227 if (kinds == FILTER_REFS_BRANCHES) {
228 const struct worktree *wt =
229 find_shared_symref("HEAD", name);
230 if (wt) {
231 error(_("Cannot delete branch '%s' "
232 "checked out at '%s'"),
233 bname.buf, wt->path);
234 ret = 1;
235 continue;
236 }
237 }
238
239 target = resolve_refdup(name,
240 RESOLVE_REF_READING
241 | RESOLVE_REF_NO_RECURSE
242 | RESOLVE_REF_ALLOW_BAD_NAME,
243 oid.hash, &flags);
244 if (!target) {
245 error(remote_branch
246 ? _("remote-tracking branch '%s' not found.")
247 : _("branch '%s' not found."), bname.buf);
248 ret = 1;
249 continue;
250 }
251
252 if (!(flags & (REF_ISSYMREF|REF_ISBROKEN)) &&
253 check_branch_commit(bname.buf, name, &oid, head_rev, kinds,
254 force)) {
255 ret = 1;
256 goto next;
257 }
258
259 if (delete_ref(NULL, name, is_null_oid(&oid) ? NULL : oid.hash,
260 REF_NODEREF)) {
261 error(remote_branch
262 ? _("Error deleting remote-tracking branch '%s'")
263 : _("Error deleting branch '%s'"),
264 bname.buf);
265 ret = 1;
266 goto next;
267 }
268 if (!quiet) {
269 printf(remote_branch
270 ? _("Deleted remote-tracking branch %s (was %s).\n")
271 : _("Deleted branch %s (was %s).\n"),
272 bname.buf,
273 (flags & REF_ISBROKEN) ? "broken"
274 : (flags & REF_ISSYMREF) ? target
275 : find_unique_abbrev(oid.hash, DEFAULT_ABBREV));
276 }
277 delete_branch_config(bname.buf);
278
279 next:
280 free(target);
281 }
282
283 free(name);
284
285 return(ret);
286}

288static int calc_maxwidth(struct ref_array *refs, int remote_bonus)
289{
290 int i, max = 0;
291 for (i = 0; i < refs->nr; i++) {
292 struct ref_array_item *it = refs->items[i];
293 const char *desc = it->refname;
294 int w;
295
296 skip_prefix(it->refname, "refs/heads/", &desc);
297 skip_prefix(it->refname, "refs/remotes/", &desc);
298 if (it->kind == FILTER_REFS_DETACHED_HEAD) {
299 char *head_desc = get_head_description();
300 w = utf8_strwidth(head_desc);
301 free(head_desc);
302 } else
303 w = utf8_strwidth(desc);
304
305 if (it->kind == FILTER_REFS_REMOTES)
306 w += remote_bonus;
307 if (w > max)
308 max = w;
309 }
310 return max;
311}

313static const char *quote_literal_for_format(const char *s)
314{
315 static struct strbuf buf = STRBUF_INIT;
316
317 strbuf_reset(&buf);
318 while (*s) {
319 const char *ep = strchrnul(s, '%');
320 if (s < ep)
321 strbuf_add(&buf, s, ep - s);
322 if (*ep == '%') {
323 strbuf_addstr(&buf, "%%");
324 s = ep + 1;
325 } else {
326 s = ep;
327 }
328 }
329 return buf.buf;
330}

332static char *build_format(struct ref_filter *filter, int maxwidth, const char *remote_prefix)
333{
334 struct strbuf fmt = STRBUF_INIT;
335 struct strbuf local = STRBUF_INIT;
336 struct strbuf remote = STRBUF_INIT;
337
338 strbuf_addf(&local, "%%(if)%%(HEAD)%%(then)* %s%%(else) %s%%(end)",
339 branch_get_color(BRANCH_COLOR_CURRENT),
340 branch_get_color(BRANCH_COLOR_LOCAL));
341 strbuf_addf(&remote, " %s",
342 branch_get_color(BRANCH_COLOR_REMOTE));
343
344 if (filter->verbose) {
345 struct strbuf obname = STRBUF_INIT;
346
347 if (filter->abbrev < 0)
348 strbuf_addf(&obname, "%%(objectname:short)");
349 else if (!filter->abbrev)
350 strbuf_addf(&obname, "%%(objectname)");
351 else
352 strbuf_addf(&obname, "%%(objectname:short=%d)", filter->abbrev);
353
354 strbuf_addf(&local, "%%(align:%d,left)%%(refname:lstrip=2)%%(end)", maxwidth);
355 strbuf_addf(&local, "%s", branch_get_color(BRANCH_COLOR_RESET));
356 strbuf_addf(&local, " %s ", obname.buf);
357
358 if (filter->verbose > 1)
359 strbuf_addf(&local, "%%(if)%%(upstream)%%(then)[%s%%(upstream:short)%s%%(if)%%(upstream:track)"
360 "%%(then): %%(upstream:track,nobracket)%%(end)] %%(end)%%(contents:subject)",
361 branch_get_color(BRANCH_COLOR_UPSTREAM), branch_get_color(BRANCH_COLOR_RESET));
362 else
363 strbuf_addf(&local, "%%(if)%%(upstream:track)%%(then)%%(upstream:track) %%(end)%%(contents:subject)");
364
365 strbuf_addf(&remote, "%%(align:%d,left)%s%%(refname:lstrip=2)%%(end)%s"
366 "%%(if)%%(symref)%%(then) -> %%(symref:short)"
367 "%%(else) %s %%(contents:subject)%%(end)",
368 maxwidth, quote_literal_for_format(remote_prefix),
369 branch_get_color(BRANCH_COLOR_RESET), obname.buf);
370 strbuf_release(&obname);
371 } else {
372 strbuf_addf(&local, "%%(refname:lstrip=2)%s%%(if)%%(symref)%%(then) -> %%(symref:short)%%(end)",
373 branch_get_color(BRANCH_COLOR_RESET));
374 strbuf_addf(&remote, "%s%%(refname:lstrip=2)%s%%(if)%%(symref)%%(then) -> %%(symref:short)%%(end)",
375 quote_literal_for_format(remote_prefix),
376 branch_get_color(BRANCH_COLOR_RESET));
377 }
378
379 strbuf_addf(&fmt, "%%(if:notequals=refs/remotes)%%(refname:rstrip=-2)%%(then)%s%%(else)%s%%(end)", local.buf, remote.buf);
380
381 strbuf_release(&local);
382 strbuf_release(&remote);
383 return strbuf_detach(&fmt, NULL);
384}

386static void print_ref_list(struct ref_filter *filter, struct ref_sorting *sorting, struct ref_format *format)
387{
388 int i;
389 struct ref_array array;
390 int maxwidth = 0;
391 const char *remote_prefix = "";
392 struct strbuf out = STRBUF_INIT;
393 char *to_free = NULL;
394
395 /*
396 * If we are listing more than just remote branches,
397 * then remote branches will have a "remotes/" prefix.
398 * We need to account for this in the width.
399 */
400 if (filter->kind != FILTER_REFS_REMOTES)
401 remote_prefix = "remotes/";
402
403 memset(&array, 0, sizeof(array));
404
405 filter_refs(&array, filter, filter->kind | FILTER_REFS_INCLUDE_BROKEN);
406
407 if (filter->verbose)
408 maxwidth = calc_maxwidth(&array, strlen(remote_prefix));
409
410 if (!format->format)
411 format->format = to_free = build_format(filter, maxwidth, remote_prefix);
412 format->use_color = branch_use_color;
413
414 if (verify_ref_format(format))
415 die(_("unable to parse format string"));
416
417 ref_array_sort(sorting, &array);
418
419 for (i = 0; i < array.nr; i++) {
420 format_ref_array_item(array.items[i], format, &out);
421 if (column_active(colopts)) {
422 assert(!filter->verbose && "--column and --verbose are incompatible");
423 /* format to a string_list to let print_columns() do its job */
424 string_list_append(&output, out.buf);
425 } else {
426 fwrite(out.buf, 1, out.len, stdout);
427 putchar('\n');
428 }
429 strbuf_release(&out);
430 }
431
432 ref_array_clear(&array);
433 free(to_free);
434}

436static void reject_rebase_or_bisect_branch(const char *target)
437{
438 struct worktree **worktrees = get_worktrees(0);
439 int i;
440
441 for (i = 0; worktrees[i]; i++) {
442 struct worktree *wt = worktrees[i];
443
444 if (!wt->is_detached)
445 continue;
446
447 if (is_worktree_being_rebased(wt, target))
448 die(_("Branch %s is being rebased at %s"),
449 target, wt->path);
450
451 if (is_worktree_being_bisected(wt, target))
452 die(_("Branch %s is being bisected at %s"),
453 target, wt->path);
454 }
455
456 free_worktrees(worktrees);
457}

459static void rename_branch(const char *oldname, const char *newname, int force)
460{
461 struct strbuf oldref = STRBUF_INIT, newref = STRBUF_INIT, logmsg = STRBUF_INIT;
462 struct strbuf oldsection = STRBUF_INIT, newsection = STRBUF_INIT;
463 int recovery = 0;
464 int clobber_head_ok;
465
466 if (!oldname)
467 die(_("cannot rename the current branch while not on any."));
468
469 if (strbuf_check_branch_ref(&oldref, oldname)) {
470 /*
471 * Bad name --- this could be an attempt to rename a
472 * ref that we used to allow to be created by accident.
473 */
474 if (ref_exists(oldref.buf))
475 recovery = 1;
476 else
477 die(_("Invalid branch name: '%s'"), oldname);
478 }
479
480 /*
481 * A command like "git branch -M currentbranch currentbranch" cannot
482 * cause the worktree to become inconsistent with HEAD, so allow it.
483 */
484 clobber_head_ok = !strcmp(oldname, newname);
485
486 validate_new_branchname(newname, &newref, force, clobber_head_ok);
487
488 reject_rebase_or_bisect_branch(oldref.buf);
489
490 strbuf_addf(&logmsg, "Branch: renamed %s to %s",
491 oldref.buf, newref.buf);
492
493 if (rename_ref(oldref.buf, newref.buf, logmsg.buf))
494 die(_("Branch rename failed"));
495
496 if (recovery)
497 warning(_("Renamed a misnamed branch '%s' away"), oldref.buf + 11);
498
499 if (replace_each_worktree_head_symref(oldref.buf, newref.buf, logmsg.buf))
500 die(_("Branch renamed to %s, but HEAD is not updated!"), newname);
501
502 strbuf_release(&logmsg);
503
504 strbuf_addf(&oldsection, "branch.%s", oldref.buf + 11);
505 strbuf_release(&oldref);
506 strbuf_addf(&newsection, "branch.%s", newref.buf + 11);
507 strbuf_release(&newref);
508 if (git_config_rename_section(oldsection.buf, newsection.buf) < 0)
509 die(_("Branch is renamed, but update of config-file failed"));
510 strbuf_release(&oldsection);
511 strbuf_release(&newsection);
512}

516static int edit_branch_description(const char *branch_name)
517{
518 struct strbuf buf = STRBUF_INIT;
519 struct strbuf name = STRBUF_INIT;
520
521 read_branch_desc(&buf, branch_name);
522 if (!buf.len || buf.buf[buf.len-1] != '\n')
523 strbuf_addch(&buf, '\n');
524 strbuf_commented_addf(&buf,
525 _("Please edit the description for the branch\n"
526 " %s\n"
527 "Lines starting with '%c' will be stripped.\n"),
528 branch_name, comment_line_char);
529 write_file_buf(edit_description(), buf.buf, buf.len);
530 strbuf_reset(&buf);
531 if (launch_editor(edit_description(), &buf, NULL)) {
532 strbuf_release(&buf);
533 return -1;
534 }
535 strbuf_stripspace(&buf, 1);
536
537 strbuf_addf(&name, "branch.%s.description", branch_name);
538 git_config_set(name.buf, buf.len ? buf.buf : NULL);
539 strbuf_release(&name);
540 strbuf_release(&buf);
541
542 return 0;
543}

545int cmd_branch(int argc, const char **argv, const char *prefix)
546{
547 int delete = 0, rename = 0, force = 0, list = 0;
548 int reflog = 0, edit_description = 0;
549 int quiet = 0, unset_upstream = 0;
550 const char *new_upstream = NULL;
551 enum branch_track track;
552 struct ref_filter filter;
553 int icase = 0;
554 static struct ref_sorting *sorting = NULL, **sorting_tail = &sorting;
555 struct ref_format format = REF_FORMAT_INIT;
556
557 struct option options[] = {
558 OPT_GROUP(N_("Generic options")),
559 OPT__VERBOSE(&filter.verbose,
560 N_("show hash and subject, give twice for upstream branch")),
561 OPT__QUIET(&quiet, N_("suppress informational messages")),
562 OPT_SET_INT('t', "track", &track, N_("set up tracking mode (see git-pull(1))"),
563 BRANCH_TRACK_EXPLICIT),
564 OPT_SET_INT( 0, "set-upstream", &track, N_("change upstream info"),
565 BRANCH_TRACK_OVERRIDE),
566 OPT_STRING('u', "set-upstream-to", &new_upstream, N_("upstream"), N_("change the upstream info")),
567 OPT_BOOL(0, "unset-upstream", &unset_upstream, N_("Unset the upstream info")),
568 OPT__COLOR(&branch_use_color, N_("use colored output")),
569 OPT_SET_INT('r', "remotes", &filter.kind, N_("act on remote-tracking branches"),
570 FILTER_REFS_REMOTES),
571 OPT_CONTAINS(&filter.with_commit, N_("print only branches that contain the commit")),
572 OPT_NO_CONTAINS(&filter.no_commit, N_("print only branches that don't contain the commit")),
573 OPT_WITH(&filter.with_commit, N_("print only branches that contain the commit")),
574 OPT_WITHOUT(&filter.no_commit, N_("print only branches that don't contain the commit")),
575 OPT__ABBREV(&filter.abbrev),
576
577 OPT_GROUP(N_("Specific git-branch actions:")),
578 OPT_SET_INT('a', "all", &filter.kind, N_("list both remote-tracking and local branches"),
579 FILTER_REFS_REMOTES | FILTER_REFS_BRANCHES),
580 OPT_BIT('d', "delete", &delete, N_("delete fully merged branch"), 1),
581 OPT_BIT('D', NULL, &delete, N_("delete branch (even if not merged)"), 2),
582 OPT_BIT('m', "move", &rename, N_("move/rename a branch and its reflog"), 1),
583 OPT_BIT('M', NULL, &rename, N_("move/rename a branch, even if target exists"), 2),
584 OPT_BOOL(0, "list", &list, N_("list branch names")),
585 OPT_BOOL('l', "create-reflog", &reflog, N_("create the branch's reflog")),
586 OPT_BOOL(0, "edit-description", &edit_description,
587 N_("edit the description for the branch")),
588 OPT__FORCE(&force, N_("force creation, move/rename, deletion")),
589 OPT_MERGED(&filter, N_("print only branches that are merged")),
590 OPT_NO_MERGED(&filter, N_("print only branches that are not merged")),
591 OPT_COLUMN(0, "column", &colopts, N_("list branches in columns")),
592 OPT_CALLBACK(0 , "sort", sorting_tail, N_("key"),
593 N_("field name to sort on"), &parse_opt_ref_sorting),
594 {
595 OPTION_CALLBACK, 0, "points-at", &filter.points_at, N_("object"),
596 N_("print only branches of the object"), 0, parse_opt_object_name
597 },
598 OPT_BOOL('i', "ignore-case", &icase, N_("sorting and filtering are case insensitive")),
599 OPT_STRING( 0 , "format", &format.format, N_("format"), N_("format to use for the output")),
600 OPT_END(),
601 };
602
603 setup_ref_filter_porcelain_msg();
604
605 memset(&filter, 0, sizeof(filter));
606 filter.kind = FILTER_REFS_BRANCHES;
607 filter.abbrev = -1;
608
609 if (argc == 2 && !strcmp(argv[1], "-h"))
610 usage_with_options(builtin_branch_usage, options);
611
612 git_config(git_branch_config, NULL);
613
614 track = git_branch_track;
615
616 head = resolve_refdup("HEAD", 0, head_oid.hash, NULL);
617 if (!head)
618 die(_("Failed to resolve HEAD as a valid ref."));
619 if (!strcmp(head, "HEAD"))
620 filter.detached = 1;
621 else if (!skip_prefix(head, "refs/heads/", &head))
622 die(_("HEAD not found below refs/heads!"));
623
624 argc = parse_options(argc, argv, prefix, options, builtin_branch_usage,
625 0);
626
627 if (!delete && !rename && !edit_description && !new_upstream && !unset_upstream && argc == 0)
628 list = 1;
629
630 if (filter.with_commit || filter.merge != REF_FILTER_MERGED_NONE || filter.points_at.nr ||
631 filter.no_commit)
632 list = 1;
633
634 if (!!delete + !!rename + !!new_upstream +
635 list + unset_upstream > 1)
636 usage_with_options(builtin_branch_usage, options);
637
638 if (filter.abbrev == -1)
639 filter.abbrev = DEFAULT_ABBREV;
640 filter.ignore_case = icase;
641
642 finalize_colopts(&colopts, -1);
643 if (filter.verbose) {
644 if (explicitly_enable_column(colopts))
645 die(_("--column and --verbose are incompatible"));
646 colopts = 0;
647 }
648
649 if (force) {
650 delete *= 2;
651 rename *= 2;
652 }
653
654 if (delete) {
655 if (!argc)
656 die(_("branch name required"));
657 return delete_branches(argc, argv, delete > 1, filter.kind, quiet);
658 } else if (list) {
659 /* git branch --local also shows HEAD when it is detached */
660 if ((filter.kind & FILTER_REFS_BRANCHES) && filter.detached)
661 filter.kind |= FILTER_REFS_DETACHED_HEAD;
662 filter.name_patterns = argv;
663 /*
664 * If no sorting parameter is given then we default to sorting
665 * by 'refname'. This would give us an alphabetically sorted
666 * array with the 'HEAD' ref at the beginning followed by
667 * local branches 'refs/heads/...' and finally remote-tacking
668 * branches 'refs/remotes/...'.
669 */
670 if (!sorting)
671 sorting = ref_default_sorting();
672 sorting->ignore_case = icase;
673 print_ref_list(&filter, sorting, &format);
674 print_columns(&output, colopts, NULL);
675 string_list_clear(&output, 0);
676 return 0;
677 }
678 else if (edit_description) {
679 const char *branch_name;
680 struct strbuf branch_ref = STRBUF_INIT;
681
682 if (!argc) {
683 if (filter.detached)
684 die(_("Cannot give description to detached HEAD"));
685 branch_name = head;
686 } else if (argc == 1)
687 branch_name = argv[0];
688 else
689 die(_("cannot edit description of more than one branch"));
690
691 strbuf_addf(&branch_ref, "refs/heads/%s", branch_name);
692 if (!ref_exists(branch_ref.buf)) {
693 strbuf_release(&branch_ref);
694
695 if (!argc)
696 return error(_("No commit on branch '%s' yet."),
697 branch_name);
698 else
699 return error(_("No branch named '%s'."),
700 branch_name);
701 }
702 strbuf_release(&branch_ref);
703
704 if (edit_branch_description(branch_name))
705 return 1;
706 } else if (rename) {
707 if (!argc)
708 die(_("branch name required"));
709 else if (argc == 1)
710 rename_branch(head, argv[0], rename > 1);
711 else if (argc == 2)
712 rename_branch(argv[0], argv[1], rename > 1);
713 else
714 die(_("too many branches for a rename operation"));
715 } else if (new_upstream) {
716 struct branch *branch = branch_get(argv[0]);
717
718 if (argc > 1)
719 die(_("too many branches to set new upstream"));
720
721 if (!branch) {
722 if (!argc || !strcmp(argv[0], "HEAD"))
723 die(_("could not set upstream of HEAD to %s when "
724 "it does not point to any branch."),
725 new_upstream);
726 die(_("no such branch '%s'"), argv[0]);
727 }
728
729 if (!ref_exists(branch->refname))
730 die(_("branch '%s' does not exist"), branch->name);
731
732 /*
733 * create_branch takes care of setting up the tracking
734 * info and making sure new_upstream is correct
735 */
736 create_branch(branch->name, new_upstream, 0, 0, 0, quiet, BRANCH_TRACK_OVERRIDE);
737 } else if (unset_upstream) {
738 struct branch *branch = branch_get(argv[0]);
739 struct strbuf buf = STRBUF_INIT;
740
741 if (argc > 1)
742 die(_("too many branches to unset upstream"));
743
744 if (!branch) {
745 if (!argc || !strcmp(argv[0], "HEAD"))
746 die(_("could not unset upstream of HEAD when "
747 "it does not point to any branch."));
748 die(_("no such branch '%s'"), argv[0]);
749 }
750
751 if (!branch_has_merge_config(branch))
752 die(_("Branch '%s' has no upstream information"), branch->name);
753
754 strbuf_addf(&buf, "branch.%s.remote", branch->name);
755 git_config_set_multivar(buf.buf, NULL, NULL, 1);
756 strbuf_reset(&buf);
757 strbuf_addf(&buf, "branch.%s.merge", branch->name);
758 git_config_set_multivar(buf.buf, NULL, NULL, 1);
759 strbuf_release(&buf);
760 } else if (argc > 0 && argc <= 2) {
761 struct branch *branch = branch_get(argv[0]);
762 int branch_existed = 0, remote_tracking = 0;
763 struct strbuf buf = STRBUF_INIT;
764
765 if (!strcmp(argv[0], "HEAD"))
766 die(_("it does not make sense to create 'HEAD' manually"));
767
768 if (!branch)
769 die(_("no such branch '%s'"), argv[0]);
770
771 if (filter.kind != FILTER_REFS_BRANCHES)
772 die(_("-a and -r options to 'git branch' do not make sense with a branch name"));
773
774 if (track == BRANCH_TRACK_OVERRIDE)
775 fprintf(stderr, _("The --set-upstream flag is deprecated and will be removed. Consider using --track or --set-upstream-to\n"));
776
777 strbuf_addf(&buf, "refs/remotes/%s", branch->name);
778 remote_tracking = ref_exists(buf.buf);
779 strbuf_release(&buf);
780
781 branch_existed = ref_exists(branch->refname);
782 create_branch(argv[0], (argc == 2) ? argv[1] : head,
783 force, reflog, 0, quiet, track);
784
785 /*
786 * We only show the instructions if the user gave us
787 * one branch which doesn't exist locally, but is the
788 * name of a remote-tracking branch.
789 */
790 if (argc == 1 && track == BRANCH_TRACK_OVERRIDE &&
791 !branch_existed && remote_tracking) {
792 fprintf(stderr, _("\nIf you wanted to make '%s' track '%s', do this:\n\n"), head, branch->name);
793 fprintf(stderr, " git branch -d %s\n", branch->name);
794 fprintf(stderr, " git branch --set-upstream-to %s\n", branch->name);
795 }
796
797 } else
798 usage_with_options(builtin_branch_usage, options);
799
800 return 0;
801}
