38int get_sha1_hex(const char *hex, unsigned char *sha1)
39{
40 int i;
41 for (i = 0; i < GIT_SHA1_RAWSZ; i++) {
42 int val = hex2chr(hex);
43 if (val < 0)
44 return -1;
45 *sha1++ = val;
46 hex += 2;
47 }
48 return 0;
49}

64char *sha1_to_hex_r(char *buffer, const unsigned char *sha1)
65{
66 static const char hex[] = "0123456789abcdef";
67 char *buf = buffer;
68 int i;
69
70 for (i = 0; i < GIT_SHA1_RAWSZ; i++) {
71 unsigned int val = *sha1++;
72 *buf++ = hex[val >> 4];
73 *buf++ = hex[val & 0xf];
74 }
75 *buf = '\0';
76
77 return buffer;
78}
