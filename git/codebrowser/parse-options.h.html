51/*
52 * `type`::
53 * holds the type of the option, you must have an OPTION_END last in your
54 * array.
55 *
56 * `short_name`::
57 * the character to use as a short option name, '\0' if none.
58 *
59 * `long_name`::
60 * the long option name, without the leading dashes, NULL if none.
61 *
62 * `value`::
63 * stores pointers to the values to be filled.
64 *
65 * `argh`::
66 * token to explain the kind of argument this option wants. Keep it
67 * homogeneous across the repository. Should be wrapped by N_() for
68 * translation.
69 *
70 * `help`::
71 * the short help associated to what the option does.
72 * Must never be NULL (except for OPTION_END).
73 * OPTION_GROUP uses this pointer to store the group header.
74 * Should be wrapped by N_() for translation.
75 *
76 * `flags`::
77 * mask of parse_opt_option_flags.
78 * PARSE_OPT_OPTARG: says that the argument is optional (not for BOOLEANs)
79 * PARSE_OPT_NOARG: says that this option does not take an argument
80 * PARSE_OPT_NONEG: says that this option cannot be negated
81 * PARSE_OPT_HIDDEN: this option is skipped in the default usage, and
82 * shown only in the full usage.
83 * PARSE_OPT_LASTARG_DEFAULT: says that this option will take the default
84 * value if no argument is given when the option
85 * is last on the command line. If the option is
86 * not last it will require an argument.
87 * Should not be used with PARSE_OPT_OPTARG.
88 * PARSE_OPT_NODASH: this option doesn't start with a dash.
89 * PARSE_OPT_LITERAL_ARGHELP: says that argh shouldn't be enclosed in brackets
90 * (i.e. '<argh>') in the help message.
91 * Useful for options with multiple parameters.
92 *
93 * `callback`::
94 * pointer to the callback to use for OPTION_CALLBACK or
95 * OPTION_LOWLEVEL_CALLBACK.
96 *
97 * `defval`::
98 * default value to fill (*->value) with for PARSE_OPT_OPTARG.
99 * OPTION_{BIT,SET_INT} store the {mask,integer} to put in the value when met.
100 * CALLBACKS can use it like they want.
101 */
102struct option {
103 enum parse_opt_type type;
104 int short_name;
105 const char *long_name;
106 void *value;
107 const char *argh;
108 const char *help;
109
110 int flags;
111 parse_opt_cb *callback;
112 intptr_t defval;
113};
