39/* We use one producer thread and THREADS consumer
40 * threads. The producer adds struct work_items to 'todo' and the
41 * consumers pick work items from the same array.
42 */
43struct work_item {
44 struct grep_source source;
45 char done;
46 struct strbuf out;
47};
48
49/* In the range [todo_done, todo_start) in 'todo' we have work_items
50 * that have been or are processed by a consumer thread. We haven't
51 * written the result for these to stdout yet.
52 *
53 * The work_items in [todo_start, todo_end) are waiting to be picked
54 * up by a consumer thread.
55 *
56 * The ranges are modulo TODO_SIZE.
57 */
58#define TODO_SIZE 128
59static struct work_item todo[TODO_SIZE];
60static int todo_start;
61static int todo_end;
62static int todo_done;
63
64/* Has all work items been added? */
65static int all_work_added;
66
67/* This lock protects all the variables above. */
68static pthread_mutex_t grep_mutex;
69
70static inline void grep_lock(void)
71{
72 assert(num_threads);
73 pthread_mutex_lock(&grep_mutex);
74}
75
76static inline void grep_unlock(void)
77{
78 assert(num_threads);
79 pthread_mutex_unlock(&grep_mutex);
80}
81
82/* Signalled when a new work_item is added to todo. */
83static pthread_cond_t cond_add;
84
85/* Signalled when the result from one work_item is written to
86 * stdout.
87 */
88static pthread_cond_t cond_write;
89
90/* Signalled when we are finished with everything. */
91static pthread_cond_t cond_result;
92
93static int skip_first_line;
94
95static void add_work(struct grep_opt *opt, enum grep_source_type type,
96 const char *name, const char *path, const void *id)
97{
98 grep_lock();
99
100 while ((todo_end+1) % ARRAY_SIZE(todo) == todo_done) {
101 pthread_cond_wait(&cond_write, &grep_mutex);
102 }
103
104 grep_source_init(&todo[todo_end].source, type, name, path, id);
105 if (opt->binary != GREP_BINARY_TEXT)
106 grep_source_load_driver(&todo[todo_end].source);
107 todo[todo_end].done = 0;
108 strbuf_reset(&todo[todo_end].out);
109 todo_end = (todo_end + 1) % ARRAY_SIZE(todo);
110
111 pthread_cond_signal(&cond_add);
112 grep_unlock();
113}
114
115static struct work_item *get_work(void)
116{
117 struct work_item *ret;
118
119 grep_lock();
120 while (todo_start == todo_end && !all_work_added) {
121 pthread_cond_wait(&cond_add, &grep_mutex);
122 }
123
124 if (todo_start == todo_end && all_work_added) {
125 ret = NULL;
126 } else {
127 ret = &todo[todo_start];
128 todo_start = (todo_start + 1) % ARRAY_SIZE(todo);
129 }
130 grep_unlock();
131 return ret;
132}
133
134static void work_done(struct work_item *w)
135{
136 int old_done;
137
138 grep_lock();
139 w->done = 1;
140 old_done = todo_done;
141 for(; todo[todo_done].done && todo_done != todo_start;
142 todo_done = (todo_done+1) % ARRAY_SIZE(todo)) {
143 w = &todo[todo_done];
144 if (w->out.len) {
145 const char *p = w->out.buf;
146 size_t len = w->out.len;
147
148 /* Skip the leading hunk mark of the first file. */
149 if (skip_first_line) {
150 while (len) {
151 len--;
152 if (*p++ == '\n')
153 break;
154 }
155 skip_first_line = 0;
156 }
157
158 write_or_die(1, p, len);
159 }
160 grep_source_clear(&w->source);
161 }
162
163 if (old_done != todo_done)
164 pthread_cond_signal(&cond_write);
165
166 if (all_work_added && todo_done == todo_end)
167 pthread_cond_signal(&cond_result);
168
169 grep_unlock();
170}

172static void *run(void *arg)
173{
174 int hit = 0;
175 struct grep_opt *opt = arg;
176
177 while (1) {
178 struct work_item *w = get_work();
179 if (!w)
180 break;
181
182 opt->output_priv = w;
183 hit |= grep_source(opt, &w->source);
184 grep_source_clear_data(&w->source);
185 work_done(w);
186 }
187 free_grep_patterns(arg);
188 free(arg);
189
190 return (void*) (intptr_t) hit;
191}

193static void strbuf_out(struct grep_opt *opt, const void *buf, size_t size)
194{
195 struct work_item *w = opt->output_priv;
196 strbuf_add(&w->out, buf, size);
197}

199static void start_threads(struct grep_opt *opt)
200{
201 int i;
202
203 pthread_mutex_init(&grep_mutex, NULL);
204 pthread_mutex_init(&grep_read_mutex, NULL);
205 pthread_mutex_init(&grep_attr_mutex, NULL);
206 pthread_cond_init(&cond_add, NULL);
207 pthread_cond_init(&cond_write, NULL);
208 pthread_cond_init(&cond_result, NULL);
209 grep_use_locks = 1;
210
211 for (i = 0; i < ARRAY_SIZE(todo); i++) {
212 strbuf_init(&todo[i].out, 0);
213 }
214
215 threads = xcalloc(num_threads, sizeof(*threads));
216 for (i = 0; i < num_threads; i++) {
217 int err;
218 struct grep_opt *o = grep_opt_dup(opt);
219 o->output = strbuf_out;
220 if (i)
221 o->debug = 0;
222 compile_grep_patterns(o);
223 err = pthread_create(&threads[i], NULL, run, o);
224
225 if (err)
226 die(_("grep: failed to create thread: %s"),
227 strerror(err));
228 }
229}

231static int wait_all(void)
232{
233 int hit = 0;
234 int i;
235
236 grep_lock();
237 all_work_added = 1;
238
239 /* Wait until all work is done. */
240 while (todo_done != todo_end)
241 pthread_cond_wait(&cond_result, &grep_mutex);
242
243 /* Wake up all the consumer threads so they can see that there
244 * is no more work to do.
245 */
246 pthread_cond_broadcast(&cond_add);
247 grep_unlock();
248
249 for (i = 0; i < num_threads; i++) {
250 void *h;
251 pthread_join(threads[i], &h);
252 hit |= (int) (intptr_t) h;
253 }
254
255 free(threads);
256
257 pthread_mutex_destroy(&grep_mutex);
258 pthread_mutex_destroy(&grep_read_mutex);
259 pthread_mutex_destroy(&grep_attr_mutex);
260 pthread_cond_destroy(&cond_add);
261 pthread_cond_destroy(&cond_write);
262 pthread_cond_destroy(&cond_result);
263 grep_use_locks = 0;
264
265 return hit;
266}

275static int grep_cmd_config(const char *var, const char *value, void *cb)
276{
277 int st = grep_config(var, value, cb);
278 if (git_default_config(var, value, cb) < 0)
279 st = -1;
280
281 if (!strcmp(var, "grep.threads")) {
282 num_threads = git_config_int(var, value);
283 if (num_threads < 0)
284 die(_("invalid number of threads specified (%d) for %s"),
285 num_threads, var);
286#ifdef NO_PTHREADS
287 else if (num_threads && num_threads != 1) {
288 /*
289 * TRANSLATORS: %s is the configuration
290 * variable for tweaking threads, currently
291 * grep.threads
292 */
293 warning(_("no threads support, ignoring %s"), var);
294 num_threads = 0;
295 }
296#endif
297 }
298
299 if (!strcmp(var, "submodule.recurse"))
300 recurse_submodules = git_config_bool(var, value);
301
302 return st;
303}

315static int grep_oid(struct grep_opt *opt, const struct object_id *oid,
316 const char *filename, int tree_name_len,
317 const char *path)
318{
319 struct strbuf pathbuf = STRBUF_INIT;
320
321 if (opt->relative && opt->prefix_length) {
322 quote_path_relative(filename + tree_name_len, opt->prefix, &pathbuf);
323 strbuf_insert(&pathbuf, 0, filename, tree_name_len);
324 } else {
325 strbuf_addstr(&pathbuf, filename);
326 }
327
328#ifndef NO_PTHREADS
329 if (num_threads) {
330 add_work(opt, GREP_SOURCE_OID, pathbuf.buf, path, oid);
331 strbuf_release(&pathbuf);
332 return 0;
333 } else
334#endif
335 {
336 struct grep_source gs;
337 int hit;
338
339 grep_source_init(&gs, GREP_SOURCE_OID, pathbuf.buf, path, oid);
340 strbuf_release(&pathbuf);
341 hit = grep_source(opt, &gs);
342
343 grep_source_clear(&gs);
344 return hit;
345 }
346}

386static void run_pager(struct grep_opt *opt, const char *prefix)
387{
388 struct string_list *path_list = opt->output_priv;
389 struct child_process child = CHILD_PROCESS_INIT;
390 int i, status;
391

393 argv_array_push(&child.args, path_list->items[i].string);
394 child.dir = prefix;
395 child.use_shell = 1;
396
397 status = run_command(&child);
398 if (status)
399 exit(status);
400}

408static int grep_submodule(struct grep_opt *opt, struct repository *superproject,
409 const struct pathspec *pathspec,
410 const struct object_id *oid,
411 const char *filename, const char *path)
412{
413 struct repository submodule;
414 int hit;
415
416 if (!is_submodule_active(superproject, path))
417 return 0;
418
419 if (repo_submodule_init(&submodule, superproject, path))
420 return 0;
421
422 repo_read_gitmodules(&submodule);
423
424 /*
425 * NEEDSWORK: This adds the submodule's object directory to the list of
426 * alternates for the single in-memory object store. This has some bad
427 * consequences for memory (processed objects will never be freed) and
428 * performance (this increases the number of pack files git has to pay
429 * attention to, to the sum of the number of pack files in all the
430 * repositories processed so far). This can be removed once the object
431 * store is no longer global and instead is a member of the repository
432 * object.
433 */
434 add_to_alternates_memory(submodule.objectdir);
435
436 if (oid) {
437 struct object *object;
438 struct tree_desc tree;
439 void *data;
440 unsigned long size;
441 struct strbuf base = STRBUF_INIT;
442
443 object = parse_object_or_die(oid, oid_to_hex(oid));
444
445 grep_read_lock();
446 data = read_object_with_reference(object->oid.hash, tree_type,
447 &size, NULL);
448 grep_read_unlock();
449
450 if (!data)
451 die(_("unable to read tree (%s)"), oid_to_hex(&object->oid));
452
453 strbuf_addstr(&base, filename);
454 strbuf_addch(&base, '/');
455
456 init_tree_desc(&tree, data, size);
457 hit = grep_tree(opt, pathspec, &tree, &base, base.len,
458 object->type == OBJ_COMMIT, &submodule);
459 strbuf_release(&base);
460 free(data);
461 } else {
462 hit = grep_cache(opt, &submodule, pathspec, 1);
463 }
464
465 repo_clear(&submodule);
466 return hit;
467}

469static int grep_cache(struct grep_opt *opt, struct repository *repo,
470 const struct pathspec *pathspec, int cached)
471{
472 int hit = 0;
473 int nr;
474 struct strbuf name = STRBUF_INIT;
475 int name_base_len = 0;
476 if (repo->submodule_prefix) {
477 name_base_len = strlen(repo->submodule_prefix);
478 strbuf_addstr(&name, repo->submodule_prefix);
479 }
480
481 repo_read_index(repo);
482
483 for (nr = 0; nr < repo->index->cache_nr; nr++) {
484 const struct cache_entry *ce = repo->index->cache[nr];
485 strbuf_setlen(&name, name_base_len);
486 strbuf_addstr(&name, ce->name);
487
488 if (S_ISREG(ce->ce_mode) &&
489 match_pathspec(pathspec, name.buf, name.len, 0, NULL,
490 S_ISDIR(ce->ce_mode) ||
491 S_ISGITLINK(ce->ce_mode))) {
492 /*
493 * If CE_VALID is on, we assume worktree file and its
494 * cache entry are identical, even if worktree file has
495 * been modified, so use cache version instead
496 */
497 if (cached || (ce->ce_flags & CE_VALID) ||
498 ce_skip_worktree(ce)) {
499 if (ce_stage(ce) || ce_intent_to_add(ce))
500 continue;
501 hit |= grep_oid(opt, &ce->oid, name.buf,
502 0, name.buf);
503 } else {
504 hit |= grep_file(opt, name.buf);
505 }
506 } else if (recurse_submodules && S_ISGITLINK(ce->ce_mode) &&
507 submodule_path_match(pathspec, name.buf, NULL)) {
508 hit |= grep_submodule(opt, repo, pathspec, NULL, ce->name, ce->name);
509 } else {
510 continue;
511 }
512
513 if (ce_stage(ce)) {
514 do {
515 nr++;
516 } while (nr < repo->index->cache_nr &&
517 !strcmp(ce->name, repo->index->cache[nr]->name));
518 nr--; /* compensate for loop control */
519 }
520 if (hit && opt->status_only)
521 break;
522 }
523
524 strbuf_release(&name);
525 return hit;
526}

528static int grep_tree(struct grep_opt *opt, const struct pathspec *pathspec,
529 struct tree_desc *tree, struct strbuf *base, int tn_len,
530 int check_attr, struct repository *repo)
531{
532 int hit = 0;
533 enum interesting match = entry_not_interesting;
534 struct name_entry entry;
535 int old_baselen = base->len;
536 struct strbuf name = STRBUF_INIT;
537 int name_base_len = 0;
538 if (repo->submodule_prefix) {
539 strbuf_addstr(&name, repo->submodule_prefix);
540 name_base_len = name.len;
541 }
542
543 while (tree_entry(tree, &entry)) {
544 int te_len = tree_entry_len(&entry);
545
546 if (match != all_entries_interesting) {
547 strbuf_addstr(&name, base->buf + tn_len);
548 match = tree_entry_interesting(&entry, &name,
549 0, pathspec);
550 strbuf_setlen(&name, name_base_len);
551
552 if (match == all_entries_not_interesting)
553 break;
554 if (match == entry_not_interesting)
555 continue;
556 }
557
558 strbuf_add(base, entry.path, te_len);
559
560 if (S_ISREG(entry.mode)) {
561 hit |= grep_oid(opt, entry.oid, base->buf, tn_len,
562 check_attr ? base->buf + tn_len : NULL);
563 } else if (S_ISDIR(entry.mode)) {
564 enum object_type type;
565 struct tree_desc sub;
566 void *data;
567 unsigned long size;
568
569 data = lock_and_read_oid_file(entry.oid, &type, &size);
570 if (!data)
571 die(_("unable to read tree (%s)"),
572 oid_to_hex(entry.oid));
573
574 strbuf_addch(base, '/');
575 init_tree_desc(&sub, data, size);
576 hit |= grep_tree(opt, pathspec, &sub, base, tn_len,
577 check_attr, repo);
578 free(data);
579 } else if (recurse_submodules && S_ISGITLINK(entry.mode)) {
580 hit |= grep_submodule(opt, repo, pathspec, entry.oid,
581 base->buf, base->buf + tn_len);
582 }
583
584 strbuf_setlen(base, old_baselen);
585
586 if (hit && opt->status_only)
587 break;
588 }
589
590 strbuf_release(&name);
591 return hit;
592}

594static int grep_object(struct grep_opt *opt, const struct pathspec *pathspec,
595 struct object *obj, const char *name, const char *path,
596 struct repository *repo)
597{
598 if (obj->type == OBJ_BLOB)
599 return grep_oid(opt, &obj->oid, name, 0, path);
600 if (obj->type == OBJ_COMMIT || obj->type == OBJ_TREE) {
601 struct tree_desc tree;
602 void *data;
603 unsigned long size;
604 struct strbuf base;
605 int hit, len;
606
607 grep_read_lock();
608 data = read_object_with_reference(obj->oid.hash, tree_type,
609 &size, NULL);
610 grep_read_unlock();
611
612 if (!data)
613 die(_("unable to read tree (%s)"), oid_to_hex(&obj->oid));
614
615 len = name ? strlen(name) : 0;
616 strbuf_init(&base, PATH_MAX + len + 1);
617 if (len) {
618 strbuf_add(&base, name, len);
619 strbuf_addch(&base, ':');
620 }
621 init_tree_desc(&tree, data, size);
622 hit = grep_tree(opt, pathspec, &tree, &base, base.len,
623 obj->type == OBJ_COMMIT, repo);
624 strbuf_release(&base);
625 free(data);
626 return hit;
627 }
628 die(_("unable to grep from object of type %s"), typename(obj->type));
629}

631static int grep_objects(struct grep_opt *opt, const struct pathspec *pathspec,
632 struct repository *repo,
633 const struct object_array *list)
634{
635 unsigned int i;
636 int hit = 0;
637 const unsigned int nr = list->nr;
638
639 for (i = 0; i < nr; i++) {
640 struct object *real_obj;
641 real_obj = deref_tag(list->objects[i].item, NULL, 0);
642
643 /* load the gitmodules file for this rev */
644 if (recurse_submodules) {
645 submodule_free();
646 gitmodules_config_oid(&real_obj->oid);
647 }
648 if (grep_object(opt, pathspec, real_obj, list->objects[i].name, list->objects[i].path,
649 repo)) {
650 hit = 1;
651 if (opt->status_only)
652 break;
653 }
654 }
655 return hit;
656}

658static int grep_directory(struct grep_opt *opt, const struct pathspec *pathspec,
659 int exc_std, int use_index)
660{
661 struct dir_struct dir;
662 int i, hit = 0;
663
664 memset(&dir, 0, sizeof(dir));
665 if (!use_index)
666 dir.flags |= DIR_NO_GITLINKS;
667 if (exc_std)
668 setup_standard_excludes(&dir);
669
670 fill_directory(&dir, &the_index, pathspec);
671 for (i = 0; i < dir.nr; i++) {
672 if (!dir_path_match(dir.entries[i], pathspec, 0, NULL))
673 continue;
674 hit |= grep_file(opt, dir.entries[i]->name);
675 if (hit && opt->status_only)
676 break;
677 }
678 return hit;
679}

682 int unset)
683{

701static int file_callback(const struct option *opt, const char *arg, int unset)
702{
703 struct grep_opt *grep_opt = opt->value;
704 int from_stdin = !strcmp(arg, "-");
705 FILE *patterns;
706 int lno = 0;
707 struct strbuf sb = STRBUF_INIT;
708
709 patterns = from_stdin ? stdin : fopen(arg, "r");
710 if (!patterns)
711 die_errno(_("cannot open '%s'"), arg);
712 while (strbuf_getline(&sb, patterns) == 0) {
713 /* ignore empty line like grep does */
714 if (sb.len == 0)
715 continue;
716
717 append_grep_pat(grep_opt, sb.buf, sb.len, arg, ++lno,
718 GREP_PATTERN);
719 }
720 if (!from_stdin)
721 fclose(patterns);
722 strbuf_release(&sb);
723 return 0;
724}

762int cmd_grep(int argc, const char **argv, const char *prefix)
763{
764 int hit = 0;
765 int cached = 0, untracked = 0, opt_exclude = -1;
766 int seen_dashdash = 0;
767 int external_grep_allowed__ignored;
768 const char *show_in_pager = NULL, *default_pager = "dummy";
769 struct grep_opt opt;
770 struct object_array list = OBJECT_ARRAY_INIT;
771 struct pathspec pathspec;

891 init_grep_defaults();
892 git_config(grep_cmd_config, NULL);
893 grep_init(&opt, prefix);
894
895 /*
896 * If there is no -- then the paths must exist in the working
897 * tree. If there is no explicit pattern specified with -e or
898 * -f, we take the first unrecognized non option to be the
899 * pattern, but then what follows it must be zero or more
900 * valid refs up to the -- (if exists), and then existing
901 * paths. If there is an explicit pattern, then the first
902 * unrecognized non option is the beginning of the refs list
903 * that continues up to the -- (if exists), and then paths.
904 */
905 argc = parse_options(argc, argv, prefix, options, grep_usage,
906 PARSE_OPT_KEEP_DASHDASH |
907 PARSE_OPT_STOP_AT_NON_OPTION);
908 grep_commit_pattern_type(pattern_type_arg, &opt);
909
910 if (use_index && !startup_info->have_repository) {
911 int fallback = 0;
912 git_config_get_bool("grep.fallbacktonoindex", &fallback);
913 if (fallback)
914 use_index = 0;
915 else
916 /* die the same way as if we did it at the beginning */
917 setup_git_directory();
918 }
919
920 /*
921 * skip a -- separator; we know it cannot be
922 * separating revisions from pathnames if
923 * we haven't even had any patterns yet
924 */
925 if (argc > 0 && !opt.pattern_list && !strcmp(argv[0], "--")) {
926 argv++;
927 argc--;
928 }
929
930 /* First unrecognized non-option token */
931 if (argc > 0 && !opt.pattern_list) {
932 append_grep_pattern(&opt, argv[0], "command line", 0,
933 GREP_PATTERN);
934 argv++;
935 argc--;
936 }
937
938 if (show_in_pager == default_pager)
939 show_in_pager = git_pager(1);
940 if (show_in_pager) {
941 opt.color = 0;
942 opt.name_only = 1;
943 opt.null_following_name = 1;
944 opt.output_priv = &path_list;
945 opt.output = append_path;
946 string_list_append(&path_list, show_in_pager);
947 }
948
949 if (!opt.pattern_list)
950 die(_("no pattern given."));
951
952 /*
953 * We have to find "--" in a separate pass, because its presence
954 * influences how we will parse arguments that come before it.
955 */
956 for (i = 0; i < argc; i++) {
957 if (!strcmp(argv[i], "--")) {
958 seen_dashdash = 1;
959 break;
960 }
961 }
962
963 /*
964 * Resolve any rev arguments. If we have a dashdash, then everything up
965 * to it must resolve as a rev. If not, then we stop at the first
966 * non-rev and assume everything else is a path.
967 */
968 allow_revs = use_index && !untracked;
969 for (i = 0; i < argc; i++) {
970 const char *arg = argv[i];
971 struct object_id oid;
972 struct object_context oc;
973 struct object *object;
974
975 if (!strcmp(arg, "--")) {
976 i++;
977 break;
978 }
979
980 if (!allow_revs) {
981 if (seen_dashdash)
982 die(_("--no-index or --untracked cannot be used with revs"));
983 break;
984 }
985
986 if (get_oid_with_context(arg, GET_OID_RECORD_PATH,
987 &oid, &oc)) {
988 if (seen_dashdash)
989 die(_("unable to resolve revision: %s"), arg);
990 break;
991 }
992
993 object = parse_object_or_die(&oid, arg);
994 if (!seen_dashdash)
995 verify_non_filename(prefix, arg);
996 add_object_array_with_path(object, arg, &list, oc.mode, oc.path);
997 free(oc.path);
998 }
999
1000 /*
1001 * Anything left over is presumed to be a path. But in the non-dashdash
1002 * "do what I mean" case, we verify and complain when that isn't true.
1003 */
1004 if (!seen_dashdash) {
1005 int j;
1006 for (j = i; j < argc; j++)
1007 verify_filename(prefix, argv[j], j == i && allow_revs);
1008 }
1009
1010 parse_pathspec(&pathspec, 0,
1011 PATHSPEC_PREFER_CWD |
1012 (opt.max_depth != -1 ? PATHSPEC_MAXDEPTH_VALID : 0),
1013 prefix, argv + i);
1014 pathspec.max_depth = opt.max_depth;
1015 pathspec.recursive = 1;
1016
1017#ifndef NO_PTHREADS
1018 if (list.nr || cached || show_in_pager)
1019 num_threads = 0;
1020 else if (num_threads == 0)
1021 num_threads = GREP_NUM_THREADS_DEFAULT;
1022 else if (num_threads < 0)
1023 die(_("invalid number of threads specified (%d)"), num_threads);
1024 if (num_threads == 1)
1025 num_threads = 0;
1026#else
1027 if (num_threads)
1028 warning(_("no threads support, ignoring --threads"));
1029 num_threads = 0;
1030#endif
1031
1032 if (!num_threads)
1033 /*
1034 * The compiled patterns on the main path are only
1035 * used when not using threading. Otherwise
1036 * start_threads() below calls compile_grep_patterns()
1037 * for each thread.
1038 */
1039 compile_grep_patterns(&opt);
1040
1041#ifndef NO_PTHREADS
1042 if (num_threads) {
1043 if (!(opt.name_only || opt.unmatch_name_only || opt.count)
1044 && (opt.pre_context || opt.post_context ||
1045 opt.file_break || opt.funcbody))
1046 skip_first_line = 1;
1047 start_threads(&opt);
1048 }
1049#endif
1050
1051 if (recurse_submodules) {
1052 gitmodules_config();
1053 }
1054
1055 if (show_in_pager && (cached || list.nr))
1056 die(_("--open-files-in-pager only works on the worktree"));
1057
1058 if (show_in_pager && opt.pattern_list && !opt.pattern_list->next) {
1059 const char *pager = path_list.items[0].string;
1060 int len = strlen(pager);
1061
1062 if (len > 4 && is_dir_sep(pager[len - 5]))
1063 pager += len - 4;
1064
1065 if (opt.ignore_case && !strcmp("less", pager))
1066 string_list_append(&path_list, "-I");
1067
1068 if (!strcmp("less", pager) || !strcmp("vi", pager)) {
1069 struct strbuf buf = STRBUF_INIT;
1070 strbuf_addf(&buf, "+/%s%s",
1071 strcmp("less", pager) ? "" : "*",
1072 opt.pattern_list->pattern);
1073 string_list_append(&path_list, buf.buf);
1074 strbuf_detach(&buf, NULL);
1075 }
1076 }
1077
1078 if (recurse_submodules && (!use_index || untracked))
1079 die(_("option not supported with --recurse-submodules."));
1080
1081 if (!show_in_pager && !opt.status_only)
1082 setup_pager();
1083
1084 if (!use_index && (untracked || cached))
1085 die(_("--cached or --untracked cannot be used with --no-index."));
1086
1087 if (!use_index || untracked) {
1088 int use_exclude = (opt_exclude < 0) ? use_index : !!opt_exclude;
1089 hit = grep_directory(&opt, &pathspec, use_exclude, use_index);
1090 } else if (0 <= opt_exclude) {
1091 die(_("--[no-]exclude-standard cannot be used for tracked contents."));
1092 } else if (!list.nr) {
1093 if (!cached)
1094 setup_work_tree();
1095
1096 hit = grep_cache(&opt, the_repository, &pathspec, cached);
1097 } else {
1098 if (cached)
1099 die(_("both --cached and trees are given."));
1100
1101 hit = grep_objects(&opt, &pathspec, the_repository, &list);
1102 }
1103
1104 if (num_threads)
1105 hit |= wait_all();
1106 if (hit && show_in_pager)
1107 run_pager(&opt, prefix);
1108 clear_pathspec(&pathspec);
1109 free_grep_patterns(&opt);
1110 return !hit;
1111}
