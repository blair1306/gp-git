17static char *get_default_remote(void)
18{
19 char *dest = NULL, *ret;
20 unsigned char sha1[20];
21 struct strbuf sb = STRBUF_INIT;
22 const char *refname = resolve_ref_unsafe("HEAD", 0, sha1, NULL);
23
24 if (!refname)
25 die(_("No such ref: %s"), "HEAD");
26
27 /* detached HEAD */
28 if (!strcmp(refname, "HEAD"))
29 return xstrdup("origin");
30
31 if (!skip_prefix(refname, "refs/heads/", &refname))
32 die(_("Expecting a full ref name, got %s"), refname);
33
34 strbuf_addf(&sb, "branch.%s.remote", refname);
35 if (git_config_get_string(sb.buf, &dest))
36 ret = xstrdup("origin");
37 else
38 ret = dest;
39
40 strbuf_release(&sb);
41 return ret;
42}

54/*
55 * Returns 1 if it was the last chop before ':'.
56 */
57static int chop_last_dir(char **remoteurl, int is_relative)
58{
59 char *rfind = find_last_dir_sep(*remoteurl);
60 if (rfind) {
61 *rfind = '\0';
62 return 0;
63 }
64
65 rfind = strrchr(*remoteurl, ':');
66 if (rfind) {
67 *rfind = '\0';
68 return 1;
69 }
70
71 if (is_relative || !strcmp(".", *remoteurl))
72 die(_("cannot strip one component off url '%s'"),
73 *remoteurl);
74
75 free(*remoteurl);
76 *remoteurl = xstrdup(".");
77 return 0;
78}

80/*
81 * The `url` argument is the URL that navigates to the submodule origin
82 * repo. When relative, this URL is relative to the superproject origin
83 * URL repo. The `up_path` argument, if specified, is the relative
84 * path that navigates from the submodule working tree to the superproject
85 * working tree. Returns the origin URL of the submodule.
86 *
87 * Return either an absolute URL or filesystem path (if the superproject
88 * origin URL is an absolute URL or filesystem path, respectively) or a
89 * relative file system path (if the superproject origin URL is a relative
90 * file system path).
91 *
92 * When the output is a relative file system path, the path is either
93 * relative to the submodule working tree, if up_path is specified, or to
94 * the superproject working tree otherwise.
95 *
96 * NEEDSWORK: This works incorrectly on the domain and protocol part.
97 * remote_url url outcome expectation
98 * http://a.com/b ../c http://a.com/c as is
99 * http://a.com/b/ ../c http://a.com/c same as previous line, but
100 * ignore trailing slash in url
101 * http://a.com/b ../../c http://c error out
102 * http://a.com/b ../../../c http:/c error out
103 * http://a.com/b ../../../../c http:c error out
104 * http://a.com/b ../../../../../c .:c error out
105 * NEEDSWORK: Given how chop_last_dir() works, this function is broken
106 * when a local part has a colon in its path component, too.
107 */
108static char *relative_url(const char *remote_url,
109 const char *url,
110 const char *up_path)
111{
112 int is_relative = 0;
113 int colonsep = 0;
114 char *out;
115 char *remoteurl = xstrdup(remote_url);
116 struct strbuf sb = STRBUF_INIT;
117 size_t len = strlen(remoteurl);
118
119 if (is_dir_sep(remoteurl[len-1]))
120 remoteurl[len-1] = '\0';
121
122 if (!url_is_local_not_ssh(remoteurl) || is_absolute_path(remoteurl))
123 is_relative = 0;
124 else {
125 is_relative = 1;
126 /*
127 * Prepend a './' to ensure all relative
128 * remoteurls start with './' or '../'
129 */
130 if (!starts_with_dot_slash(remoteurl) &&
131 !starts_with_dot_dot_slash(remoteurl)) {
132 strbuf_reset(&sb);
133 strbuf_addf(&sb, "./%s", remoteurl);
134 free(remoteurl);
135 remoteurl = strbuf_detach(&sb, NULL);
136 }
137 }
138 /*
139 * When the url starts with '../', remove that and the
140 * last directory in remoteurl.
141 */
142 while (url) {
143 if (starts_with_dot_dot_slash(url)) {
144 url += 3;
145 colonsep |= chop_last_dir(&remoteurl, is_relative);
146 } else if (starts_with_dot_slash(url))
147 url += 2;
148 else
149 break;
150 }
151 strbuf_reset(&sb);
152 strbuf_addf(&sb, "%s%s%s", remoteurl, colonsep ? ":" : "/", url);
153 if (ends_with(url, "/"))
154 strbuf_setlen(&sb, sb.len - 1);
155 free(remoteurl);
156
157 if (starts_with_dot_slash(sb.buf))
158 out = xstrdup(sb.buf + 2);
159 else
160 out = xstrdup(sb.buf);
161 strbuf_reset(&sb);
162
163 if (!up_path || !is_relative)
164 return out;
165
166 strbuf_addf(&sb, "%s%s", up_path, out);
167 free(out);
168 return strbuf_detach(&sb, NULL);
169}

171static int resolve_relative_url(int argc, const char **argv, const char *prefix)
172{
173 char *remoteurl = NULL;
174 char *remote = get_default_remote();
175 const char *up_path = NULL;
176 char *res;
177 const char *url;
178 struct strbuf sb = STRBUF_INIT;
179
180 if (argc != 2 && argc != 3)
181 die("resolve-relative-url only accepts one or two arguments");
182
183 url = argv[1];
184 strbuf_addf(&sb, "remote.%s.url", remote);
185 free(remote);
186
187 if (git_config_get_string(sb.buf, &remoteurl))
188 /* the repository is its own authoritative upstream */
189 remoteurl = xgetcwd();
190
191 if (argc == 3)
192 up_path = argv[2];
193
194 res = relative_url(remoteurl, url, up_path);
195 puts(res);
196 free(res);
197 free(remoteurl);
198 return 0;
199}

229static int module_list_compute(int argc, const char **argv,
230 const char *prefix,
231 struct pathspec *pathspec,
232 struct module_list *list)
233{
234 int i, result = 0;
235 char *ps_matched = NULL;
236 parse_pathspec(pathspec, 0,
237 PATHSPEC_PREFER_FULL,
238 prefix, argv);
239
240 if (pathspec->nr)
241 ps_matched = xcalloc(pathspec->nr, 1);
242
243 if (read_cache() < 0)
244 die(_("index file corrupt"));
245
246 for (i = 0; i < active_nr; i++) {
247 const struct cache_entry *ce = active_cache[i];
248
249 if (!match_pathspec(pathspec, ce->name, ce_namelen(ce),
250 0, ps_matched, 1) ||
251 !S_ISGITLINK(ce->ce_mode))
252 continue;
253
254 ALLOC_GROW(list->entries, list->nr + 1, list->alloc);
255 list->entries[list->nr++] = ce;
256 while (i + 1 < active_nr &&
257 !strcmp(ce->name, active_cache[i + 1]->name))
258 /*
259 * Skip entries with the same name in different stages
260 * to make sure an entry is returned only once.
261 */
262 i++;
263 }
264
265 if (ps_matched && report_path_error(ps_matched, pathspec, prefix))
266 result = -1;
267
268 free(ps_matched);
269
270 return result;
271}

273static void module_list_active(struct module_list *list)
274{
275 int i;
276 struct module_list active_modules = MODULE_LIST_INIT;
277
278 gitmodules_config();
279
280 for (i = 0; i < list->nr; i++) {
281 const struct cache_entry *ce = list->entries[i];
282
283 if (!is_submodule_active(the_repository, ce->name))

286 ALLOC_GROW(active_modules.entries,

291
292 free(list->entries);
293 *list = active_modules;
294}

334static void init_submodule(const char *path, const char *prefix, int quiet)
335{
336 const struct submodule *sub;
337 struct strbuf sb = STRBUF_INIT;
338 char *upd = NULL, *url = NULL, *displaypath;
339
340 /* Only loads from .gitmodules, no overlay with .git/config */
341 gitmodules_config();
342
343 if (prefix && get_super_prefix())
344 die("BUG: cannot have prefix and superprefix");
345 else if (prefix)
346 displaypath = xstrdup(relative_path(path, prefix, &sb));
347 else if (get_super_prefix()) {
348 strbuf_addf(&sb, "%s%s", get_super_prefix(), path);
349 displaypath = strbuf_detach(&sb, NULL);
350 } else
351 displaypath = xstrdup(path);
352
353 sub = submodule_from_path(&null_oid, path);
354
355 if (!sub)
356 die(_("No url found for submodule path '%s' in .gitmodules"),
357 displaypath);
358
359 /*
360 * NEEDSWORK: In a multi-working-tree world, this needs to be
361 * set in the per-worktree config.
362 *
363 * Set active flag for the submodule being initialized
364 */
365 if (!is_submodule_active(the_repository, path)) {
366 strbuf_reset(&sb);
367 strbuf_addf(&sb, "submodule.%s.active", sub->name);
368 git_config_set_gently(sb.buf, "true");
369 }
370
371 /*
372 * Copy url setting when it is not set yet.
373 * To look up the url in .git/config, we must not fall back to
374 * .gitmodules, so look it up directly.
375 */
376 strbuf_reset(&sb);
377 strbuf_addf(&sb, "submodule.%s.url", sub->name);
378 if (git_config_get_string(sb.buf, &url)) {
379 if (!sub->url)
380 die(_("No url found for submodule path '%s' in .gitmodules"),
381 displaypath);
382
383 url = xstrdup(sub->url);
384
385 /* Possibly a url relative to parent */
386 if (starts_with_dot_dot_slash(url) ||
387 starts_with_dot_slash(url)) {
388 char *remoteurl, *relurl;
389 char *remote = get_default_remote();
390 struct strbuf remotesb = STRBUF_INIT;
391 strbuf_addf(&remotesb, "remote.%s.url", remote);
392 free(remote);
393
394 if (git_config_get_string(remotesb.buf, &remoteurl)) {
395 warning(_("could not lookup configuration '%s'. Assuming this repository is its own authoritative upstream."), remotesb.buf);
396 remoteurl = xgetcwd();
397 }
398 relurl = relative_url(remoteurl, url, NULL);
399 strbuf_release(&remotesb);
400 free(remoteurl);
401 free(url);
402 url = relurl;
403 }
404
405 if (git_config_set_gently(sb.buf, url))
406 die(_("Failed to register url for submodule path '%s'"),
407 displaypath);
408 if (!quiet)
409 fprintf(stderr,
410 _("Submodule '%s' (%s) registered for path '%s'\n"),
411 sub->name, url, displaypath);
412 }
413
414 /* Copy "update" setting when it is not set yet */
415 strbuf_reset(&sb);
416 strbuf_addf(&sb, "submodule.%s.update", sub->name);
417 if (git_config_get_string(sb.buf, &upd) &&
418 sub->update_strategy.type != SM_UPDATE_UNSPECIFIED) {
419 if (sub->update_strategy.type == SM_UPDATE_COMMAND) {
420 fprintf(stderr, _("warning: command update mode suggested for submodule '%s'\n"),
421 sub->name);
422 upd = xstrdup("none");
423 } else
424 upd = xstrdup(submodule_strategy_to_string(&sub->update_strategy));
425
426 if (git_config_set_gently(sb.buf, upd))
427 die(_("Failed to register update mode for submodule path '%s'"), displaypath);
428 }
429 strbuf_release(&sb);
430 free(displaypath);
431 free(url);
432 free(upd);
433}

435static int module_init(int argc, const char **argv, const char *prefix)
436{
437 struct pathspec pathspec;
438 struct module_list list = MODULE_LIST_INIT;
439 int quiet = 0;
440 int i;
441
442 struct option module_init_options[] = {
443 OPT__QUIET(&quiet, N_("Suppress output for initializing a submodule")),
444 OPT_END()
445 };
446
447 const char *const git_submodule_helper_usage[] = {
448 N_("git submodule--helper init [<path>]"),
449 NULL
450 };
451
452 argc = parse_options(argc, argv, prefix, module_init_options,
453 git_submodule_helper_usage, 0);
454
455 if (module_list_compute(argc, argv, prefix, &pathspec, &list) < 0)
456 return 1;
457
458 /*
459 * If there are no path args and submodule.active is set then,
460 * by default, only initialize 'active' modules.
461 */
462 if (!argc && git_config_get_value_multi("submodule.active"))
463 module_list_active(&list);
464
465 for (i = 0; i < list.nr; i++)
466 init_submodule(list.entries[i]->name, prefix, quiet);
467
468 return 0;
469}

490static int clone_submodule(const char *path, const char *gitdir, const char *url,
491 const char *depth, struct string_list *reference,
492 int quiet, int progress)
493{
494 struct child_process cp = CHILD_PROCESS_INIT;
495
496 argv_array_push(&cp.args, "clone");
497 argv_array_push(&cp.args, "--no-checkout");
498 if (quiet)
499 argv_array_push(&cp.args, "--quiet");
500 if (progress)
501 argv_array_push(&cp.args, "--progress");
502 if (depth && *depth)
503 argv_array_pushl(&cp.args, "--depth", depth, NULL);
504 if (reference->nr) {
505 struct string_list_item *item;
506 for_each_string_list_item(item, reference)
507 argv_array_pushl(&cp.args, "--reference",
508 item->string, NULL);
509 }
510 if (gitdir && *gitdir)
511 argv_array_pushl(&cp.args, "--separate-git-dir", gitdir, NULL);
512
513 argv_array_push(&cp.args, url);
514 argv_array_push(&cp.args, path);
515
516 cp.git_cmd = 1;
517 prepare_submodule_repo_env(&cp.env_array);
518 cp.no_stdin = 1;
519
520 return run_command(&cp);
521}

535static int add_possible_reference_from_superproject(
536 struct alternate_object_database *alt, void *sas_cb)
537{
538 struct submodule_alternate_setup *sas = sas_cb;
539
540 /*
541 * If the alternate object store is another repository, try the
542 * standard layout with .git/(modules/<name>)+/objects
543 */
544 if (ends_with(alt->path, "/objects")) {
545 char *sm_alternate;
546 struct strbuf sb = STRBUF_INIT;
547 struct strbuf err = STRBUF_INIT;
548 strbuf_add(&sb, alt->path, strlen(alt->path) - strlen("objects"));
549
550 /*
551 * We need to end the new path with '/' to mark it as a dir,
552 * otherwise a submodule name containing '/' will be broken
553 * as the last part of a missing submodule reference would
554 * be taken as a file name.
555 */
556 strbuf_addf(&sb, "modules/%s/", sas->submodule_name);
557
558 sm_alternate = compute_alternate_path(sb.buf, &err);
559 if (sm_alternate) {
560 string_list_append(sas->reference, xstrdup(sb.buf));
561 free(sm_alternate);
562 } else {
563 switch (sas->error_mode) {
564 case SUBMODULE_ALTERNATE_ERROR_DIE:
565 die(_("submodule '%s' cannot add alternate: %s"),
566 sas->submodule_name, err.buf);
567 case SUBMODULE_ALTERNATE_ERROR_INFO:
568 fprintf(stderr, _("submodule '%s' cannot add alternate: %s"),
569 sas->submodule_name, err.buf);
570 case SUBMODULE_ALTERNATE_ERROR_IGNORE:
571 ; /* nothing */
572 }
573 }
574 strbuf_release(&sb);
575 }
576
577 return 0;
578}

580static void prepare_possible_alternates(const char *sm_name,
581 struct string_list *reference)
582{
583 char *sm_alternate = NULL, *error_strategy = NULL;
584 struct submodule_alternate_setup sas = SUBMODULE_ALTERNATE_SETUP_INIT;
585
586 git_config_get_string("submodule.alternateLocation", &sm_alternate);
587 if (!sm_alternate)
588 return;
589
590 git_config_get_string("submodule.alternateErrorStrategy", &error_strategy);
591
592 if (!error_strategy)
593 error_strategy = xstrdup("die");
594
595 sas.submodule_name = sm_name;
596 sas.reference = reference;
597 if (!strcmp(error_strategy, "die"))
598 sas.error_mode = SUBMODULE_ALTERNATE_ERROR_DIE;
599 else if (!strcmp(error_strategy, "info"))
600 sas.error_mode = SUBMODULE_ALTERNATE_ERROR_INFO;
601 else if (!strcmp(error_strategy, "ignore"))
602 sas.error_mode = SUBMODULE_ALTERNATE_ERROR_IGNORE;
603 else
604 die(_("Value '%s' for submodule.alternateErrorStrategy is not recognized"), error_strategy);
605
606 if (!strcmp(sm_alternate, "superproject"))
607 foreach_alt_odb(add_possible_reference_from_superproject, &sas);
608 else if (!strcmp(sm_alternate, "no"))
609 ; /* do nothing */
610 else
611 die(_("Value '%s' for submodule.alternateLocation is not recognized"), sm_alternate);
612
613 free(sm_alternate);
614 free(error_strategy);
615}

617static int module_clone(int argc, const char **argv, const char *prefix)
618{
619 const char *name = NULL, *url = NULL, *depth = NULL;
620 int quiet = 0;
621 int progress = 0;
622 char *p, *path = NULL, *sm_gitdir;
623 struct strbuf sb = STRBUF_INIT;
624 struct string_list reference = STRING_LIST_INIT_NODUP;
625 char *sm_alternate = NULL, *error_strategy = NULL;
626
627 struct option module_clone_options[] = {
628 OPT_STRING(0, "prefix", &prefix,
629 N_("path"),
630 N_("alternative anchor for relative paths")),
631 OPT_STRING(0, "path", &path,
632 N_("path"),
633 N_("where the new submodule will be cloned to")),
634 OPT_STRING(0, "name", &name,
635 N_("string"),
636 N_("name of the new submodule")),
637 OPT_STRING(0, "url", &url,
638 N_("string"),
639 N_("url where to clone the submodule from")),
640 OPT_STRING_LIST(0, "reference", &reference,
641 N_("repo"),
642 N_("reference repository")),
643 OPT_STRING(0, "depth", &depth,
644 N_("string"),
645 N_("depth for shallow clones")),
646 OPT__QUIET(&quiet, "Suppress output for cloning a submodule"),
647 OPT_BOOL(0, "progress", &progress,
648 N_("force cloning progress")),
649 OPT_END()
650 };
651
652 const char *const git_submodule_helper_usage[] = {
653 N_("git submodule--helper clone [--prefix=<path>] [--quiet] "
654 "[--reference <repository>] [--name <name>] [--depth <depth>] "
655 "--url <url> --path <path>"),
656 NULL
657 };
658
659 argc = parse_options(argc, argv, prefix, module_clone_options,
660 git_submodule_helper_usage, 0);
661
662 if (argc || !url || !path || !*path)
663 usage_with_options(git_submodule_helper_usage,
664 module_clone_options);
665
666 strbuf_addf(&sb, "%s/modules/%s", get_git_dir(), name);
667 sm_gitdir = absolute_pathdup(sb.buf);
668 strbuf_reset(&sb);
669
670 if (!is_absolute_path(path)) {
671 strbuf_addf(&sb, "%s/%s", get_git_work_tree(), path);
672 path = strbuf_detach(&sb, NULL);
673 } else
674 path = xstrdup(path);
675
676 if (!file_exists(sm_gitdir)) {
677 if (safe_create_leading_directories_const(sm_gitdir) < 0)
678 die(_("could not create directory '%s'"), sm_gitdir);
679
680 prepare_possible_alternates(name, &reference);
681
682 if (clone_submodule(path, sm_gitdir, url, depth, &reference,
683 quiet, progress))
684 die(_("clone of '%s' into submodule path '%s' failed"),
685 url, path);
686 } else {
687 if (safe_create_leading_directories_const(path) < 0)
688 die(_("could not create directory '%s'"), path);
689 strbuf_addf(&sb, "%s/index", sm_gitdir);
690 unlink_or_warn(sb.buf);
691 strbuf_reset(&sb);
692 }
693
694 /* Connect module worktree and git dir */
695 connect_work_tree_and_git_dir(path, sm_gitdir);
696
697 p = git_pathdup_submodule(path, "config");
698 if (!p)
699 die(_("could not get submodule directory for '%s'"), path);
700
701 /* setup alternateLocation and alternateErrorStrategy in the cloned submodule if needed */
702 git_config_get_string("submodule.alternateLocation", &sm_alternate);
703 if (sm_alternate)
704 git_config_set_in_file(p, "submodule.alternateLocation",
705 sm_alternate);
706 git_config_get_string("submodule.alternateErrorStrategy", &error_strategy);
707 if (error_strategy)
708 git_config_set_in_file(p, "submodule.alternateErrorStrategy",
709 error_strategy);
710
711 free(sm_alternate);
712 free(error_strategy);
713
714 strbuf_release(&sb);
715 free(sm_gitdir);
716 free(path);
717 free(p);
718 return 0;
719}

721struct submodule_update_clone {
722 /* index into 'list', the list of submodules to look into for cloning */
723 int current;
724 struct module_list list;

773/**
774 * Determine whether 'ce' needs to be cloned. If so, prepare the 'child' to
775 * run the clone. Returns 1 if 'ce' needs to be cloned, 0 otherwise.
776 */
777static int prepare_to_clone_next_submodule(const struct cache_entry *ce,
778 struct child_process *child,
779 struct submodule_update_clone *suc,
780 struct strbuf *out)
781{
782 const struct submodule *sub = NULL;
783 struct strbuf displaypath_sb = STRBUF_INIT;
784 struct strbuf sb = STRBUF_INIT;
785 const char *displaypath = NULL;
786 int needs_cloning = 0;
787
788 if (ce_stage(ce)) {
789 if (suc->recursive_prefix)
790 strbuf_addf(&sb, "%s/%s", suc->recursive_prefix, ce->name);
791 else
792 strbuf_addstr(&sb, ce->name);
793 strbuf_addf(out, _("Skipping unmerged submodule %s"), sb.buf);
794 strbuf_addch(out, '\n');
795 goto cleanup;
796 }
797
798 sub = submodule_from_path(&null_oid, ce->name);
799
800 if (suc->recursive_prefix)
801 displaypath = relative_path(suc->recursive_prefix,
802 ce->name, &displaypath_sb);
803 else
804 displaypath = ce->name;
805
806 if (!sub) {
807 next_submodule_warn_missing(suc, out, displaypath);
808 goto cleanup;
809 }
810
811 if (suc->update.type == SM_UPDATE_NONE
812 || (suc->update.type == SM_UPDATE_UNSPECIFIED
813 && sub->update_strategy.type == SM_UPDATE_NONE)) {
814 strbuf_addf(out, _("Skipping submodule '%s'"), displaypath);
815 strbuf_addch(out, '\n');
816 goto cleanup;
817 }
818
819 /* Check if the submodule has been initialized. */
820 if (!is_submodule_active(the_repository, ce->name)) {
821 next_submodule_warn_missing(suc, out, displaypath);
822 goto cleanup;
823 }
824
825 strbuf_reset(&sb);
826 strbuf_addf(&sb, "%s/.git", ce->name);
827 needs_cloning = !file_exists(sb.buf);
828
829 strbuf_reset(&sb);
830 strbuf_addf(&sb, "%06o %s %d %d\t%s\n", ce->ce_mode,
831 oid_to_hex(&ce->oid), ce_stage(ce),
832 needs_cloning, ce->name);
833 string_list_append(&suc->projectlines, sb.buf);
834
835 if (!needs_cloning)
836 goto cleanup;
837
838 child->git_cmd = 1;
839 child->no_stdin = 1;
840 child->stdout_to_stderr = 1;
841 child->err = -1;
842 argv_array_push(&child->args, "submodule--helper");
843 argv_array_push(&child->args, "clone");
844 if (suc->progress)
845 argv_array_push(&child->args, "--progress");
846 if (suc->quiet)
847 argv_array_push(&child->args, "--quiet");
848 if (suc->prefix)
849 argv_array_pushl(&child->args, "--prefix", suc->prefix, NULL);
850 if (suc->recommend_shallow && sub->recommend_shallow == 1)
851 argv_array_push(&child->args, "--depth=1");
852 argv_array_pushl(&child->args, "--path", sub->path, NULL);
853 argv_array_pushl(&child->args, "--name", sub->name, NULL);
854 argv_array_pushl(&child->args, "--url", sub->url, NULL);
855 if (suc->references.nr) {
856 struct string_list_item *item;
857 for_each_string_list_item(item, &suc->references)
858 argv_array_pushl(&child->args, "--reference", item->string, NULL);
859 }
860 if (suc->depth)
861 argv_array_push(&child->args, suc->depth);
862
863cleanup:
864 strbuf_reset(&displaypath_sb);
865 strbuf_reset(&sb);
866
867 return needs_cloning;
868}

870static int update_clone_get_next_task(struct child_process *child,
871 struct strbuf *err,
872 void *suc_cb,
873 void **idx_task_cb)
874{
875 struct submodule_update_clone *suc = suc_cb;
876 const struct cache_entry *ce;
877 int index;
878
879 for (; suc->current < suc->list.nr; suc->current++) {
880 ce = suc->list.entries[suc->current];
881 if (prepare_to_clone_next_submodule(ce, child, suc, err)) {
882 int *p = xmalloc(sizeof(*p));
883 *p = suc->current;
884 *idx_task_cb = p;
885 suc->current++;
886 return 1;
887 }
888 }
889
890 /*
891 * The loop above tried cloning each submodule once, now try the
892 * stragglers again, which we can imagine as an extension of the
893 * entry list.
894 */
895 index = suc->current - suc->list.nr;
896 if (index < suc->failed_clones_nr) {
897 int *p;
898 ce = suc->failed_clones[index];
899 if (!prepare_to_clone_next_submodule(ce, child, suc, err)) {
900 suc->current ++;
901 strbuf_addstr(err, "BUG: submodule considered for "
902 "cloning, doesn't need cloning "
903 "any more?\n");
904 return 0;
905 }
906 p = xmalloc(sizeof(*p));
907 *p = suc->current;
908 *idx_task_cb = p;
909 suc->current ++;
910 return 1;
911 }
912
913 return 0;
914}

925static int update_clone_task_finished(int result,
926 struct strbuf *err,
927 void *suc_cb,
928 void *idx_task_cb)
929{
930 const struct cache_entry *ce;
931 struct submodule_update_clone *suc = suc_cb;
932
933 int *idxP = idx_task_cb;
934 int idx = *idxP;
935 free(idxP);
936
937 if (!result)
938 return 0;
939
940 if (idx < suc->list.nr) {
941 ce = suc->list.entries[idx];
942 strbuf_addf(err, _("Failed to clone '%s'. Retry scheduled"),
943 ce->name);
944 strbuf_addch(err, '\n');
945 ALLOC_GROW(suc->failed_clones,
946 suc->failed_clones_nr + 1,
947 suc->failed_clones_alloc);
948 suc->failed_clones[suc->failed_clones_nr++] = ce;
949 return 0;
950 } else {
951 idx -= suc->list.nr;
952 ce = suc->failed_clones[idx];
953 strbuf_addf(err, _("Failed to clone '%s' a second time, aborting"),
954 ce->name);
955 strbuf_addch(err, '\n');
956 suc->quickstop = 1;
957 return 1;
958 }
959
960 return 0;
961}

972static int update_clone(int argc, const char **argv, const char *prefix)
973{
974 const char *update = NULL;
975 int max_jobs = 1;
976 struct string_list_item *item;
977 struct pathspec pathspec;
978 struct submodule_update_clone suc = SUBMODULE_UPDATE_CLONE_INIT;
979
980 struct option module_update_clone_options[] = {
981 OPT_STRING(0, "prefix", &prefix,
982 N_("path"),
983 N_("path into the working tree")),
984 OPT_STRING(0, "recursive-prefix", &suc.recursive_prefix,
985 N_("path"),
986 N_("path into the working tree, across nested "
987 "submodule boundaries")),
988 OPT_STRING(0, "update", &update,
989 N_("string"),
990 N_("rebase, merge, checkout or none")),
991 OPT_STRING_LIST(0, "reference", &suc.references, N_("repo"),
992 N_("reference repository")),
993 OPT_STRING(0, "depth", &suc.depth, "<depth>",
994 N_("Create a shallow clone truncated to the "
995 "specified number of revisions")),
996 OPT_INTEGER('j', "jobs", &max_jobs,
997 N_("parallel jobs")),
998 OPT_BOOL(0, "recommend-shallow", &suc.recommend_shallow,
999 N_("whether the initial clone should follow the shallow recommendation")),
1000 OPT__QUIET(&suc.quiet, N_("don't print cloning progress")),
1001 OPT_BOOL(0, "progress", &suc.progress,
1002 N_("force cloning progress")),
1003 OPT_END()
1004 };
1005
1006 const char *const git_submodule_helper_usage[] = {
1007 N_("git submodule--helper update_clone [--prefix=<path>] [<path>...]"),
1008 NULL
1009 };
1010 suc.prefix = prefix;
1011
1012 config_from_gitmodules(gitmodules_update_clone_config, &max_jobs);
1013 git_config(gitmodules_update_clone_config, &max_jobs);
1014
1015 argc = parse_options(argc, argv, prefix, module_update_clone_options,
1016 git_submodule_helper_usage, 0);
1017
1018 if (update)
1019 if (parse_submodule_update_strategy(update, &suc.update) < 0)
1020 die(_("bad value for update parameter"));
1021
1022 if (module_list_compute(argc, argv, prefix, &pathspec, &suc.list) < 0)
1023 return 1;
1024
1025 if (pathspec.nr)
1026 suc.warn_if_uninitialized = 1;
1027
1028 /* Overlay the parsed .gitmodules file with .git/config */
1029 gitmodules_config();
1030 git_config(submodule_config, NULL);
1031
1032 run_processes_parallel(max_jobs,
1033 update_clone_get_next_task,
1034 update_clone_start_failure,
1035 update_clone_task_finished,
1036 &suc);
1037
1038 /*
1039 * We saved the output and put it out all at once now.
1040 * That means:
1041 * - the listener does not have to interleave their (checkout)
1042 * work with our fetching. The writes involved in a
1043 * checkout involve more straightforward sequential I/O.
1044 * - the listener can avoid doing any work if fetching failed.
1045 */
1046 if (suc.quickstop)
1047 return 1;
1048
1049 for_each_string_list_item(item, &suc.projectlines)
1050 fprintf(stdout, "%s", item->string);
1051
1052 return 0;
1053}

1066static const char *remote_submodule_branch(const char *path)
1067{
1068 const struct submodule *sub;
1069 gitmodules_config();
1070 git_config(submodule_config, NULL);
1071
1072 sub = submodule_from_path(&null_oid, path);
1073 if (!sub)
1074 return NULL;
1075
1076 if (!sub->branch)
1077 return "master";
1078
1079 if (!strcmp(sub->branch, ".")) {
1080 unsigned char sha1[20];
1081 const char *refname = resolve_ref_unsafe("HEAD", 0, sha1, NULL);
1082
1083 if (!refname)
1084 die(_("No such ref: %s"), "HEAD");
1085
1086 /* detached HEAD */
1087 if (!strcmp(refname, "HEAD"))
1088 die(_("Submodule (%s) branch configured to inherit "
1089 "branch from superproject, but the superproject "
1090 "is not on any branch"), sub->name);
1091
1092 if (!skip_prefix(refname, "refs/heads/", &refname))
1093 die(_("Expecting a full ref name, got %s"), refname);
1094 return refname;
1095 }
1096
1097 return sub->branch;
1098}

1117static int push_check(int argc, const char **argv, const char *prefix)
1118{
1119 struct remote *remote;
1120 const char *superproject_head;
1121 char *head;
1122 int detached_head = 0;
1123 struct object_id head_oid;
1124
1125 if (argc < 3)
1126 die("submodule--helper push-check requires at least 2 arguments");
1127
1128 /*
1129 * superproject's resolved head ref.
1130 * if HEAD then the superproject is in a detached head state, otherwise
1131 * it will be the resolved head ref.
1132 */
1133 superproject_head = argv[1];
1134 argv++;
1135 argc--;
1136 /* Get the submodule's head ref and determine if it is detached */
1137 head = resolve_refdup("HEAD", 0, head_oid.hash, NULL);
1138 if (!head)
1139 die(_("Failed to resolve HEAD as a valid ref."));
1140 if (!strcmp(head, "HEAD"))
1141 detached_head = 1;
1142
1143 /*
1144 * The remote must be configured.
1145 * This is to avoid pushing to the exact same URL as the parent.
1146 */
1147 remote = pushremote_get(argv[1]);
1148 if (!remote || remote->origin == REMOTE_UNCONFIGURED)
1149 die("remote '%s' not configured", argv[1]);
1150
1151 /* Check the refspec */
1152 if (argc > 2) {
1153 int i, refspec_nr = argc - 2;
1154 struct ref *local_refs = get_local_heads();
1155 struct refspec *refspec = parse_push_refspec(refspec_nr,
1156 argv + 2);
1157
1158 for (i = 0; i < refspec_nr; i++) {
1159 struct refspec *rs = refspec + i;
1160
1161 if (rs->pattern || rs->matching)
1162 continue;
1163
1164 /* LHS must match a single ref */
1165 switch (count_refspec_match(rs->src, local_refs, NULL)) {
1166 case 1:
1167 break;
1168 case 0:
1169 /*
1170 * If LHS matches 'HEAD' then we need to ensure
1171 * that it matches the same named branch
1172 * checked out in the superproject.
1173 */
1174 if (!strcmp(rs->src, "HEAD")) {
1175 if (!detached_head &&
1176 !strcmp(head, superproject_head))
1177 break;
1178 die("HEAD does not match the named branch in the superproject");
1179 }
1180 default:
1181 die("src refspec '%s' must name a ref",
1182 rs->src);
1183 }
1184 }
1185 free_refspec(refspec_nr, refspec);
1186 }
1187 free(head);
1188
1189 return 0;
1190}
