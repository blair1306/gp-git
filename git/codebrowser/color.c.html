30/* An individual foreground or background color. */
31struct color {
32 enum {
33 COLOR_UNSPECIFIED = 0,
34 COLOR_NORMAL,
35 COLOR_ANSI, /* basic 0-7 ANSI colors */
36 COLOR_256,
37 COLOR_RGB
38 } type;
39 /* The numeric value for ANSI and 256-color modes */
40 unsigned char value;
41 /* 24-bit RGB color values */
42 unsigned char red, green, blue;
43};

45/*
46 * "word" is a buffer of length "len"; does it match the NUL-terminated
47 * "match" exactly?
48 */
49static int match_word(const char *word, int len, const char *match)
50{
51 return !strncasecmp(word, match, len) && !match[len];
52}

54static int get_hex_color(const char *in, unsigned char *out)
55{
56 unsigned int val;
57 val = (hexval(in[0]) << 4) | hexval(in[1]);
58 if (val & ~0xff)
59 return -1;
60 *out = val;
61 return 0;
62}

64static int parse_color(struct color *out, const char *name, int len)
65{
66 /* Positions in array must match ANSI color codes */
67 static const char * const color_names[] = {
68 "black", "red", "green", "yellow",
69 "blue", "magenta", "cyan", "white"
70 };
71 char *end;
72 int i;
73 long val;
74
75 /* First try the special word "normal"... */
76 if (match_word(name, len, "normal")) {
77 out->type = COLOR_NORMAL;
78 return 0;
79 }
80
81 /* Try a 24-bit RGB value */
82 if (len == 7 && name[0] == '#') {
83 if (!get_hex_color(name + 1, &out->red) &&
84 !get_hex_color(name + 3, &out->green) &&
85 !get_hex_color(name + 5, &out->blue)) {
86 out->type = COLOR_RGB;
87 return 0;
88 }
89 }
90
91 /* Then pick from our human-readable color names... */
92 for (i = 0; i < ARRAY_SIZE(color_names); i++) {
93 if (match_word(name, len, color_names[i])) {
94 out->type = COLOR_ANSI;
95 out->value = i;
96 return 0;
97 }
98 }
99
100 /* And finally try a literal 256-color-mode number */
101 val = strtol(name, &end, 10);
102 if (end - name == len) {
103 /*
104 * Allow "-1" as an alias for "normal", but other negative
105 * numbers are bogus.
106 */
107 if (val < -1)
108 ; /* fall through to error */
109 else if (val < 0) {
110 out->type = COLOR_NORMAL;
111 return 0;
112 /* Rewrite low numbers as more-portable standard colors. */
113 } else if (val < 8) {
114 out->type = COLOR_ANSI;
115 out->value = val;
116 return 0;
117 } else if (val < 256) {
118 out->type = COLOR_256;
119 out->value = val;
120 return 0;
121 }
122 }
123
124 return -1;
125}

127static int parse_attr(const char *name, size_t len)
128{
129 static const struct {
130 const char *name;
131 size_t len;
132 int val, neg;
133 } attrs[] = {
134#define ATTR(x, val, neg) { (x), sizeof(x)-1, (val), (neg) }
135 ATTR("bold", 1, 22),
136 ATTR("dim", 2, 22),
137 ATTR("italic", 3, 23),
138 ATTR("ul", 4, 24),
139 ATTR("blink", 5, 25),
140 ATTR("reverse", 7, 27),
141 ATTR("strike", 9, 29)
142#undef ATTR
143 };
144 int negate = 0;
145 int i;
146
147 if (skip_prefix_mem(name, len, "no", &name, &len)) {
148 skip_prefix_mem(name, len, "-", &name, &len);
149 negate = 1;
150 }
151
152 for (i = 0; i < ARRAY_SIZE(attrs); i++) {
153 if (attrs[i].len == len && !memcmp(attrs[i].name, name, len))
154 return negate ? attrs[i].neg : attrs[i].val;
155 }
156 return -1;
157}

169/*
170 * Write the ANSI color codes for "c" to "out"; the string should
171 * already have the ANSI escape code in it. "out" should have enough
172 * space in it to fit any color.
173 */
174static char *color_output(char *out, int len, const struct color *c, char type)
175{
176 switch (c->type) {
177 case COLOR_UNSPECIFIED:
178 case COLOR_NORMAL:
179 break;
180 case COLOR_ANSI:
181 if (len < 2)
182 die("BUG: color parsing ran out of space");
183 *out++ = type;
184 *out++ = '0' + c->value;
185 break;
186 case COLOR_256:
187 out += xsnprintf(out, len, "%c8;5;%d", type, c->value);
188 break;
189 case COLOR_RGB:
190 out += xsnprintf(out, len, "%c8;2;%d;%d;%d", type,
191 c->red, c->green, c->blue);
192 break;
193 }
194 return out;
195}

202int color_parse_mem(const char *value, int value_len, char *dst)
203{
204 const char *ptr = value;
205 int len = value_len;
206 char *end = dst + COLOR_MAXLEN;
207 unsigned int attr = 0;
208 struct color fg = { COLOR_UNSPECIFIED };
209 struct color bg = { COLOR_UNSPECIFIED };
210
211 while (len > 0 && isspace(*ptr)) {
212 ptr++;
213 len--;
214 }
215
216 if (!len) {
217 dst[0] = '\0';
218 return 0;
219 }
220
221 if (!strncasecmp(ptr, "reset", len)) {
222 xsnprintf(dst, end - dst, GIT_COLOR_RESET);
223 return 0;
224 }
225
226 /* [fg [bg]] [attr]... */
227 while (len > 0) {
228 const char *word = ptr;
229 struct color c = { COLOR_UNSPECIFIED };
230 int val, wordlen = 0;
231
232 while (len > 0 && !isspace(word[wordlen])) {
233 wordlen++;
234 len--;
235 }
236
237 ptr = word + wordlen;
238 while (len > 0 && isspace(*ptr)) {
239 ptr++;
240 len--;
241 }
242
243 if (!parse_color(&c, word, wordlen)) {
244 if (fg.type == COLOR_UNSPECIFIED) {
245 fg = c;
246 continue;
247 }
248 if (bg.type == COLOR_UNSPECIFIED) {
249 bg = c;
250 continue;
251 }
252 goto bad;
253 }
254 val = parse_attr(word, wordlen);
255 if (0 <= val)
256 attr |= (1 << val);
257 else
258 goto bad;
259 }
260
261#undef OUT
262#define OUT(x) do { \
263 if (dst == end) \
264 die("BUG: color parsing ran out of space"); \
265 *dst++ = (x); \
266} while(0)
267
268 if (attr || !color_empty(&fg) || !color_empty(&bg)) {
269 int sep = 0;
270 int i;
271
272 OUT('\033');
273 OUT('[');
274
275 for (i = 0; attr; i++) {
276 unsigned bit = (1 << i);
277 if (!(attr & bit))
278 continue;
279 attr &= ~bit;
280 if (sep++)
281 OUT(';');
282 dst += xsnprintf(dst, end - dst, "%d", i);
283 }
284 if (!color_empty(&fg)) {
285 if (sep++)
286 OUT(';');
287 /* foreground colors are all in the 3x range */
288 dst = color_output(dst, end - dst, &fg, '3');
289 }
290 if (!color_empty(&bg)) {
291 if (sep++)
292 OUT(';');
293 /* background colors are all in the 4x range */
294 dst = color_output(dst, end - dst, &bg, '4');
295 }
296 OUT('m');
297 }
298 OUT(0);
299 return 0;
300bad:
301 return error(_("invalid color value: %.*s"), value_len, value);
302#undef OUT
303}

305int git_config_colorbool(const char *var, const char *value)
306{
307 if (value) {
308 if (!strcasecmp(value, "never"))
309 return 0;
310 if (!strcasecmp(value, "always"))
311 return 1;
312 if (!strcasecmp(value, "auto"))
313 return GIT_COLOR_AUTO;
314 }
315
316 if (!var)
317 return -1;
318
319 /* Missing or explicit false to turn off colorization */
320 if (!git_config_bool(var, value))
321 return 0;
322
323 /* any normal truth value defaults to 'auto' */
324 return GIT_COLOR_AUTO;
325}

339int want_color(int var)
340{
341 static int want_auto = -1;
342
343 if (var < 0)
344 var = git_use_color_default;
345
346 if (var == GIT_COLOR_AUTO) {
347 if (want_auto < 0)
