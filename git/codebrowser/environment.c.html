126static char *expand_namespace(const char *raw_namespace)
127{
128 struct strbuf buf = STRBUF_INIT;
129 struct strbuf **components, **c;
130
131 if (!raw_namespace || !*raw_namespace)
132 return xstrdup("");
133
134 strbuf_addstr(&buf, raw_namespace);
135 components = strbuf_split(&buf, '/');
136 strbuf_reset(&buf);
137 for (c = components; *c; c++)
138 if (strcmp((*c)->buf, "/") != 0)
139 strbuf_addf(&buf, "refs/namespaces/%s", (*c)->buf);
140 strbuf_list_free(components);
141 if (check_refname_format(buf.buf, 0))
142 die("bad git namespace path \"%s\"", raw_namespace);
143 strbuf_addch(&buf, '/');
144 return strbuf_detach(&buf, NULL);
145}

147void setup_git_env(void)
148{
149 const char *shallow_file;
150 const char *replace_ref_base;
151
152 if (getenv(NO_REPLACE_OBJECTS_ENVIRONMENT))
153 check_replace_refs = 0;
154 replace_ref_base = getenv(GIT_REPLACE_REF_BASE_ENVIRONMENT);
155 git_replace_ref_base = xstrdup(replace_ref_base ? replace_ref_base
156 : "refs/replace/");
157 namespace = expand_namespace(getenv(GIT_NAMESPACE_ENVIRONMENT));
158 shallow_file = getenv(GIT_SHALLOW_FILE_ENVIRONMENT);
159 if (shallow_file)
160 set_alternate_shallow_file(shallow_file, 0);
161}

216/*
217 * Note. This works only before you used a work tree. This was added
218 * primarily to support git-clone to work in a new repository it just
219 * created, and is not meant to flip between different work trees.
220 */
221void set_git_work_tree(const char *new_work_tree)
222{
223 if (git_work_tree_initialized) {
224 new_work_tree = real_path(new_work_tree);
225 if (strcmp(new_work_tree, the_repository->worktree))
226 die("internal error: work tree has already been set\n"
227 "Current worktree: %s\nNew worktree: %s",
228 the_repository->worktree, new_work_tree);
229 return;
230 }
231 git_work_tree_initialized = 1;
232 repo_set_worktree(the_repository, new_work_tree);
233}

247int odb_mkstemp(struct strbuf *template, const char *pattern)
248{
249 int fd;
250 /*
251 * we let the umask do its job, don't try to be more
252 * restrictive except to remove write permission.
253 */
254 int mode = 0444;
255 git_path_buf(template, "objects/%s", pattern);
256 fd = git_mkstemp_mode(template->buf, mode);
257 if (0 <= fd)
258 return fd;
259
260 /* slow path */
261 /* some mkstemp implementations erase template on failure */
262 git_path_buf(template, "objects/%s", pattern);
263 safe_create_leading_directories(template->buf);
264 return xmkstemp_mode(template->buf, mode);
265}

267int odb_pack_keep(const char *name)
268{
269 int fd;
270
271 fd = open(name, O_RDWR|O_CREAT|O_EXCL, 0600);
272 if (0 <= fd)
273 return fd;
274
275 /* slow path */
276 safe_create_leading_directories_const(name);
277 return open(name, O_RDWR|O_CREAT|O_EXCL, 0600);
278}

314static int the_shared_repository = PERM_UMASK;
315static int need_shared_repository_from_config = 1;
316
317void set_shared_repository(int value)
318{
319 the_shared_repository = value;
320 need_shared_repository_from_config = 0;
321}
322
323int get_shared_repository(void)
324{
325 if (need_shared_repository_from_config) {
326 const char *var = "core.sharedrepository";
327 const char *value;
328 if (!git_config_get_value(var, &value))
329 the_shared_repository = git_config_perm(var, value);
330 need_shared_repository_from_config = 0;
331 }
332 return the_shared_repository;
333}

335void reset_shared_repository(void)
336{
337 need_shared_repository_from_config = 1;
