28/*
29 * "Normalize" a key argument by converting NULL to our trace_default,
30 * and otherwise passing through the value. All caller-facing functions
31 * should normalize their inputs in this way, though most get it
32 * for free by calling get_trace_fd() (directly or indirectly).
33 */
34static void normalize_trace_key(struct trace_key **key)
35{
36 static struct trace_key trace_default = { "GIT_TRACE" };
37 if (!*key)
38 *key = &trace_default;
39}
40
41/* Get a trace file descriptor from "key" env variable. */
42static int get_trace_fd(struct trace_key *key)
43{
44 const char *trace;
45
46 normalize_trace_key(&key);
47
48 /* don't open twice */
49 if (key->initialized)
50 return key->fd;
51
52 trace = getenv(key->key);
53
54 if (!trace || !strcmp(trace, "") ||
55 !strcmp(trace, "0") || !strcasecmp(trace, "false"))
56 key->fd = 0;
57 else if (!strcmp(trace, "1") || !strcasecmp(trace, "true"))
58 key->fd = STDERR_FILENO;
59 else if (strlen(trace) == 1 && isdigit(*trace))
60 key->fd = atoi(trace);
61 else if (is_absolute_path(trace)) {
62 int fd = open(trace, O_WRONLY | O_APPEND | O_CREAT, 0666);
63 if (fd == -1) {
64 warning("could not open '%s' for tracing: %s",
65 trace, strerror(errno));
66 trace_disable(key);
67 } else {
68 key->fd = fd;
69 key->need_close = 1;
70 }
71 } else {
72 warning("unknown trace value for '%s': %s\n"
73 " If you want to trace into a file, then please set %s\n"
74 " to an absolute pathname (starting with /)",
75 key->key, trace, key->key);
76 trace_disable(key);
77 }
78
79 key->initialized = 1;
80 return key->fd;
81}
82
83void trace_disable(struct trace_key *key)
84{
85 normalize_trace_key(&key);
86
87 if (key->need_close)
88 close(key->fd);
89 key->fd = 0;
90 key->initialized = 1;
91 key->need_close = 0;
92}

94static int prepare_trace_line(const char *file, int line,
95 struct trace_key *key, struct strbuf *buf)
96{
97 static struct trace_key trace_bare = TRACE_KEY_INIT(BARE);
98 struct timeval tv;
99 struct tm tm;
100 time_t secs;
101
102 if (!trace_want(key))
103 return 0;
104
105 set_try_to_free_routine(NULL); /* is never reset */
106
107 /* unit tests may want to disable additional trace output */
108 if (trace_want(&trace_bare))
109 return 1;
110
111 /* print current timestamp */
112 gettimeofday(&tv, NULL);
113 secs = tv.tv_sec;
114 localtime_r(&secs, &tm);
115 strbuf_addf(buf, "%02d:%02d:%02d.%06ld ", tm.tm_hour, tm.tm_min,
116 tm.tm_sec, (long) tv.tv_usec);
117
118#ifdef HAVE_VARIADIC_MACROS
119 /* print file:line */
120 strbuf_addf(buf, "%s:%d ", file, line);
121 /* align trace output (column 40 catches most files names in git) */
122 while (buf->len < 40)
123 strbuf_addch(buf, ' ');
124#endif
125
126 return 1;
127}

129static void trace_write(struct trace_key *key, const void *buf, unsigned len)
130{
131 if (write_in_full(get_trace_fd(key), buf, len) < 0) {
132 normalize_trace_key(&key);
133 warning("unable to write trace for %s: %s",
134 key->key, strerror(errno));
135 trace_disable(key);
136 }
137}
138
139void trace_verbatim(struct trace_key *key, const void *buf, unsigned len)
140{
141 if (!trace_want(key))
142 return;
143 trace_write(key, buf, len);
144}
145
146static void print_trace_line(struct trace_key *key, struct strbuf *buf)
147{
148 strbuf_complete_line(buf);
149 trace_write(key, buf->buf, buf->len);
150 strbuf_release(buf);
151}

153static void trace_vprintf_fl(const char *file, int line, struct trace_key *key,
154 const char *format, va_list ap)
155{
156 struct strbuf buf = STRBUF_INIT;
157
158 if (!prepare_trace_line(file, line, key, &buf))
159 return;
160
161 strbuf_vaddf(&buf, format, ap);
162 print_trace_line(key, &buf);
163}

165static void trace_argv_vprintf_fl(const char *file, int line,
166 const char **argv, const char *format,
167 va_list ap)
168{
169 struct strbuf buf = STRBUF_INIT;
170
171 if (!prepare_trace_line(file, line, NULL, &buf))
172 return;
173
174 strbuf_vaddf(&buf, format, ap);
175
176 sq_quote_argv(&buf, argv, 0);
177 print_trace_line(NULL, &buf);
178}

180void trace_strbuf_fl(const char *file, int line, struct trace_key *key,

192static struct trace_key trace_perf_key = TRACE_KEY_INIT(PERFORMANCE);
193
194static void trace_performance_vprintf_fl(const char *file, int line,
195 uint64_t nanos, const char *format,
196 va_list ap)
197{
198 struct strbuf buf = STRBUF_INIT;
199
200 if (!prepare_trace_line(file, line, &trace_perf_key, &buf))
201 return;
202
203 strbuf_addf(&buf, "performance: %.9f s", (double) nanos / 1000000000);
204
205 if (format && *format) {
206 strbuf_addstr(&buf, ": ");
207 strbuf_vaddf(&buf, format, ap);
208 }
209
210 print_trace_line(&trace_perf_key, &buf);
211}

293static const char *quote_crnl(const char *path)
294{
295 static struct strbuf new_path = STRBUF_INIT;
296
297 if (!path)
298 return NULL;
299
300 strbuf_reset(&new_path);
301
302 while (*path) {
303 switch (*path) {
304 case '\\': strbuf_addstr(&new_path, "\\\\"); break;
305 case '\n': strbuf_addstr(&new_path, "\\n"); break;
306 case '\r': strbuf_addstr(&new_path, "\\r"); break;
307 default:
308 strbuf_addch(&new_path, *path);
309 }
310 path++;
311 }
312 return new_path.buf;
313}

315/* FIXME: move prefix to startup_info struct and get rid of this arg */
316void trace_repo_setup(const char *prefix)
317{
318 static struct trace_key key = TRACE_KEY_INIT(SETUP);
319 const char *git_work_tree;
320 char *cwd;
321
322 if (!trace_want(&key))
323 return;
324
325 cwd = xgetcwd();
326
327 if (!(git_work_tree = get_git_work_tree()))
328 git_work_tree = "(null)";
329
330 if (!prefix)
331 prefix = "(null)";
332
333 trace_printf_key(&key, "setup: git_dir: %s\n", quote_crnl(get_git_dir()));
334 trace_printf_key(&key, "setup: git_common_dir: %s\n", quote_crnl(get_git_common_dir()));
335 trace_printf_key(&key, "setup: worktree: %s\n", quote_crnl(git_work_tree));
336 trace_printf_key(&key, "setup: cwd: %s\n", quote_crnl(cwd));
337 trace_printf_key(&key, "setup: prefix: %s\n", quote_crnl(prefix));
338
339 free(cwd);
340}

347#if defined(HAVE_CLOCK_GETTIME) && defined(HAVE_CLOCK_MONOTONIC)
348
349static inline uint64_t highres_nanos(void)
350{
351 struct timespec ts;
352 if (clock_gettime(CLOCK_MONOTONIC, &ts))
353 return 0;
354 return (uint64_t) ts.tv_sec * 1000000000 + ts.tv_nsec;
355}
356
357#elif defined (GIT_WINDOWS_NATIVE)
358
359static inline uint64_t highres_nanos(void)
360{
361 static uint64_t high_ns, scaled_low_ns;
362 static int scale;
363 LARGE_INTEGER cnt;
364
365 if (!scale) {
366 if (!QueryPerformanceFrequency(&cnt))
367 return 0;
368
369 /* high_ns = number of ns per cnt.HighPart */
370 high_ns = (1000000000LL << 32) / (uint64_t) cnt.QuadPart;
371
372 /*
373 * Number of ns per cnt.LowPart is 10^9 / frequency (or
374 * high_ns >> 32). For maximum precision, we scale this factor
375 * so that it just fits within 32 bit (i.e. won't overflow if
376 * multiplied with cnt.LowPart).
377 */
378 scaled_low_ns = high_ns;
379 scale = 32;
380 while (scaled_low_ns >= 0x100000000LL) {
381 scaled_low_ns >>= 1;
382 scale--;
383 }
384 }
385
386 /* if QPF worked on initialization, we expect QPC to work as well */
387 QueryPerformanceCounter(&cnt);
388
389 return (high_ns * cnt.HighPart) +
390 ((scaled_low_ns * cnt.LowPart) >> scale);
391}
392
393#else
394# define highres_nanos() 0
395#endif

397static inline uint64_t gettimeofday_nanos(void)
398{
399 struct timeval tv;
400 gettimeofday(&tv, NULL);
401 return (uint64_t) tv.tv_sec * 1000000000 + tv.tv_usec * 1000;
402}

404/*
405 * Returns nanoseconds since the epoch (01/01/1970), for performance tracing
406 * (i.e. favoring high precision over wall clock time accuracy).
407 */
408uint64_t getnanotime(void)
409{
410 static uint64_t offset;
411 if (offset > 1) {
412 /* initialization succeeded, return offset + high res time */
413 return offset + highres_nanos();
414 } else if (offset == 1) {
415 /* initialization failed, fall back to gettimeofday */
416 return gettimeofday_nanos();
417 } else {
418 /* initialize offset if high resolution timer works */
419 uint64_t now = gettimeofday_nanos();
420 uint64_t highres = highres_nanos();
421 if (highres)
422 offset = now - highres;
423 else
424 offset = 1;
425 return now;
426 }
427}

429static uint64_t command_start_time;
430static struct strbuf command_line = STRBUF_INIT;
431
432static void print_command_performance_atexit(void)
433{
434 trace_performance_since(command_start_time, "git command:%s",
435 command_line.buf);
436}
437
438void trace_command_performance(const char **argv)
439{
440 if (!trace_want(&trace_perf_key))
441 return;
442
443 if (!command_start_time)
444 atexit(print_command_performance_atexit);
445
446 strbuf_reset(&command_line);
447 sq_quote_argv(&command_line, argv, 0);
448 command_start_time = getnanotime();
449}
