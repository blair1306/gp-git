95#define DIFF_OPT_TST(opts, flag) ((opts)->flags & DIFF_OPT_##flag)
96#define DIFF_OPT_TOUCHED(opts, flag) ((opts)->touched_flags & DIFF_OPT_##flag)
97#define DIFF_OPT_SET(opts, flag) (((opts)->flags |= DIFF_OPT_##flag),((opts)->touched_flags |= DIFF_OPT_##flag))
98#define DIFF_OPT_CLR(opts, flag) (((opts)->flags &= ~DIFF_OPT_##flag),((opts)->touched_flags |= DIFF_OPT_##flag))
99#define DIFF_XDL_TST(opts, flag) ((opts)->xdl_opts & XDF_##flag)
100#define DIFF_XDL_SET(opts, flag) ((opts)->xdl_opts |= XDF_##flag)
101#define DIFF_XDL_CLR(opts, flag) ((opts)->xdl_opts &= ~XDF_##flag)
102
103#define DIFF_WITH_ALG(opts, flag) (((opts)->xdl_opts & ~XDF_DIFF_ALGORITHM_MASK) | XDF_##flag)

366/*
367 * Fill the contents of the filespec "df", respecting any textconv defined by
368 * its userdiff driver. The "driver" parameter must come from a
369 * previous call to get_textconv(), and therefore should either be NULL or have
370 * textconv enabled.
371 *
372 * Note that the memory ownership of the resulting buffer depends on whether
373 * the driver field is NULL. If it is, then the memory belongs to the filespec
374 * struct. If it is non-NULL, then "outbuf" points to a newly allocated buffer
375 * that should be freed by the caller.
376 */
377extern size_t fill_textconv(struct userdiff_driver *driver,
378 struct diff_filespec *df,
379 char **outbuf);
