57static void write_eolinfo(const struct index_state *istate,
58 const struct cache_entry *ce, const char *path)
59{
60 if (show_eol) {
61 struct stat st;
62 const char *i_txt = "";
63 const char *w_txt = "";
64 const char *a_txt = get_convert_attr_ascii(path);
65 if (ce && S_ISREG(ce->ce_mode))
66 i_txt = get_cached_convert_stats_ascii(istate,
67 ce->name);
68 if (!lstat(path, &st) && S_ISREG(st.st_mode))
69 w_txt = get_wt_convert_stats_ascii(path);
70 printf("i/%-5s w/%-5s attr/%-17s\t", i_txt, w_txt, a_txt);
71 }
72}

84static const char *get_tag(const struct cache_entry *ce, const char *tag)
85{
86 static char alttag[4];
87
88 if (tag && *tag && show_valid_bit && (ce->ce_flags & CE_VALID)) {
89 memcpy(alttag, tag, 3);
90
91 if (isalpha(tag[0])) {
92 alttag[0] = tolower(tag[0]);
93 } else if (tag[0] == '?') {
94 alttag[0] = '!';
95 } else {
96 alttag[0] = 'v';
97 alttag[1] = tag[0];
98 alttag[2] = ' ';
99 alttag[3] = 0;
100 }
101
102 tag = alttag;
103 }
104
105 return tag;
106}

108static void print_debug(const struct cache_entry *ce)
109{
110 if (debug_mode) {
111 const struct stat_data *sd = &ce->ce_stat_data;
112
113 printf(" ctime: %d:%d\n", sd->sd_ctime.sec, sd->sd_ctime.nsec);
114 printf(" mtime: %d:%d\n", sd->sd_mtime.sec, sd->sd_mtime.nsec);
115 printf(" dev: %d\tino: %d\n", sd->sd_dev, sd->sd_ino);

117 printf(" size: %d\tflags: %x\n", sd->sd_size, ce->ce_flags);
118 }

121static void show_dir_entry(const char *tag, struct dir_entry *ent)
122{
123 int len = max_prefix_len;
124
125 if (len > ent->len)
126 die("git ls-files: internal error - directory entry not superset of prefix");
127
128 if (!dir_path_match(ent, &pathspec, len, ps_matched))
129 return;
130
131 fputs(tag, stdout);
132 write_eolinfo(NULL, NULL, ent->name);
133 write_name(ent->name);
134}

136static void show_other_files(const struct index_state *istate,
137 const struct dir_struct *dir)

149static void show_killed_files(const struct index_state *istate,
150 const struct dir_struct *dir)
151{
152 int i;
153 for (i = 0; i < dir->nr; i++) {
154 struct dir_entry *ent = dir->entries[i];
155 char *cp, *sp;
156 int pos, len, killed = 0;
157
158 for (cp = ent->name; cp - ent->name < ent->len; cp = sp + 1) {
159 sp = strchr(cp, '/');
160 if (!sp) {
161 /* If ent->name is prefix of an entry in the
162 * cache, it will be killed.
163 */
164 pos = index_name_pos(istate, ent->name, ent->len);
165 if (0 <= pos)
166 die("BUG: killed-file %.*s not found",
167 ent->len, ent->name);
168 pos = -pos - 1;
169 while (pos < istate->cache_nr &&
170 ce_stage(istate->cache[pos]))
171 pos++; /* skip unmerged */
172 if (istate->cache_nr <= pos)
173 break;
174 /* pos points at a name immediately after
175 * ent->name in the cache. Does it expect
176 * ent->name to be a directory?
177 */
178 len = ce_namelen(istate->cache[pos]);
179 if ((ent->len < len) &&
180 !strncmp(istate->cache[pos]->name,
181 ent->name, ent->len) &&
182 istate->cache[pos]->name[ent->len] == '/')
183 killed = 1;
184 break;
185 }
186 if (0 <= index_name_pos(istate, ent->name, sp - ent->name)) {
187 /* If any of the leading directories in
188 * ent->name is registered in the cache,
189 * ent->name will be killed.
190 */
191 killed = 1;
192 break;
193 }
194 }
195 if (killed)
196 show_dir_entry(tag_killed, dir->entries[i]);
197 }
198}

202static void show_submodule(struct repository *superproject,
203 struct dir_struct *dir, const char *path)
204{
205 struct repository submodule;
206
207 if (repo_submodule_init(&submodule, superproject, path))
208 return;
209
210 if (repo_read_index(&submodule) < 0)
211 die("index file corrupt");
212

214
215 show_files(&submodule, dir);
216
217 repo_clear(&submodule);

220static void show_ce(struct repository *repo, struct dir_struct *dir,
221 const struct cache_entry *ce, const char *fullname,
222 const char *tag)
223{
224 if (max_prefix_len > strlen(fullname))
225 die("git ls-files: internal error - cache entry not superset of prefix");
226
227 if (recurse_submodules && S_ISGITLINK(ce->ce_mode) &&
228 is_submodule_active(repo, ce->name)) {
229 show_submodule(repo, dir, ce->name);
230 } else if (match_pathspec(&pathspec, fullname, strlen(fullname),
231 max_prefix_len, ps_matched,
232 S_ISDIR(ce->ce_mode) ||
233 S_ISGITLINK(ce->ce_mode))) {
234 tag = get_tag(ce, tag);
235
236 if (!show_stage) {
237 fputs(tag, stdout);
238 } else {
239 printf("%s%06o %s %d\t",
240 tag,
241 ce->ce_mode,
242 find_unique_abbrev(ce->oid.hash, abbrev),
243 ce_stage(ce));
244 }
245 write_eolinfo(repo->index, ce, fullname);
246 write_name(fullname);
247 print_debug(ce);
248 }
249}

251static void show_ru_info(const struct index_state *istate)
252{
253 struct string_list_item *item;
254
255 if (!istate->resolve_undo)
256 return;
257
258 for_each_string_list_item(item, istate->resolve_undo) {
259 const char *path = item->string;
260 struct resolve_undo_info *ui = item->util;
261 int i, len;
262
263 len = strlen(path);
264 if (len < max_prefix_len)
265 continue; /* outside of the prefix */
266 if (!match_pathspec(&pathspec, path, len,
267 max_prefix_len, ps_matched, 0))
268 continue; /* uninterested */
269 for (i = 0; i < 3; i++) {
270 if (!ui->mode[i])
271 continue;
272 printf("%s%06o %s %d\t", tag_resolve_undo, ui->mode[i],
273 find_unique_abbrev(ui->sha1[i], abbrev),
274 i + 1);
275 write_name(path);
276 }
277 }

296static void show_files(struct repository *repo, struct dir_struct *dir)
297{
298 int i;
299 struct strbuf fullname = STRBUF_INIT;
300
301 /* For cached/deleted files we don't need to even do the readdir */
302 if (show_others || show_killed) {
303 if (!show_others)
304 dir->flags |= DIR_COLLECT_KILLED_ONLY;
305 fill_directory(dir, repo->index, &pathspec);
306 if (show_others)
307 show_other_files(repo->index, dir);
308 if (show_killed)
309 show_killed_files(repo->index, dir);
310 }
311 if (show_cached || show_stage) {
312 for (i = 0; i < repo->index->cache_nr; i++) {
313 const struct cache_entry *ce = repo->index->cache[i];
314
315 construct_fullname(&fullname, repo, ce);
316
317 if ((dir->flags & DIR_SHOW_IGNORED) &&
318 !ce_excluded(dir, repo->index, fullname.buf, ce))
319 continue;
320 if (show_unmerged && !ce_stage(ce))
321 continue;
322 if (ce->ce_flags & CE_UPDATE)
323 continue;
324 show_ce(repo, dir, ce, fullname.buf,
325 ce_stage(ce) ? tag_unmerged :
326 (ce_skip_worktree(ce) ? tag_skip_worktree :
327 tag_cached));
328 }
329 }
330 if (show_deleted || show_modified) {
331 for (i = 0; i < repo->index->cache_nr; i++) {
332 const struct cache_entry *ce = repo->index->cache[i];
333 struct stat st;
334 int err;
335
336 construct_fullname(&fullname, repo, ce);
337
338 if ((dir->flags & DIR_SHOW_IGNORED) &&
339 !ce_excluded(dir, repo->index, fullname.buf, ce))
340 continue;
341 if (ce->ce_flags & CE_UPDATE)
342 continue;
343 if (ce_skip_worktree(ce))
344 continue;
345 err = lstat(fullname.buf, &st);
346 if (show_deleted && err)
347 show_ce(repo, dir, ce, fullname.buf, tag_removed);
348 if (show_modified && ie_modified(repo->index, ce, &st, 0))
349 show_ce(repo, dir, ce, fullname.buf, tag_modified);
350 }
351 }
352
353 strbuf_release(&fullname);
354}

356/*
357 * Prune the index to only contain stuff starting with "prefix"
358 */
359static void prune_index(struct index_state *istate,
360 const char *prefix, size_t prefixlen)
361{
362 int pos;
363 unsigned int first, last;
364
365 if (!prefix || !istate->cache_nr)
366 return;
367 pos = index_name_pos(istate, prefix, prefixlen);
368 if (pos < 0)
369 pos = -pos-1;
370 first = pos;
371 last = istate->cache_nr;
372 while (last > first) {
373 int next = (last + first) >> 1;
374 const struct cache_entry *ce = istate->cache[next];
375 if (!strncmp(ce->name, prefix, prefixlen)) {
376 first = next+1;
377 continue;
378 }
379 last = next;
380 }
381 MOVE_ARRAY(istate->cache, istate->cache + pos, last - pos);
382 istate->cache_nr = last - pos;
383}

385static int get_common_prefix_len(const char *common_prefix)
386{
387 int common_prefix_len;
388
389 if (!common_prefix)
390 return 0;
391
392 common_prefix_len = strlen(common_prefix);
393
394 /*
395 * If the prefix has a trailing slash, strip it so that submodules wont
396 * be pruned from the index.
397 */
398 if (common_prefix[common_prefix_len - 1] == '/')
399 common_prefix_len--;
400
401 return common_prefix_len;
402}

404/*
405 * Read the tree specified with --with-tree option
406 * (typically, HEAD) into stage #1 and then
407 * squash them down to stage #0. This is used for
408 * --error-unmatch to list and check the path patterns
409 * that were given from the command line. We are not
410 * going to write this index out.
411 */
412void overlay_tree_on_index(struct index_state *istate,
413 const char *tree_name, const char *prefix)
414{
415 struct tree *tree;
416 struct object_id oid;
417 struct pathspec pathspec;
418 struct cache_entry *last_stage0 = NULL;
419 int i;
420
421 if (get_oid(tree_name, &oid))
422 die("tree-ish %s not found.", tree_name);
423 tree = parse_tree_indirect(&oid);
424 if (!tree)
425 die("bad tree-ish %s", tree_name);
426
427 /* Hoist the unmerged entries up to stage #3 to make room */
428 for (i = 0; i < istate->cache_nr; i++) {
429 struct cache_entry *ce = istate->cache[i];
430 if (!ce_stage(ce))
431 continue;
432 ce->ce_flags |= CE_STAGEMASK;
433 }
434
435 if (prefix) {
436 static const char *(matchbuf[1]);
437 matchbuf[0] = NULL;
438 parse_pathspec(&pathspec, PATHSPEC_ALL_MAGIC,
439 PATHSPEC_PREFER_CWD, prefix, matchbuf);
440 } else
441 memset(&pathspec, 0, sizeof(pathspec));
442 if (read_tree(tree, 1, &pathspec, istate))
443 die("unable to read tree entries %s", tree_name);
444
445 for (i = 0; i < istate->cache_nr; i++) {
446 struct cache_entry *ce = istate->cache[i];
447 switch (ce_stage(ce)) {
448 case 0:
449 last_stage0 = ce;
450 /* fallthru */
451 default:
452 continue;
453 case 1:
454 /*
455 * If there is stage #0 entry for this, we do not
456 * need to show it. We use CE_UPDATE bit to mark
457 * such an entry.
458 */
459 if (last_stage0 &&
460 !strcmp(last_stage0->name, ce->name))
461 ce->ce_flags |= CE_UPDATE;
462 }
463 }
464}

504int cmd_ls_files(int argc, const char **argv, const char *cmd_prefix)
505{
506 int require_work_tree = 0, show_tag = 0, i;
507 const char *max_prefix;
508 struct dir_struct dir;
509 struct exclude_list *el;
510 struct string_list exclude_list = STRING_LIST_INIT_NODUP;
511 struct option builtin_ls_files_options[] = {
512 /* Think twice before adding "--nul" synonym to this */
513 OPT_SET_INT('z', NULL, &line_terminator,
514 N_("paths are separated with NUL character"), '\0'),
515 OPT_BOOL('t', NULL, &show_tag,
516 N_("identify the file status with tags")),
517 OPT_BOOL('v', NULL, &show_valid_bit,
518 N_("use lowercase letters for 'assume unchanged' files")),
519 OPT_BOOL('c', "cached", &show_cached,
520 N_("show cached files in the output (default)")),
521 OPT_BOOL('d', "deleted", &show_deleted,
522 N_("show deleted files in the output")),
523 OPT_BOOL('m', "modified", &show_modified,
524 N_("show modified files in the output")),
525 OPT_BOOL('o', "others", &show_others,
526 N_("show other files in the output")),
527 OPT_BIT('i', "ignored", &dir.flags,
528 N_("show ignored files in the output"),
529 DIR_SHOW_IGNORED),
530 OPT_BOOL('s', "stage", &show_stage,
531 N_("show staged contents' object name in the output")),
532 OPT_BOOL('k', "killed", &show_killed,
533 N_("show files on the filesystem that need to be removed")),
534 OPT_BIT(0, "directory", &dir.flags,
535 N_("show 'other' directories' names only"),
536 DIR_SHOW_OTHER_DIRECTORIES),
537 OPT_BOOL(0, "eol", &show_eol, N_("show line endings of files")),
538 OPT_NEGBIT(0, "empty-directory", &dir.flags,
539 N_("don't show empty directories"),
540 DIR_HIDE_EMPTY_DIRECTORIES),
541 OPT_BOOL('u', "unmerged", &show_unmerged,
542 N_("show unmerged files in the output")),
543 OPT_BOOL(0, "resolve-undo", &show_resolve_undo,
544 N_("show resolve-undo information")),
545 { OPTION_CALLBACK, 'x', "exclude", &exclude_list, N_("pattern"),
546 N_("skip files matching pattern"),
547 0, option_parse_exclude },
548 { OPTION_CALLBACK, 'X', "exclude-from", &dir, N_("file"),
549 N_("exclude patterns are read from <file>"),
550 0, option_parse_exclude_from },
551 OPT_STRING(0, "exclude-per-directory", &dir.exclude_per_dir, N_("file"),
552 N_("read additional per-directory exclude patterns in <file>")),
553 { OPTION_CALLBACK, 0, "exclude-standard", &dir, NULL,
554 N_("add the standard git exclusions"),
555 PARSE_OPT_NOARG, option_parse_exclude_standard },
556 { OPTION_SET_INT, 0, "full-name", &prefix_len, NULL,
557 N_("make the output relative to the project top directory"),
558 PARSE_OPT_NOARG | PARSE_OPT_NONEG, NULL },
559 OPT_BOOL(0, "recurse-submodules", &recurse_submodules,
560 N_("recurse through submodules")),
561 OPT_BOOL(0, "error-unmatch", &error_unmatch,
562 N_("if any <file> is not in the index, treat this as an error")),
563 OPT_STRING(0, "with-tree", &with_tree, N_("tree-ish"),
564 N_("pretend that paths removed since <tree-ish> are still present")),
565 OPT__ABBREV(&abbrev),
566 OPT_BOOL(0, "debug", &debug_mode, N_("show debugging data")),
567 OPT_END()
568 };
569
570 if (argc == 2 && !strcmp(argv[1], "-h"))
571 usage_with_options(ls_files_usage, builtin_ls_files_options);
572
573 memset(&dir, 0, sizeof(dir));
574 prefix = cmd_prefix;
575 if (prefix)
576 prefix_len = strlen(prefix);
577 git_config(git_default_config, NULL);
578
579 if (repo_read_index(the_repository) < 0)
580 die("index file corrupt");
581
582 argc = parse_options(argc, argv, prefix, builtin_ls_files_options,
583 ls_files_usage, 0);
584 el = add_exclude_list(&dir, EXC_CMDL, "--exclude option");
585 for (i = 0; i < exclude_list.nr; i++) {
586 add_exclude(exclude_list.items[i].string, "", 0, el, --exclude_args);
587 }
588 if (show_tag || show_valid_bit) {
589 tag_cached = "H ";
590 tag_unmerged = "M ";
591 tag_removed = "R ";
592 tag_modified = "C ";
593 tag_other = "? ";
594 tag_killed = "K ";
595 tag_skip_worktree = "S ";
596 tag_resolve_undo = "U ";
597 }
598 if (show_modified || show_others || show_deleted || (dir.flags & DIR_SHOW_IGNORED) || show_killed)
599 require_work_tree = 1;
600 if (show_unmerged)
601 /*
602 * There's no point in showing unmerged unless
603 * you also show the stage information.
604 */
605 show_stage = 1;
606 if (dir.exclude_per_dir)
607 exc_given = 1;
608
609 if (require_work_tree && !is_inside_work_tree())
610 setup_work_tree();
611
612 if (recurse_submodules)
613 repo_read_gitmodules(the_repository);
614
615 if (recurse_submodules &&
616 (show_stage || show_deleted || show_others || show_unmerged ||
617 show_killed || show_modified || show_resolve_undo || with_tree))
618 die("ls-files --recurse-submodules unsupported mode");
619
620 if (recurse_submodules && error_unmatch)
621 die("ls-files --recurse-submodules does not support "
622 "--error-unmatch");
623
624 parse_pathspec(&pathspec, 0,
625 PATHSPEC_PREFER_CWD,
626 prefix, argv);
627
628 /*
629 * Find common prefix for all pathspec's
630 * This is used as a performance optimization which unfortunately cannot
631 * be done when recursing into submodules because when a pathspec is
632 * given which spans repository boundaries you can't simply remove the
633 * submodule entry because the pathspec may match something inside the
634 * submodule.
635 */
636 if (recurse_submodules)
637 max_prefix = NULL;
638 else
639 max_prefix = common_prefix(&pathspec);
640 max_prefix_len = get_common_prefix_len(max_prefix);
641
642 prune_index(the_repository->index, max_prefix, max_prefix_len);
643
644 /* Treat unmatching pathspec elements as errors */
645 if (pathspec.nr && error_unmatch)
646 ps_matched = xcalloc(pathspec.nr, 1);
647
648 if ((dir.flags & DIR_SHOW_IGNORED) && !exc_given)
649 die("ls-files --ignored needs some exclude pattern");
650
651 /* With no flags, we default to showing the cached files */
652 if (!(show_stage || show_deleted || show_others || show_unmerged ||
653 show_killed || show_modified || show_resolve_undo))
654 show_cached = 1;
655
656 if (with_tree) {
657 /*
658 * Basic sanity check; show-stages and show-unmerged
659 * would not make any sense with this option.
660 */
661 if (show_stage || show_unmerged)
662 die("ls-files --with-tree is incompatible with -s or -u");
663 overlay_tree_on_index(the_repository->index, with_tree, max_prefix);
664 }
665
666 show_files(the_repository, &dir);
667
668 if (show_resolve_undo)
669 show_ru_info(the_repository->index);
670
671 if (ps_matched) {
672 int bad;
673 bad = report_path_error(ps_matched, &pathspec, prefix);
674 if (bad)
675 fprintf(stderr, "Did you forget to 'git add'?\n");
676
677 return bad ? 1 : 0;
678 }
679
680 return 0;
681}
