45/*
46 * These variables record the "current" config source, which
47 * can be accessed by parsing callbacks.
48 *
49 * The "cf" variable will be non-NULL only when we are actually parsing a real
50 * config source (file, blob, cmdline, etc).
51 *
52 * The "current_config_kvi" variable will be non-NULL only when we are feeding
53 * cached config from a configset into a callback.
54 *
55 * They should generally never be non-NULL at the same time. If they are both
56 * NULL, then we aren't parsing anything (and depending on the function looking
57 * at the variables, it's either a bug for it to be called in the first place,
58 * or it's a function which can be reused for non-config purposes, and should
59 * fall back to some sane behavior).
60 */
61static struct config_source *cf;
62static struct key_value_info *current_config_kvi;

101static int config_buf_ungetc(int c, struct config_source *conf)
102{
103 if (conf->u.buf.pos > 0) {
104 conf->u.buf.pos--;
105 if (conf->u.buf.buf[conf->u.buf.pos] != c)
106 die("BUG: config_buf can only ungetc the same character");
107 return c;
108 }
109
110 return EOF;
111}

121" %s\n"
122"from\n"

124"Do you have circular includes?";
125static int handle_path_include(const char *path, struct config_include_data *inc)
126{
127 int ret = 0;
128 struct strbuf buf = STRBUF_INIT;
129 char *expanded;
130
131 if (!path)
132 return config_error_nonbool("include.path");
133
134 expanded = expand_user_path(path, 0);
135 if (!expanded)
136 return error("could not expand include path '%s'", path);
137 path = expanded;
138
139 /*
140 * Use an absolute path as-is, but interpret relative paths
141 * based on the including config file.
142 */
143 if (!is_absolute_path(path)) {
144 char *slash;
145
146 if (!cf || !cf->path)
147 return error("relative config includes must come from files");
148
149 slash = find_last_dir_sep(cf->path);
150 if (slash)
151 strbuf_add(&buf, cf->path, slash - cf->path + 1);
152 strbuf_addstr(&buf, path);
153 path = buf.buf;
154 }
155
156 if (!access_or_die(path, R_OK, 0)) {
157 if (++inc->depth > MAX_INCLUDE_DEPTH)
158 die(include_depth_advice, MAX_INCLUDE_DEPTH, path,
159 !cf ? "<unknown>" :
160 cf->name ? cf->name :
161 "the command line");
162 ret = git_config_from_file(git_config_include, path, inc);
163 inc->depth--;
164 }
165 strbuf_release(&buf);
166 free(expanded);
167 return ret;
168}

170static int prepare_include_condition_pattern(struct strbuf *pat)
171{
172 struct strbuf path = STRBUF_INIT;
173 char *expanded;
174 int prefix = 0;
175
176 expanded = expand_user_path(pat->buf, 1);
177 if (expanded) {
178 strbuf_reset(pat);
179 strbuf_addstr(pat, expanded);
180 free(expanded);
181 }
182
183 if (pat->buf[0] == '.' && is_dir_sep(pat->buf[1])) {
184 const char *slash;
185
186 if (!cf || !cf->path)
187 return error(_("relative config include "
188 "conditionals must come from files"));
189
190 strbuf_realpath(&path, cf->path, 1);
191 slash = find_last_dir_sep(path.buf);
192 if (!slash)
193 die("BUG: how is this possible?");
194 strbuf_splice(pat, 0, 1, path.buf, slash - path.buf);
195 prefix = slash - path.buf + 1 /* slash */;
196 } else if (!is_absolute_path(pat->buf))
197 strbuf_insert(pat, 0, "**/", 3);
198
199 if (pat->len && is_dir_sep(pat->buf[pat->len - 1]))
200 strbuf_addstr(pat, "**");
201
202 strbuf_release(&path);
203 return prefix;
204}

206static int include_by_gitdir(const struct config_options *opts,
207 const char *cond, size_t cond_len, int icase)
208{
209 struct strbuf text = STRBUF_INIT;
210 struct strbuf pattern = STRBUF_INIT;
211 int ret = 0, prefix;
212 const char *git_dir;
213 int already_tried_absolute = 0;
214
215 if (opts->git_dir)
216 git_dir = opts->git_dir;
217 else
218 goto done;
219
220 strbuf_realpath(&text, git_dir, 1);
221 strbuf_add(&pattern, cond, cond_len);
222 prefix = prepare_include_condition_pattern(&pattern);
223
224again:
225 if (prefix < 0)
226 goto done;
227
228 if (prefix > 0) {
229 /*
230 * perform literal matching on the prefix part so that
231 * any wildcard character in it can't create side effects.
232 */
233 if (text.len < prefix)
234 goto done;
235 if (!icase && strncmp(pattern.buf, text.buf, prefix))
236 goto done;
237 if (icase && strncasecmp(pattern.buf, text.buf, prefix))
238 goto done;
239 }
240
241 ret = !wildmatch(pattern.buf + prefix, text.buf + prefix,
242 icase ? WM_CASEFOLD : 0);
243
244 if (!ret && !already_tried_absolute) {
245 /*
246 * We've tried e.g. matching gitdir:~/work, but if
247 * ~/work is a symlink to /mnt/storage/work
248 * strbuf_realpath() will expand it, so the rule won't
249 * match. Let's match against a
250 * strbuf_add_absolute_path() version of the path,
251 * which'll do the right thing
252 */
253 strbuf_reset(&text);
254 strbuf_add_absolute_path(&text, git_dir);
255 already_tried_absolute = 1;
256 goto again;
257 }
258done:
259 strbuf_release(&pattern);
260 strbuf_release(&text);
261 return ret;
262}

264static int include_condition_is_true(const struct config_options *opts,
265 const char *cond, size_t cond_len)
266{
267
268 if (skip_prefix_mem(cond, cond_len, "gitdir:", &cond, &cond_len))
269 return include_by_gitdir(opts, cond, cond_len, 0);
270 else if (skip_prefix_mem(cond, cond_len, "gitdir/i:", &cond, &cond_len))
271 return include_by_gitdir(opts, cond, cond_len, 1);
272
273 /* unknown conditionals are always false */
274 return 0;
275}

277int git_config_include(const char *var, const char *value, void *data)
278{
279 struct config_include_data *inc = data;
280 const char *cond, *key;
281 int cond_len;
282 int ret;
283
284 /*
285 * Pass along all values, including "include" directives; this makes it
286 * possible to query information on the includes themselves.
287 */
288 ret = inc->fn(var, value, inc->data);
289 if (ret < 0)
290 return ret;
291
292 if (!strcmp(var, "include.path"))
293 ret = handle_path_include(value, inc);
294
295 if (!parse_config_key(var, "includeif", &cond, &cond_len, &key) &&
296 (cond && include_condition_is_true(inc->opts, cond, cond_len)) &&
297 !strcmp(key, "path"))
298 ret = handle_path_include(value, inc);
299
300 return ret;
301}

321/*
322 * Auxiliary function to sanity-check and split the key into the section
323 * identifier and variable name.
324 *
325 * Returns 0 on success, -1 when there is an invalid character in the key and
326 * -2 if there is no section name in the key.
327 *
328 * store_key - pointer to char* which will hold a copy of the key with
329 * lowercase section and variable name
330 * baselen - pointer to int which will hold the length of the
331 * section + subsection part, can be NULL
332 */
333static int git_config_parse_key_1(const char *key, char **store_key, int *baselen_, int quiet)
334{
335 int i, dot, baselen;
336 const char *last_dot = strrchr(key, '.');
337
338 /*
339 * Since "key" actually contains the section name and the real
340 * key name separated by a dot, we have to know where the dot is.
341 */
342
343 if (last_dot == NULL || last_dot == key) {
344 if (!quiet)
345 error("key does not contain a section: %s", key);
346 return -CONFIG_NO_SECTION_OR_NAME;
347 }
348
349 if (!last_dot[1]) {
350 if (!quiet)
351 error("key does not contain variable name: %s", key);
352 return -CONFIG_NO_SECTION_OR_NAME;
353 }
354
355 baselen = last_dot - key;
356 if (baselen_)
357 *baselen_ = baselen;
358
359 /*
360 * Validate the key and while at it, lower case it for matching.
361 */
362 if (store_key)
363 *store_key = xmallocz(strlen(key));
364
365 dot = 0;
366 for (i = 0; key[i]; i++) {
367 unsigned char c = key[i];
368 if (c == '.')
369 dot = 1;
370 /* Leave the extended basename untouched.. */
371 if (!dot || i > baselen) {
372 if (!iskeychar(c) ||
373 (i == baselen + 1 && !isalpha(c))) {
374 if (!quiet)
375 error("invalid key: %s", key);
376 goto out_free_ret_1;
377 }
378 c = tolower(c);
379 } else if (c == '\n') {
380 if (!quiet)
381 error("invalid key (newline): %s", key);
382 goto out_free_ret_1;
383 }
384 if (store_key)
385 (*store_key)[i] = c;
386 }
387
388 return 0;
389
390out_free_ret_1:
391 if (store_key) {
392 FREE_AND_NULL(*store_key);
393 }
394 return -CONFIG_INVALID_KEY;
395}

397int git_config_parse_key(const char *key, char **store_key, int *baselen)
398{
399 return git_config_parse_key_1(key, store_key, baselen, 0);

407int git_config_parse_parameter(const char *text,
408 config_fn_t fn, void *data)
409{
410 const char *value;
411 char *canonical_name;
412 struct strbuf **pair;
413 int ret;
414
415 pair = strbuf_split_str(text, '=', 2);
416 if (!pair[0])
417 return error("bogus config parameter: %s", text);
418
419 if (pair[0]->len && pair[0]->buf[pair[0]->len - 1] == '=') {
420 strbuf_setlen(pair[0], pair[0]->len - 1);
421 value = pair[1] ? pair[1]->buf : "";
422 } else {
423 value = NULL;
424 }
425
426 strbuf_trim(pair[0]);
427 if (!pair[0]->len) {
428 strbuf_list_free(pair);
429 return error("bogus config parameter: %s", text);
430 }
431
432 if (git_config_parse_key(pair[0]->buf, &canonical_name, NULL)) {
433 ret = -1;
434 } else {
435 ret = (fn(canonical_name, value, data) < 0) ? -1 : 0;
436 free(canonical_name);
437 }
438 strbuf_list_free(pair);
439 return ret;
440}

442int git_config_from_parameters(config_fn_t fn, void *data)
443{
444 const char *env = getenv(CONFIG_DATA_ENVIRONMENT);
445 int ret = 0;
446 char *envw;
447 const char **argv = NULL;
448 int nr = 0, alloc = 0;
449 int i;
450 struct config_source source;
451
452 if (!env)
453 return 0;
454
455 memset(&source, 0, sizeof(source));
456 source.prev = cf;
457 source.origin_type = CONFIG_ORIGIN_CMDLINE;
458 cf = &source;
459
460 /* sq_dequote will write over it */
461 envw = xstrdup(env);
462
463 if (sq_dequote_to_argv(envw, &argv, &nr, &alloc) < 0) {
464 ret = error("bogus format in " CONFIG_DATA_ENVIRONMENT);
465 goto out;
466 }
467
468 for (i = 0; i < nr; i++) {
469 if (git_config_parse_parameter(argv[i], fn, data) < 0) {
470 ret = -1;
471 goto out;
472 }
473 }
474
475out:
476 free(argv);
477 free(envw);
478 cf = source.prev;
479 return ret;
480}

482static int get_next_char(void)
483{
484 int c = cf->do_fgetc(cf);
485
486 if (c == '\r') {
487 /* DOS like systems */
488 c = cf->do_fgetc(cf);
489 if (c != '\n') {
490 if (c != EOF)
491 cf->do_ungetc(c, cf);
492 c = '\r';
493 }
494 }
495 if (c == '\n')
496 cf->linenr++;
497 if (c == EOF) {
498 cf->eof = 1;
499 cf->linenr++;
500 c = '\n';
501 }
502 return c;
503}

505static char *parse_value(void)
506{
507 int quote = 0, comment = 0, space = 0;
508
509 strbuf_reset(&cf->value);
510 for (;;) {
511 int c = get_next_char();
512 if (c == '\n') {
513 if (quote) {
514 cf->linenr--;
515 return NULL;
516 }
517 return cf->value.buf;
518 }
519 if (comment)
520 continue;
521 if (isspace(c) && !quote) {
522 if (cf->value.len)
523 space++;
524 continue;
525 }
526 if (!quote) {
527 if (c == ';' || c == '#') {
528 comment = 1;
529 continue;
530 }
531 }
532 for (; space; space--)
533 strbuf_addch(&cf->value, ' ');
534 if (c == '\\') {
535 c = get_next_char();
536 switch (c) {
537 case '\n':
538 continue;
539 case 't':
540 c = '\t';
541 break;
542 case 'b':
543 c = '\b';
544 break;
545 case 'n':
546 c = '\n';
547 break;
548 /* Some characters escape as themselves */
549 case '\\': case '"':
550 break;
551 /* Reject unknown escape sequences */
552 default:
553 return NULL;
554 }
555 strbuf_addch(&cf->value, c);
556 continue;
557 }
558 if (c == '"') {
559 quote = 1-quote;
560 continue;
561 }
562 strbuf_addch(&cf->value, c);
563 }
564}

566static int get_value(config_fn_t fn, void *data, struct strbuf *name)
567{
568 int c;
569 char *value;
570 int ret;
571
572 /* Get the full name */
573 for (;;) {
574 c = get_next_char();
575 if (cf->eof)
576 break;
577 if (!iskeychar(c))
578 break;
579 strbuf_addch(name, tolower(c));
580 }
581
582 while (c == ' ' || c == '\t')
583 c = get_next_char();
584
585 value = NULL;
586 if (c != '\n') {
587 if (c != '=')
588 return -1;
589 value = parse_value();
590 if (!value)
591 return -1;
592 }
593 /*
594 * We already consumed the \n, but we need linenr to point to
595 * the line we just parsed during the call to fn to get
596 * accurate line number in error messages.
597 */
598 cf->linenr--;
599 ret = fn(name->buf, value, data);
600 if (ret >= 0)
601 cf->linenr++;
602 return ret;
603}

605static int get_extended_base_var(struct strbuf *name, int c)
606{
607 do {
608 if (c == '\n')
609 goto error_incomplete_line;
610 c = get_next_char();
611 } while (isspace(c));
612
613 /* We require the format to be '[base "extension"]' */
614 if (c != '"')
615 return -1;
616 strbuf_addch(name, '.');
617
618 for (;;) {
619 int c = get_next_char();
620 if (c == '\n')
621 goto error_incomplete_line;
622 if (c == '"')
623 break;
624 if (c == '\\') {
625 c = get_next_char();
626 if (c == '\n')
627 goto error_incomplete_line;
628 }
629 strbuf_addch(name, c);
630 }
631
632 /* Final ']' */
633 if (get_next_char() != ']')
634 return -1;
635 return 0;
636error_incomplete_line:
637 cf->linenr--;
638 return -1;
639}

641static int get_base_var(struct strbuf *name)
642{
643 for (;;) {
644 int c = get_next_char();
645 if (cf->eof)
646 return -1;
647 if (c == ']')
648 return 0;
649 if (isspace(c))
650 return get_extended_base_var(name, c);
651 if (!iskeychar(c) && c != '.')
652 return -1;
653 strbuf_addch(name, tolower(c));
654 }
655}

657static int git_parse_source(config_fn_t fn, void *data)
658{
659 int comment = 0;
660 int baselen = 0;
661 struct strbuf *var = &cf->var;
662 int error_return = 0;
663 char *error_msg = NULL;
664
665 /* U+FEFF Byte Order Mark in UTF8 */
666 const char *bomptr = utf8_bom;
667
668 for (;;) {
669 int c = get_next_char();
670 if (bomptr && *bomptr) {
671 /* We are at the file beginning; skip UTF8-encoded BOM
672 * if present. Sane editors won't put this in on their
673 * own, but e.g. Windows Notepad will do it happily. */
674 if (c == (*bomptr & 0377)) {
675 bomptr++;
676 continue;
677 } else {
678 /* Do not tolerate partial BOM. */
679 if (bomptr != utf8_bom)
680 break;
681 /* No BOM at file beginning. Cool. */
682 bomptr = NULL;
683 }
684 }
685 if (c == '\n') {
686 if (cf->eof)
687 return 0;
688 comment = 0;
689 continue;
690 }
691 if (comment || isspace(c))
692 continue;
693 if (c == '#' || c == ';') {
694 comment = 1;
695 continue;
696 }
697 if (c == '[') {
698 /* Reset prior to determining a new stem */
699 strbuf_reset(var);
700 if (get_base_var(var) < 0 || var->len < 1)
701 break;
702 strbuf_addch(var, '.');
703 baselen = var->len;
704 continue;
705 }
706 if (!isalpha(c))
707 break;
708 /*
709 * Truncate the var name back to the section header
710 * stem prior to grabbing the suffix part of the name
711 * and the value.
712 */
713 strbuf_setlen(var, baselen);
714 strbuf_addch(var, tolower(c));
715 if (get_value(fn, data, var) < 0)
716 break;
717 }
718
719 switch (cf->origin_type) {
720 case CONFIG_ORIGIN_BLOB:
721 error_msg = xstrfmt(_("bad config line %d in blob %s"),
722 cf->linenr, cf->name);
723 break;
724 case CONFIG_ORIGIN_FILE:
725 error_msg = xstrfmt(_("bad config line %d in file %s"),
726 cf->linenr, cf->name);
727 break;
728 case CONFIG_ORIGIN_STDIN:
729 error_msg = xstrfmt(_("bad config line %d in standard input"),
730 cf->linenr);
731 break;
732 case CONFIG_ORIGIN_SUBMODULE_BLOB:
733 error_msg = xstrfmt(_("bad config line %d in submodule-blob %s"),
734 cf->linenr, cf->name);
735 break;
736 case CONFIG_ORIGIN_CMDLINE:
737 error_msg = xstrfmt(_("bad config line %d in command line %s"),
738 cf->linenr, cf->name);
739 break;
740 default:
741 error_msg = xstrfmt(_("bad config line %d in %s"),
742 cf->linenr, cf->name);
743 }
744
745 if (cf->die_on_error)
746 die("%s", error_msg);
747 else
748 error_return = error("%s", error_msg);
749
750 free(error_msg);
751 return error_return;
752}

754static int parse_unit_factor(const char *end, uintmax_t *val)
755{
756 if (!*end)
757 return 1;
758 else if (!strcasecmp(end, "k")) {
759 *val *= 1024;
760 return 1;
761 }
762 else if (!strcasecmp(end, "m")) {
763 *val *= 1024 * 1024;
764 return 1;
765 }
766 else if (!strcasecmp(end, "g")) {
767 *val *= 1024 * 1024 * 1024;
768 return 1;
769 }
770 return 0;
771}

773static int git_parse_signed(const char *value, intmax_t *ret, intmax_t max)
774{
775 if (value && *value) {
776 char *end;
777 intmax_t val;
778 uintmax_t uval;
779 uintmax_t factor = 1;
780
781 errno = 0;
782 val = strtoimax(value, &end, 0);
783 if (errno == ERANGE)
784 return 0;
785 if (!parse_unit_factor(end, &factor)) {
786 errno = EINVAL;
787 return 0;
788 }
789 uval = labs(val);
790 uval *= factor;
791 if (uval > max || labs(val) > uval) {
792 errno = ERANGE;
793 return 0;
794 }
795 val *= factor;
796 *ret = val;
797 return 1;
798 }
799 errno = EINVAL;
800 return 0;
801}

803static int git_parse_unsigned(const char *value, uintmax_t *ret, uintmax_t max)
804{
805 if (value && *value) {
806 char *end;
807 uintmax_t val;
808 uintmax_t oldval;
809
810 errno = 0;
811 val = strtoumax(value, &end, 0);
812 if (errno == ERANGE)
813 return 0;
814 oldval = val;
815 if (!parse_unit_factor(end, &val)) {
816 errno = EINVAL;
817 return 0;
818 }
819 if (val > max || oldval > val) {
820 errno = ERANGE;
821 return 0;
822 }
823 *ret = val;
824 return 1;
825 }
826 errno = EINVAL;
827 return 0;

868{
869 const char * error_type = (errno == ERANGE)? _("out of range"):_("invalid unit");

871 if (!value)

873

875 die(_("bad numeric config value '%s' for '%s': %s"),

884 value, name, cf->name, error_type);
885 case CONFIG_ORIGIN_STDIN:

888 case CONFIG_ORIGIN_SUBMODULE_BLOB:
889 die(_("bad numeric config value '%s' for '%s' in submodule-blob %s: %s"),
890 value, name, cf->name, error_type);
891 case CONFIG_ORIGIN_CMDLINE:
892 die(_("bad numeric config value '%s' for '%s' in command line %s: %s"),
893 value, name, cf->name, error_type);
894 default:
895 die(_("bad numeric config value '%s' for '%s' in %s: %s"),

897 }

1387/*
1388 * All source specific fields in the union, die_on_error, name and the callbacks
1389 * fgetc, ungetc, ftell of top need to be initialized before calling
1390 * this function.
1391 */
1392static int do_config_from(struct config_source *top, config_fn_t fn, void *data)
1393{
1394 int ret;
1395
1396 /* push config-file parsing state stack */
1397 top->prev = cf;
1398 top->linenr = 1;
1399 top->eof = 0;
1400 strbuf_init(&top->value, 1024);
1401 strbuf_init(&top->var, 1024);
1402 cf = top;
1403
1404 ret = git_parse_source(fn, data);
1405
1406 /* pop config-file parsing state stack */
1407 strbuf_release(&top->value);
1408 strbuf_release(&top->var);
1409 cf = top->prev;
1410
1411 return ret;
1412}

1542static int do_git_config_sequence(const struct config_options *opts,
1543 config_fn_t fn, void *data)
1544{
1545 int ret = 0;
1546 char *xdg_config = xdg_config_home("config");
1547 char *user_config = expand_user_path("~/.gitconfig", 0);
1548 char *repo_config;
1549
1550 if (opts->commondir)
1551 repo_config = mkpathdup("%s/config", opts->commondir);
1552 else
1553 repo_config = NULL;
1554
1555 current_parsing_scope = CONFIG_SCOPE_SYSTEM;
1556 if (git_config_system() && !access_or_die(git_etc_gitconfig(), R_OK, 0))
1557 ret += git_config_from_file(fn, git_etc_gitconfig(),
1558 data);
1559
1560 current_parsing_scope = CONFIG_SCOPE_GLOBAL;
1561 if (xdg_config && !access_or_die(xdg_config, R_OK, ACCESS_EACCES_OK))
1562 ret += git_config_from_file(fn, xdg_config, data);
1563
1564 if (user_config && !access_or_die(user_config, R_OK, ACCESS_EACCES_OK))
1565 ret += git_config_from_file(fn, user_config, data);
1566
1567 current_parsing_scope = CONFIG_SCOPE_REPO;
1568 if (repo_config && !access_or_die(repo_config, R_OK, 0))
1569 ret += git_config_from_file(fn, repo_config, data);
1570
1571 current_parsing_scope = CONFIG_SCOPE_CMDLINE;
1572 if (git_config_from_parameters(fn, data) < 0)
1573 die(_("unable to parse command-line config"));
1574
1575 current_parsing_scope = CONFIG_SCOPE_UNKNOWN;
1576 free(xdg_config);
1577 free(user_config);
1578 free(repo_config);
1579 return ret;
1580}

1582int config_with_options(config_fn_t fn, void *data,
1583 struct git_config_source *config_source,
1584 const struct config_options *opts)
1585{
1586 struct config_include_data inc = CONFIG_INCLUDE_INIT;
1587
1588 if (opts->respect_includes) {
1589 inc.fn = fn;
1590 inc.data = data;
1591 inc.opts = opts;
1592 fn = git_config_include;
1593 data = &inc;
1594 }
1595

1597 * If we have a specific filename, use it. Otherwise, follow the
1598 * regular lookup sequence.
1599 */
1600 if (config_source && config_source->use_stdin)
1601 return git_config_from_stdin(fn, data);
1602 else if (config_source && config_source->file)
1603 return git_config_from_file(fn, config_source->file, data);
1604 else if (config_source && config_source->blob)
1605 return git_config_from_blob_ref(fn, config_source->blob, data);

2210 if (store.seen == 1 && store.multi_replace == 0) {
2211 warning(_("%s has multiple values"), key);
2212 }
2213
2214 ALLOC_GROW(store.offset, store.seen + 1,
2215 store.offset_alloc);
2216
2217 store.offset[store.seen] = cf->do_ftell(cf);
2218 store.seen++;
2219 }
2220 break;
2221 case SECTION_SEEN:
2222 /*
2223 * What we are looking for is in store.key (both
2224 * section and var), and its section part is baselen
2225 * long. We found key (again, both section and var).
2226 * We would want to know if this key is in the same
2227 * section as what we are looking for. We already
2228 * know we are in the same section as what should
2229 * hold store.key.
2230 */
2231 ep = strrchr(key, '.');
2232 section_len = ep - key;
2233
2234 if ((section_len != store.baselen) ||
2235 memcmp(key, store.key, section_len+1)) {
2236 store.state = SECTION_END_SEEN;
2237 break;
2238 }
2239
2240 /*
2241 * Do not increment matches: this is no match, but we
2242 * just made sure we are in the desired section.
2243 */
2244 ALLOC_GROW(store.offset, store.seen + 1,
2245 store.offset_alloc);
2246 store.offset[store.seen] = cf->do_ftell(cf);
2247 /* fallthru */
2248 case SECTION_END_SEEN:
2249 case START:
2250 if (matches(key, value)) {
2251 ALLOC_GROW(store.offset, store.seen + 1,
2252 store.offset_alloc);
2253 store.offset[store.seen] = cf->do_ftell(cf);
2254 store.state = KEY_SEEN;
2255 store.seen++;
2256 } else {
2257 if (strrchr(key, '.') - key == store.baselen &&
2258 !strncmp(key, store.key, store.baselen)) {
2259 store.state = SECTION_SEEN;
2260 ALLOC_GROW(store.offset,
2261 store.seen + 1,
2262 store.offset_alloc);
2263 store.offset[store.seen] = cf->do_ftell(cf);
2264 }
2265 }
2266 }
2267 return 0;
2268}

2326 length, key + store.baselen + 1, quote);

2333 case '\t':

2341 break;
2342 }
2343 strbuf_addf(&sb, "%s\n", quote);

2346 strbuf_release(&sb);

2399/*
2400 * If value==NULL, unset in (remove from) config,
2401 * if value_regex!=NULL, disregard key/value pairs where value does not match.
2402 * if value_regex==CONFIG_REGEX_NONE, do not match any existing values
2403 * (only add a new one)
2404 * if multi_replace==0, nothing, or only one matching key/value is replaced,
2405 * else all matching key/values (regardless how many) are removed,
2406 * before the new pair is written.
2407 *
2408 * Returns 0 on success.
2409 *
2410 * This function does this:
2411 *
2412 * - it locks the config file by creating ".git/config.lock"
2413 *
2414 * - it then parses the config using store_aux() as validator to find
2415 * the position on the key/value pair to replace. If it is to be unset,
2416 * it must be found exactly once.
2417 *
2418 * - the config file is mmap()ed and the part before the match (if any) is
2419 * written to the lock file, then the changed part and the rest.
2420 *
2421 * - the config file is removed and the lock file rename()d to it.
2422 *
2423 */
2424int git_config_set_multivar_in_file_gently(const char *config_filename,
2425 const char *key, const char *value,
2426 const char *value_regex,
2427 int multi_replace)
2428{
2429 int fd = -1, in_fd = -1;
2430 int ret;
2431 struct lock_file *lock = NULL;
2432 char *filename_buf = NULL;
2433 char *contents = NULL;
2434 size_t contents_sz;
2435
2436 /* parse-key returns negative; flip the sign to feed exit(3) */
2437 ret = 0 - git_config_parse_key(key, &store.key, &store.baselen);
2438 if (ret)
2439 goto out_free;
2440
2441 store.multi_replace = multi_replace;
2442
2443 if (!config_filename)
2444 config_filename = filename_buf = git_pathdup("config");
2445
2446 /*
2447 * The lock serves a purpose in addition to locking: the new
2448 * contents of .git/config will be written into it.
2449 */
2450 lock = xcalloc(1, sizeof(struct lock_file));
2451 fd = hold_lock_file_for_update(lock, config_filename, 0);
2452 if (fd < 0) {
2453 error_errno("could not lock config file %s", config_filename);
2454 free(store.key);
2455 ret = CONFIG_NO_LOCK;
2456 goto out_free;
2457 }
2458
2459 /*
2460 * If .git/config does not exist yet, write a minimal version.
2461 */
2462 in_fd = open(config_filename, O_RDONLY);
2463 if ( in_fd < 0 ) {
2464 free(store.key);
2465
2466 if ( ENOENT != errno ) {
2467 error_errno("opening %s", config_filename);
2468 ret = CONFIG_INVALID_FILE; /* same as "invalid config file" */
2469 goto out_free;
2470 }
2471 /* if nothing to unset, error out */
2472 if (value == NULL) {
2473 ret = CONFIG_NOTHING_SET;
2474 goto out_free;
2475 }
2476
2477 store.key = (char *)key;
2478 if (!store_write_section(fd, key) ||
2479 !store_write_pair(fd, key, value))
2480 goto write_err_out;
2481 } else {
2482 struct stat st;
2483 size_t copy_begin, copy_end;
2484 int i, new_line = 0;
2485
2486 if (value_regex == NULL)
2487 store.value_regex = NULL;
2488 else if (value_regex == CONFIG_REGEX_NONE)
2489 store.value_regex = CONFIG_REGEX_NONE;
2490 else {
2491 if (value_regex[0] == '!') {
2492 store.do_not_match = 1;
2493 value_regex++;
2494 } else
2495 store.do_not_match = 0;
2496
2497 store.value_regex = (regex_t*)xmalloc(sizeof(regex_t));
2498 if (regcomp(store.value_regex, value_regex,
2499 REG_EXTENDED)) {
2500 error("invalid pattern: %s", value_regex);
2501 free(store.value_regex);
2502 ret = CONFIG_INVALID_PATTERN;
2503 goto out_free;
2504 }
2505 }
2506
2507 ALLOC_GROW(store.offset, 1, store.offset_alloc);
2508 store.offset[0] = 0;
2509 store.state = START;
2510 store.seen = 0;
2511
2512 /*
2513 * After this, store.offset will contain the *end* offset
2514 * of the last match, or remain at 0 if no match was found.
2515 * As a side effect, we make sure to transform only a valid
2516 * existing config file.
2517 */
2518 if (git_config_from_file(store_aux, config_filename, NULL)) {
2519 error("invalid config file %s", config_filename);
2520 free(store.key);
2521 if (store.value_regex != NULL &&
2522 store.value_regex != CONFIG_REGEX_NONE) {
2523 regfree(store.value_regex);
2524 free(store.value_regex);
2525 }
2526 ret = CONFIG_INVALID_FILE;
2527 goto out_free;
2528 }
2529
2530 free(store.key);
2531 if (store.value_regex != NULL &&
2532 store.value_regex != CONFIG_REGEX_NONE) {
2533 regfree(store.value_regex);
2534 free(store.value_regex);
2535 }
2536
2537 /* if nothing to unset, or too many matches, error out */
2538 if ((store.seen == 0 && value == NULL) ||
2539 (store.seen > 1 && multi_replace == 0)) {
2540 ret = CONFIG_NOTHING_SET;
2541 goto out_free;
2542 }
2543
2544 if (fstat(in_fd, &st) == -1) {
2545 error_errno(_("fstat on %s failed"), config_filename);
2546 ret = CONFIG_INVALID_FILE;
2547 goto out_free;
2548 }
2549
2550 contents_sz = xsize_t(st.st_size);
2551 contents = xmmap_gently(NULL, contents_sz, PROT_READ,
2552 MAP_PRIVATE, in_fd, 0);
2553 if (contents == MAP_FAILED) {
2554 if (errno == ENODEV && S_ISDIR(st.st_mode))
2555 errno = EISDIR;
2556 error_errno("unable to mmap '%s'", config_filename);
2557 ret = CONFIG_INVALID_FILE;
2558 contents = NULL;
2559 goto out_free;
2560 }
2561 close(in_fd);
2562 in_fd = -1;
2563
2564 if (chmod(get_lock_file_path(lock), st.st_mode & 07777) < 0) {
2565 error_errno("chmod on %s failed", get_lock_file_path(lock));
2566 ret = CONFIG_NO_WRITE;
2567 goto out_free;
2568 }
2569
2570 if (store.seen == 0)
2571 store.seen = 1;
2572
2573 for (i = 0, copy_begin = 0; i < store.seen; i++) {
2574 if (store.offset[i] == 0) {
2575 store.offset[i] = copy_end = contents_sz;
2576 } else if (store.state != KEY_SEEN) {
2577 copy_end = store.offset[i];
2578 } else
2579 copy_end = find_beginning_of_line(
2580 contents, contents_sz,
2581 store.offset[i]-2, &new_line);
2582
2583 if (copy_end > 0 && contents[copy_end-1] != '\n')
2584 new_line = 1;
2585
2586 /* write the first part of the config */
2587 if (copy_end > copy_begin) {
2588 if (write_in_full(fd, contents + copy_begin,
2589 copy_end - copy_begin) <
2590 copy_end - copy_begin)
2591 goto write_err_out;
2592 if (new_line &&
2593 write_str_in_full(fd, "\n") != 1)
2594 goto write_err_out;
2595 }
2596 copy_begin = store.offset[i];
2597 }
2598
2599 /* write the pair (value == NULL means unset) */
2600 if (value != NULL) {
2601 if (store.state == START) {
2602 if (!store_write_section(fd, key))
2603 goto write_err_out;
2604 }
2605 if (!store_write_pair(fd, key, value))
2606 goto write_err_out;
2607 }
2608
2609 /* write the rest of the config */
2610 if (copy_begin < contents_sz)
2611 if (write_in_full(fd, contents + copy_begin,
2612 contents_sz - copy_begin) <
2613 contents_sz - copy_begin)
2614 goto write_err_out;
2615
2616 munmap(contents, contents_sz);
2617 contents = NULL;
2618 }
2619
2620 if (commit_lock_file(lock) < 0) {
2621 error_errno("could not write config file %s", config_filename);
2622 ret = CONFIG_NO_WRITE;
2623 lock = NULL;
2624 goto out_free;
2625 }
2626
2627 /*
2628 * lock is committed, so don't try to roll it back below.
2629 * NOTE: Since lockfile.c keeps a linked list of all created
2630 * lock_file structures, it isn't safe to free(lock). It's
2631 * better to just leave it hanging around.
2632 */
2633 lock = NULL;
2634 ret = 0;
2635
2636 /* Invalidate the config cache */
2637 git_config_clear();
2638
2639out_free:
2640 if (lock)
2641 rollback_lock_file(lock);
2642 free(filename_buf);
2643 if (contents)
2644 munmap(contents, contents_sz);
2645 if (in_fd >= 0)
2646 close(in_fd);
2647 return ret;
2648
2649write_err_out:
2650 ret = write_error(get_lock_file_path(lock));
2651 goto out_free;
2652
2653}

2682static int section_name_match (const char *buf, const char *name)
2683{
2684 int i = 0, j = 0, dot = 0;
2685 if (buf[i] != '[')
2686 return 0;
2687 for (i = 1; buf[i] && buf[i] != ']'; i++) {
2688 if (!dot && isspace(buf[i])) {
2689 dot = 1;
2690 if (name[j++] != '.')
2691 break;
2692 for (i++; isspace(buf[i]); i++)
2693 ; /* do nothing */
2694 if (buf[i] != '"')
2695 break;
2696 continue;
2697 }
2698 if (buf[i] == '\\' && dot)
2699 i++;
2700 else if (buf[i] == '"' && dot) {
2701 for (i++; isspace(buf[i]); i++)
2702 ; /* do_nothing */
2703 break;
2704 }
2705 if (buf[i] != name[j++])
2706 break;
2707 }
2708 if (buf[i] == ']' && name[j] == 0) {
2709 /*
2710 * We match, now just find the right length offset by
2711 * gobbling up any whitespace after it, as well
2712 */
2713 i++;
2714 for (; buf[i] && isspace(buf[i]); i++)
2715 ; /* do nothing */
2716 return i;
2717 }
2718 return 0;
2719}

2721static int section_name_is_ok(const char *name)
2722{
2723 /* Empty section names are bogus. */
2724 if (!*name)
2725 return 0;
2726
2727 /*
2728 * Before a dot, we must be alphanumeric or dash. After the first dot,
2729 * anything goes, so we can stop checking.
2730 */
2731 for (; *name && *name != '.'; name++)
2732 if (*name != '-' && !isalnum(*name))
2733 return 0;
2734 return 1;
2735}

2737/* if new_name == NULL, the section is removed instead */
2738int git_config_rename_section_in_file(const char *config_filename,
2739 const char *old_name, const char *new_name)
2740{
2741 int ret = 0, remove = 0;
2742 char *filename_buf = NULL;
2743 struct lock_file *lock;
2744 int out_fd;
2745 char buf[1024];
2746 FILE *config_file = NULL;
2747 struct stat st;
2748
2749 if (new_name && !section_name_is_ok(new_name)) {
2750 ret = error("invalid section name: %s", new_name);
2751 goto out_no_rollback;
2752 }
2753
2754 if (!config_filename)
2755 config_filename = filename_buf = git_pathdup("config");
2756
2757 lock = xcalloc(1, sizeof(struct lock_file));
2758 out_fd = hold_lock_file_for_update(lock, config_filename, 0);
2759 if (out_fd < 0) {
2760 ret = error("could not lock config file %s", config_filename);
2761 goto out;
2762 }
2763
2764 if (!(config_file = fopen(config_filename, "rb"))) {
2765 ret = warn_on_fopen_errors(config_filename);
2766 if (ret)
2767 goto out;
2768 /* no config file means nothing to rename, no error */
2769 goto commit_and_out;
2770 }
2771
2772 if (fstat(fileno(config_file), &st) == -1) {
2773 ret = error_errno(_("fstat on %s failed"), config_filename);
2774 goto out;
2775 }
2776
2777 if (chmod(get_lock_file_path(lock), st.st_mode & 07777) < 0) {
2778 ret = error_errno("chmod on %s failed",
2779 get_lock_file_path(lock));
2780 goto out;
2781 }
2782
2783 while (fgets(buf, sizeof(buf), config_file)) {
2784 int i;
2785 int length;
2786 char *output = buf;
2787 for (i = 0; buf[i] && isspace(buf[i]); i++)
2788 ; /* do nothing */
2789 if (buf[i] == '[') {
2790 /* it's a section */
2791 int offset = section_name_match(&buf[i], old_name);
2792 if (offset > 0) {
2793 ret++;
2794 if (new_name == NULL) {
2795 remove = 1;
2796 continue;
2797 }
2798 store.baselen = strlen(new_name);
2799 if (!store_write_section(out_fd, new_name)) {
2800 ret = write_error(get_lock_file_path(lock));
2801 goto out;
2802 }
2803 /*
2804 * We wrote out the new section, with
2805 * a newline, now skip the old
2806 * section's length
2807 */
2808 output += offset + i;
2809 if (strlen(output) > 0) {
2810 /*
2811 * More content means there's
2812 * a declaration to put on the
2813 * next line; indent with a
2814 * tab
2815 */
2816 output -= 1;
2817 output[0] = '\t';
2818 }
2819 }
2820 remove = 0;
2821 }
2822 if (remove)
2823 continue;
2824 length = strlen(output);
2825 if (write_in_full(out_fd, output, length) != length) {
2826 ret = write_error(get_lock_file_path(lock));
2827 goto out;
2828 }
2829 }
2830 fclose(config_file);
2831 config_file = NULL;
2832commit_and_out:
2833 if (commit_lock_file(lock) < 0)
2834 ret = error_errno("could not write config file %s",
2835 config_filename);
2836out:
2837 if (config_file)
2838 fclose(config_file);
2839 rollback_lock_file(lock);
2840out_no_rollback:
2841 free(filename_buf);
2842 return ret;
2843}

2862 const char **subsection, int *subsection_len,
2863 const char **key)
2864{

2868 if (!skip_prefix(var, section, &var) || *var != '.')
2869 return -1;

2872 * Find the key; we don't know yet if we have a subsection, but we must
2873 * parse backwards from the end, since the subsection may have dots in
2874 * it, too.
2875 */
2876 dot = strrchr(var, '.');

2880 if (dot == var) {

2882 *subsection = NULL;

2886 else {
2887 if (!subsection)
2888 return -1;
2889 *subsection = var + 1;
2890 *subsection_len = dot - *subsection;
2891 }

2893 return 0;
